35   /** 
36    * A nominal to output code filter; it choose to which subset (0 or 1) each attribute value 
37    * drop uniformly and independently at random. 
38    * 
39    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
40    * @version $Revision: 1.0 $ 
41    * @see oaidtb.filters.AbstractNominalToOCFilter 
42    * @see oaidtb.filters.AbstractNominalToRandomOCFilter 
43    */ 
44   public class NominalToRandomOCFilter extends AbstractNominalToRandomOCFilter{ 
45    
46     /** 
47      * Generate a new partition at specified position even though it has already been generated 
48      * (it will be overwritten). 
49      * 
50      * It doesn't use the instances information (so it can be null). 
51      * 
52      * @param instances Perhaps a subclasss will use some information of them. 
53      * @param partitionNumber The number of the partition to generate. 
54      * 
55      * @throws Exception If partitionNumber is incorrect or another error occurs. 
56      */ 
57     public void newPartition(Instances instances, int partitionNumber) throws Exception{ 
58    
59       if (partitionNumber > m_NumGeneratedPartitions || partitionNumber < 0) 
60         throw new Exception("Partition index must be between 0 and " 
61                             + String.valueOf(m_NumGeneratedPartitions + 1)); 
62    
63       for (int i = 0; i < m_OutputCode.length; i++) 
64         if (m_Random.nextBoolean()) 
65           m_OutputCode[i].set(partitionNumber); 
66         else 
67           m_OutputCode[i].clear(partitionNumber); 
68    
69       if (partitionNumber == m_NumGeneratedPartitions) 
70         m_NumGeneratedPartitions++; 
71     } 
72    
73     /** 
74      * Generate a new partition incrementing the number of partitions generated, 
75      * and so incrementing by 1 the length of all code words. It's equivalent 
76      * to newPartition(instances, m_NumGeneratedPartitions), but faster. 
77      * 
78      * @param instances Perhaps a subclasss will use some information of them. 
79      */ 
80     public void newPartition(Instances instances){ 
81    
82       for (int i = 0; i < m_OutputCode.length; i++) 
83         if (m_Random.nextBoolean()) 
84           m_OutputCode[i].set(m_NumGeneratedPartitions); 
85    
86       m_NumGeneratedPartitions++; 
87     } 
88    
89     /** 
90      * Sets the format of the input instances. 
91      * 
92      * @param instanceInfo an Instances object containing the input instance structure. 
93      * 
94      * @return true if the outputFormat may be collected immediately 
95      * @exception Exception if the format couldn't be set successfully 
96      */ 
97     public boolean setInputFormat(Instances instanceInfo) throws Exception{ 
98    
99       super.setInputFormat(instanceInfo); 
100   
101      m_OutputCode = new BitSet[instanceInfo.numDistinctValues(m_ProcessedAttribute)]; 
102   
103      for (int i = 0; i < m_OutputCode.length; i++) 
104        m_OutputCode[i] = new BitSet(); 
105   
106      if (m_Random == null) 
107        m_Random = new Random(m_Seed); 
108   
109      newPartition(null); 
110   
111      return true; 
112    }
