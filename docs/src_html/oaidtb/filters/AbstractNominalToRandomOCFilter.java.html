23   package oaidtb.filters; 
24    
25   import weka.core.Option; 
26   import weka.core.Utils; 
27   import weka.core.Instances; 
28    
29   import java.util.Enumeration; 
30   import java.util.Random; 
31   import java.util.Vector; 
32    
33   /** 
34    * Abstract class defining the structure and common code to "nominal to output code" 
35    * filters which exhibit randomness in its behaviour (ie. they use a random number generator 
36    * when constructing new partitions). 
37    * 
38    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
39    * @version $Revision: 1.0 $ 
40    * @see oaidtb.filters.AbstractNominalToOCFilter 
41    */ 
42   public abstract class AbstractNominalToRandomOCFilter extends AbstractNominalToOCFilter{ 
43    
44     /** The random number generator used to create each partition. */ 
45     protected Random m_Random = new Random(); 
46    
47     /** The m_Random's seed, unless it's a clock based seed. */ 
48     protected long m_Seed = 0; 
49    
50     /** 
51      * Get the random number generator used to get the random behavior of the filter. 
52      * 
53      * @return The m_Random number generator. 
54      */ 
55     public Random getRandom(){ 
56       return m_Random; 
57     } 
58    
59     /** 
60      * Set the random number generator to be used as the basis for partition generation. 
61      * 
62      * @param random The random number generator. 
63      */ 
64     public void setRandom(Random random){ 
65       m_Random = random; 
66     } 
67    
68     /** 
69      * Set the seed used by the random number generator. 
70      * 
71      * @param seed The seed. 
72      */ 
73     public void setSeed(long seed){ 
74       m_Seed = seed; 
75       m_Random = new Random(m_Seed); //Use m_Random.setResampleSeed 
76     } 
77    
78     /** 
79      * Sets the format of the input instances. 
80      * 
81      * @param instanceInfo an Instances object containing the input instance structure. 
82      * 
83      * @return true if the outputFormat may be collected immediately 
84      * @exception Exception if the format couldn't be set successfully 
85      */ 
86     public boolean setInputFormat(Instances instanceInfo) throws Exception{ 
87    
88       //It must give the same results when running twice over the same dataset 
89       if(super.setInputFormat(instanceInfo)){ 
90         m_Random = new Random(m_Seed); 
91         return true; 
92       } 
93    
94       return false; 
95     } 
96    
97     /** 
98      * Get the seed used by the random number generator. 
99      * 
100     * @return The seed. 
101     */ 
102    public long getSeed(){ 
103      return m_Seed; 
104    } 
105   
106    /** 
107     * Parses a given list of options controlling the behaviour of this object. 
108     * Valid options are:<p> 
109     * 
110     * -R seed<br> 
111     * Random number seed for the filter. Default is 0. If you call it with a -R 0, it will 
112     * use a clock based seed. 
113     * <p> 
114     * 
115     * @param options the list of options as an array of strings 
116     * @exception Exception if an option is not supported 
117     */ 
118    public void setOptions(String[] options) throws Exception{ 
119   
120      super.setOptions(options); 
121   
122      String seedString = Utils.getOption('R', options); 
123      if (seedString.length() != 0){ 
124        long seed = Long.parseLong(seedString); 
125        if (seed != 0) 
126          setSeed(seed); 
127        else 
128          setRandom(new Random()); //Clock based. 
129      } 
130      else 
131        setSeed(0); 
132    } 
133   
134    /** 
135     * Gets the current settings of the filter. 
136     * 
137     * @return an array of strings suitable for passing to setOptions 
138     */ 
139    public String[] getOptions(){ 
140   
141      String[] otherOptions = super.getOptions(); 
142   
143      String[] options = new String[otherOptions.length + 2]; 
144   
145      int current = 0; 
146   
147      options[current++] = "-R"; 
148      options[current++] = "" + getSeed(); 
149   
150      System.arraycopy(otherOptions, 0, 
151                       options, current, 
152                       otherOptions.length); 
153   
154      current += otherOptions.length; 
155      while (current < options.length){ 
156        options[current++] = ""; 
157      } 
158   
159      return options; 
160    } 
161   
162    /** 
163     * Returns an enumeration describing the available options 
164     * 
165     * @return an enumeration of all the available options 
166     */ 
167    public Enumeration listOptions(){ 
168   
169      Vector newVector = new Vector(2); 
170   
171      newVector.addElement(new Option( 
172        "\tRandom number seed for the filter.\n", 
173        "R", 1, "-R <class name>")); 
174   
175      Enumeration enum = super.listOptions(); 
176      while (enum.hasMoreElements()){ 
177        newVector.addElement(enum.nextElement()); 
178      } 
179   
180      return newVector.elements(); 
181    } 
182  }
