23   package oaidtb.boosters; 
24    
25   import weka.core.Instance; 
26    
27   /** 
28    * Interface for boosters which can handle binary nominal problems and can 
29    * be extended to the multiclass case using a "one-per-class" approach. 
30    * 
31    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
32    * @version $Revision: 1.0 $ 
33    */ 
34   public interface MulticlassExtensibleBooster{ 
35    
36     /** 
37      * Must the booster use its own train data copy? 
38      * 
39      * @param b True if the booster must use its own train data copy. 
40      */ 
41     void setUseOwnTrainData(boolean b); 
42    
43     /** 
44      * Return the class predicted (<0 == class 0 and >0 == class 1) and the 
45      * confidence of this prediction (its absolute value). 
46      * 
47      * @param instance The instance to be classified 
48      * 
49      * @return The class predicted and its confidence. 
50      * 
51      * @throws Exception if the instance can´t be classified succesfully. 
52      */ 
53     double confidenceAndSign(Instance instance) throws Exception; 
54    
55     /** 
56      * Return the class predicted (<0 == class 0 and >0 == class 1) and the 
57      * confidence of this prediction (its absolute value) of the classifier indicated. 
58      * Similar to Booster.distributionForInstance(instance,classifierIndex). 
59      * 
60      * @param instance The instance to be classified 
61      * @param classifierIndex The classifier's index 
62      * 
63      * @return The class predicted and its confidence. 
64      * 
65      * @throws Exception if an error happens. 
66      */ 
67     double confidenceAndSign(Instance instance, int classifierIndex) throws Exception; 
68   }
