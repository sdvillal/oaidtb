23   package oaidtb.boosters; 
24    
25   import weka.classifiers.Evaluation; 
26    
27   /** 
28    * A modification of AdaBoostM1 which works better with weak base classifiers; for more information see:<p> 
29    * 
30    * [1] - Günther Eibl & Karl Peter Pfeiffer 
31    * <i>How to make AdaBoost.M1 work for weak base classifiers by changing only one line of the code.</i> 
32    * </p> 
33    * 
34    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
35    * @version $Revision: 1.0 $ 
36    */ 
37   public class AdaBoostM1W extends oaidtb.boosters.AdaBoostM1{ 
38    
39     /** 1 / m_NumClasses; avoid operation overload when calculating the guessing error upper bound */ 
40     private double m_NumClassesInverse; 
41    
42     /** Avoid operation overload when calculating the guessing error upper bound */ 
43     private double m_GuesserCalculationConstant; 
44    
45     /** 
46      * Calculate the classifier's vote weight in the final combined hypothesis based on its error. 
47      * 
48      * @param error The training error of the base classifier. 
49      * 
50      * @return The classifier's vote weight in the final combined hypothesis 
51      */ 
52     protected double calculateBeta(double error){ 
53       return Math.log(((m_NumClasses - 1) * (1 - error) + NO_DIVISION_BY_ZERO) / (error + NO_DIVISION_BY_ZERO)); 
54     } 
55    
56     /** 
57      * The default error value for which the algorithm will stop to iterate following the 
58      * customizable criterion of the number of (consecutive or not) retries the base classifier reaches a error 
59      * worse than this value. 
60      * 
61      * @return  The ad-hoc stopping criterion proposed in [1] 
62      */ 
63     protected double defaultTooBigErrorValue(){ 
64       if(m_NumClasses != 0) 
65         return 1 - 1 / m_NumClasses; 
66       return 0; 
67     } 
68    
69     /** Initialize the guessing error of the training set upper bound. */ 
70     protected void initializeErrorUpperBound(){ 
71       if (m_CalculateErrorUpperBound){ 
72         m_NumClassesInverse = 1.0 / m_NumClasses; 
73         m_GuesserCalculationConstant = 1.0 / (Math.pow(1.0 - m_NumClassesInverse, 1.0 - m_NumClassesInverse) * 
74           Math.pow(m_NumClassesInverse, m_NumClassesInverse)); 
75    
76         if (m_Debug) 
77           System.err.println("Calculating the guessing error upper bound."); 
78       } 
79       m_ErrorUpperBound = 1.0; 
80     } 
81    
82     /** 
83      * Get the guessing error upper bound. See [1] for more info. 
84      * 
85      * @return The guessing error upper bound or -1 if no iterations have been done. 
86      */ 
87     public double getErrorUpperBound(){ 
88       if (m_NumIterations > 0){ 
89         if (m_CalculateErrorUpperBound) 
90           return Double.isNaN(m_ErrorUpperBound) ? 1 : m_ErrorUpperBound; 
91         return 1; 
92       } 
93       return -1; 
94     } 
95    
96     /** 
97      * Update the theoretical guessing error upper bound.<PRE> 
98      * 
99      *  - Note that if the error of the base classifier is greater than 1/m_NumClasses the bound is undetermined, 
100     *    and so its calculation will be aborted. 
101     * </PRE> 
102     * 
103     * @param error The current base classifier's error 
104     */ 
105    protected void updateErrorUpperBound(double error){ 
106      if (m_CalculateErrorUpperBound && !Double.isNaN(m_ErrorUpperBound)){ 
107        if (error > (1.0 - m_NumClassesInverse)){ 
108          m_ErrorUpperBound = Double.NaN; 
109          if (m_Debug) 
110            System.err.println("Can't bound the guessing error upper boud: base classifier's error is greater than 1/2."); 
111        } 
112        else{ 
113          m_ErrorUpperBound *= (Math.pow(error, 1 - m_NumClassesInverse) * Math.pow(1 - error, m_NumClassesInverse) 
114            * m_GuesserCalculationConstant); 
115          if (m_Debug) 
116            System.err.println("\tGesser <= " + m_ErrorUpperBound); 
117        } 
118      } 
119    } 
120   
121    /** @return The error upper bound string (by example, "Training error upper bound.") */ 
122    public String getErrorUpperBoundName(){ 
123      return "Guessing error of the training set upper bound"; 
124    } 
125   
126    public String calculateErrorUpperBoundTipText(){ 
127   
128      return "Calculate or not the " +getErrorUpperBoundName() +" (only if no iteration have been done yet).\n" + 
129        "See Eibl & Pfeiffer paper."; 
130    } 
131

133     * Main method for testing this class. 
134     * 
135     * @param argv the options 
136     */ 
137    public static void main(String[] argv){ 
138   
139      try{ 
140        System.out.println(Evaluation.evaluateModel(new AdaBoostM1W(), argv)); 
141      } 
142      catch (Exception e){ 
143        System.err.println(e.getMessage()); 
144      } 
145    }
