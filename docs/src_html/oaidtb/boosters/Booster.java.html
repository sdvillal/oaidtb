148    /** 
149     * Parses a given list of options. Valid options are:<p> 
150     * 
151     * -D <br> 
152     * Turn on debugging output.<p> 
153     * 
154     * -W classname <br> 
155     * Specify the full class name of a classifier as the basis for 
156     * boosting (required).<p> 
157     * 
158     * -I num <br> 
159     * Set the number of boost iterations on first call to BuildClassifier (default 10). <p> 
160     * 
161     * -Q <br> 
162     * Use resampling instead of reweighting.<p> 
163     * 
164     * -S seed <br> 
165     * Random number seed for resampling (default 1).<p> 
166     * 
167     * -N normFactor <br> 
168     * Normalization policy and factor; options: <PRE> 
169     * 
170     *   - If <0  --> Sum of weights at each iteration will not be normalized 
171     *   - If ==0 --> Sum of weights at each iteration will be the original sum of weights in the training set 
172     *   - If >0  --> Sum of weights at each iteration will be normFactor 
173     * </PRE> 
174     * 
175     * Options after -- are passed to the designated classifier.<p> 
176     * 
177     * @param options the list of options as an array of strings 
178     * @exception Exception if an option is not supported 
179     */ 
180    public void setOptions(String[] options) throws Exception{ 
181   
182      setDebug(Utils.getFlag('D', options)); 
183   
184      String boostIterations = Utils.getOption('I', options); 
185      if (boostIterations.length() != 0){ 
186        setInitialIterations(Integer.parseInt(boostIterations)); 
187      } 
188      else{ 
189        setInitialIterations(10); 
190      } 
191   
192      setUseResampling(Utils.getFlag('Q', options)); 
193   
194      String seedString = Utils.getOption('S', options); 
195      if (seedString.length() != 0){ 
196        setResampleSeed(Integer.parseInt(seedString)); 
197      } 
198      else{ 
199        setResampleSeed(1); 
200      } 
201   
202      String classifierName = Utils.getOption('W', options); 
203      if (classifierName.length() == 0){ 
204        throw new Exception("A classifier must be specified with" 
205                            + " the -W option."); 
206      } 
207      setClassifier(Classifier.forName(classifierName, 
208                                       Utils.partitionOptions(options))); 
209   
210      String normFactor = Utils.getOption('N', options); 
211      if (normFactor.length() != 0) 
212        setNormFactor(Double.parseDouble(normFactor)); 
213    }

215    /** 
216     * Gets the current settings of the Classifier. 
217     * 
218     * @return an array of strings suitable for passing to setOptions 
219     */ 
220    public String[] getOptions(){ 
221   
222      String[] classifierOptions = new String[0]; 
223      if ((m_Classifier != null) && (m_Classifier instanceof OptionHandler)) 
224        classifierOptions = ((OptionHandler) m_Classifier).getOptions(); 
225   
226   
227      String[] options = new String[classifierOptions.length + 12]; 
228      int current = 0; 
229      if (getDebug()){ 
230        options[current++] = "-D"; 
231      } 
232      if (getUseResampling()){ 
233        options[current++] = "-Q"; 
234      } 
235      options[current++] = "-I"; 
236      options[current++] = "" + getInitialIterations(); 
237      options[current++] = "-S"; 
238      options[current++] = "" + getResampleSeed(); 
239      options[current++] = "-N"; 
240      if (m_NormFactorUsed == SUM_OF_WEIGHTS_NORM_FACTOR) 
241        options[current++] = "" + 0; 
242      else if (m_NormFactorUsed == NOT_NORMALIZE) 
243        options[current++] = "" + -1; 
244      else 
245        options[current++] = "" + getNormFactor(); 
246   
247   
248      if (getClassifier() != null){ 
249        options[current++] = "-W"; 
250        options[current++] = getClassifier().getClass().getName(); 
251      } 
252   
253      options[current++] = "--"; 
254   
255      System.arraycopy(classifierOptions, 0, options, current, 
256                       classifierOptions.length); 
257      current += classifierOptions.length; 
258      while (current < options.length) 
259        options[current++] = ""; 
260   
261      return options; 
262    }

404    /** 
405     * Set the normalization factor (and policy) which will be used; options: <PRE> 
406     * 
407     *   - If <0  --> Sum of weights at each iteration will not be normalized 
408     *   - If ==0 --> Sum of weights at each iteration will be the original sum of weights in the training set 
409     *   - If >0  --> Sum of weights at each iteration will be normFactor 
410     * 
411     * </PRE> 
412     * 
413     * @param normFactor Normalization policy indicator & sum of weights at each iteration if > 0 
414     */ 
415    public void setNormFactor(double normFactor){ 
416      if (normFactor < 0){ 
417        m_NormFactor = -1; 
418        m_NormFactorUsed = NOT_NORMALIZE; 
419      } 
420      else if (normFactor == 0) 
421      //m_NormFactor will be calculated every time buildClassifier is called with initialNormFactor method. 
422        m_NormFactorUsed = SUM_OF_WEIGHTS_NORM_FACTOR; 
423      else{ 
424        m_NormFactor = normFactor; 
425        m_NormFactorUsed = CUSTOM_NORM_FACTOR; 
426      } 
427    }

438    /** Initializes the normalization factor if it must be the initial sum of weights. */ 
439    protected void initializeNormFactor(){ 
440      if (m_NormFactorUsed == SUM_OF_WEIGHTS_NORM_FACTOR) 
441        m_NormFactor = m_TrainData.sumOfWeights(); 
442   
443      if (m_Debug) 
444        switch (m_NormFactorUsed){ 
445          case SUM_OF_WEIGHTS_NORM_FACTOR: 
446            System.err.println("Weights sum at each iteration will be the original weights sum: " + m_NormFactor); 
447            break; 
448          case CUSTOM_NORM_FACTOR: 
449            System.err.println("Weights sum at each iteration will be: " + m_NormFactor); 
450            break; 
451          case NOT_NORMALIZE: 
452            System.err.println("Weights sum will not be normalized in any way."); 
453          default : 
454        } 
455    }

457    /** Initializes the weights if they must sum a custom number. */ 
458    protected void initializeWeights(){ 
459      if (m_NormFactorUsed == CUSTOM_NORM_FACTOR) 
460        normalizeWeights(m_TrainData.sumOfWeights()); 
461    }

463    /** 
464     * Normalize the weight sum of the train data following the predefined directives. 
465     * 
466     * @param weightSum The actual sum of weights of the train data. 
467     */ 
468    protected final void normalizeWeights(double weightSum){ 
469      if (m_NormFactorUsed != Booster.NOT_NORMALIZE){ 
470   
471        weightSum /= m_NormFactor; 
472        weightSum += Booster.NO_DIVISION_BY_ZERO; 
473   
474        //Normalize (it isn´t usually necessary for properly work of the classifiers). 
475        for (int i = 0; i < m_NumInstances; i++){ 
476          Instance instance = m_TrainData.instance(i); 
477          instance.setWeight((instance.weight() + Booster.NO_DIVISION_BY_ZERO) / weightSum); 
478        } 
479      } 
480    }

603     * Eliminate the last numIterations performed from the final combined hypothesis.

605     * <li>Note: To free the memory allocated, this method must be overriden 
606     * <li>Note: The weights distribution of the training instances will not return back in the time.

613        throw new Exception("There aren't so many iterations."); 
614   
615      m_NumIterations -= numIterations;

618    /** 
619     * Classify the given instance using only the first numIterationsToUse base classifiers 
620     * 
621     * @param instance The instance to be classified 
622     * @param numIterationsToUse Number of base classifiers to use 
623     * @return The instance clasification distribution 
624     * @throws Exception If an error happens 
625     */ 
626    public double[] distributionForInstance(Instance instance, int numIterationsToUse) throws Exception{ 
627   
628      if (numIterationsToUse > m_NumIterations || numIterationsToUse < 0) 
629        throw new Exception("Invaled numIterations parameter."); 
630   
631      int swapTmp = m_NumIterations; 
632      m_NumIterations = numIterationsToUse; 
633   
634      try{ 
635        double[] distributionForInstance = distributionForInstance(instance); 
636        m_NumIterations = swapTmp; 
637        return distributionForInstance; 
638      } 
639      catch (Exception e){ 
640        m_NumIterations = swapTmp; 
641        throw e; 
642      } 
643    }

657   
658      int swapTmp = m_NumIterations; 
659      m_NumIterations = numIterationsToUse;
