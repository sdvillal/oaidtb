30   /** 
31    * One of the variants of CSB proposed in:<p> 
32    * 
33    * <a href="http://www.gscit.monash.edu.au/~kmting/Papers/icml00.ps"> 
34    * Ting, K.M., <i>A Comparative Study of Cost-Sensitive Boosting Algorithms.</i>. 
35    * Proceedings of The Seventeenth International Conference on Machine Learning. pp. 983-990. 
36    * San Francisco: Morgan Kaufmann. Stanford University, June 29 - July 2, 2000. 
37    * </a><p> 
38    * 
39    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
40    * @version $Revision: 1.0 $ 
41    */ 
42   public class CSB0 extends AbstractCSB{ 
43    
44     /** 
45      * Assign new weights for the train instances. 
46      * 
47      * @param classifier The classifier to be used to calculate new instances weights. 
48      * @param alfa The classifier's vote weight 
49      */ 
50     protected void reweight(DistributionClassifier classifier, double alfa) throws Exception{ 
51    
52       Instance instance; 
53       double newWeightsSum = 0; 
54    
55       for (int i = 0; i < m_NumInstances; i++){ 
56         instance = m_TrainData.instance(i); 
57         int trueClassValue = (int) instance.classValue(); 
58         int classifiedAs = (int) classifier.classifyInstance(instance); 
59         if (classifiedAs != trueClassValue) 
60           instance.setWeight(instance.weight() * m_CostMatrix.getElement(trueClassValue, classifiedAs)); 
61         newWeightsSum += instance.weight(); 
62       } 
63    
64       normalizeWeights(newWeightsSum); 
65     } 
66    
67     /** 
68      * Main method for testing this class. 
69      * 
70      * @param argv the options 
71      */ 
72     public static void main(String[] argv){ 
73    
74       try{ 
75         System.out.println(Evaluation.evaluateModel(new CSB0(), argv)); 
76       } 
77       catch (Exception e){ 
78         System.err.println(e.getMessage()); 
79       } 
80     }
