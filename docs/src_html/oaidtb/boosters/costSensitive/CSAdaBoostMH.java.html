40   /** 
41    * A class that implements a modification of AdaBoostMH for cost sensitive classification; it uses 
42    * an AbstractCSB booster to construct an hypothesis for each class, providing it with a 2x2 cost matrix 
43    * where the element (0,1) is the cost of misclassifying any other class as that class and the element 
44    * (1,0) is the cost of misclassifying that class as any other class. 
45    * 
46    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
47    * @version $Revision: 1.0 $

155    /** 
156     * Boosting method. 
157     * Reset the model and initialize data. 
158     * 
159     * @param data the training data to be used for generating the 
160     * boosted classifier. 
161     * @exception Exception if the classifier could not be built successfully 
162     */ 
163    public void buildClassifier(Instances data) throws Exception{ 
164   
165      //Check glogal settings. 
166      if (m_Booster == null) 
167        throw new Exception("A base classifier has not been specified!"); 
168   
169      if (!(m_Booster instanceof MulticlassExtensibleBooster)) 
170        throw new Exception("Base classifier must be a \"Multiclass Extensible Booster\"."); 
171   
172      if (!data.classAttribute().isNominal()) 
173        throw new Exception("AdaBoostMH can't handle not nominal classes!"); 
174   
175      //We copy them thus ensuring AdaBoostMH nor other class mess it up. 
176      m_TrainData = new Instances(data); 
177      m_TrainData.deleteWithMissingClass(); 
178   
179      if ((m_NumInstances = m_TrainData.numInstances()) == 0) 
180        throw new Exception("No train instances without class missing!"); 
181   
182      if (3 > (m_NumClasses = m_TrainData.numClasses())) 
183        System.err.println("AdaBoostMH could not be useful with binary class problems."); 
184   
185      ((MulticlassExtensibleBooster) m_Booster).setUseOwnTrainData(false); 
186   
187      m_OriginalDataClasses = new double[m_NumInstances]; 
188   
189      for (int i = 0; i < m_NumInstances; i++) 
190        m_OriginalDataClasses[i] = m_TrainData.instance(i).classValue(); 
191   
192      //Set up the cost matrix 
193      if (m_MatrixSource == MATRIX_ON_DEMAND){ 
194        String costName = data.relationName() + CostMatrix.FILE_EXTENSION; 
195        File costFile = new File(getOnDemandDirectory(), costName); 
196        if (!costFile.exists()){ 
197          throw new Exception("On-demand cost file doesn't exist: " + costFile); 
198        } 
199        setCostMatrix(new CostMatrix(new BufferedReader( 
200          new FileReader(costFile)))); 
201      } 
202      else if (m_MatrixSource == MATRIX_SUPPLIED){ 
203        if (m_CostMatrix == null){ 
204          // try loading an old format cost file 
205          m_CostMatrix = new CostMatrix(data.numClasses()); 
206          m_CostMatrix.readOldFormat(new BufferedReader( 
207            new FileReader(m_CostFile))); 
208        } 
209      } 
210      else{ 
211        m_CostMatrix = AbstractCSB.generateMinorityClassSensitiveCostMatrix(m_TrainData, m_DefaultMatrixCostFactor); 
212        m_MatrixSource = DEFAULT_MATRIX; 
213      } 
214   
215      //Creamos el nuevo atributo que representará la "metaclase". 
216      FastVector my_nominal_values = new FastVector(2); 
217   
218      my_nominal_values.addElement("0"); 
219      my_nominal_values.addElement("1"); 
220   
221      int classIndex = m_TrainData.classIndex(); 
222      m_TrainData.setClassIndex(-1); 
223   
224      m_TrainData.deleteAttributeAt(classIndex); 
225      m_TrainData.insertAttributeAt(new Attribute("PseudoClass", my_nominal_values), classIndex); 
226      m_TrainData.setClassIndex(classIndex); 
227   
228      m_Boosters = new Booster[m_NumClasses]; 
229      SerializedObject serializedBooster = new SerializedObject(m_Booster); 
230      for (int i = 0; i < m_NumClasses; i++) 
231        m_Boosters[i] = (Booster) serializedBooster.getObject(); 
232   
233      m_InstanceWeights = new double[m_NumClasses][m_NumInstances]; 
234   
235      boolean showDebugInfo = m_Booster.getDebug(); 
236   
237      for (int i = 0; i < m_NumClasses; i++){ 
238        CostMatrix tmp = getBinaryCostMatrix(i); 
239        if (showDebugInfo){ 
240          System.err.println("\n*-*-*-*-*Building booster for class " + i + "*-*-*-*-*"); 
241          System.err.println("Cost of missclassify the class " + i + ": " + tmp.getElement(1, 0)); 
242          System.err.println("Cost of missclassify as class " + i + ": " + tmp.getElement(0, 1)); 
243  //        System.err.println("NOTE: Due to a ¿bug? in Matrix.toString() they can be shown rounded.") 
244        } 
245        relabel(i); 
246        ((AbstractCSB) m_Boosters[i]).setCostMatrix(tmp); 
247        m_Boosters[i].buildClassifier(m_TrainData); 
248        saveBoosterWeights(i); 
249      } 
250    }

252    /** 
253     * Given a class index, returns a 2x2 cost matrix where where the element (0,1) is the cost of misclassifying 
254     * any other class as that class and the element (1,0) is the cost of misclassifying that class as any other class. 
255     * 
256     * @param classIndex The index of the class we will construct a booster for 
257     * 
258     * @return The binary cost matrix 
259     */ 
260    private CostMatrix getBinaryCostMatrix(int classIndex){ 
261   
262      CostMatrix c = new CostMatrix(2); 
263   
264      //Costo de clasificar la clase en cuestión como cualquier otra. 
265      double tmp = 0.0; 
266      for (int i = 0; i < m_NumClasses; i++) 
267        tmp += m_CostMatrix.getElement(classIndex, i); 
268   
269      c.setElement(1, 0, tmp); 
270   
271      //Costo de clasificar cualquier otra otra clase como la clase en cuestión. 
272      tmp = 0.0; 
273      for (int i = 0; i < m_NumClasses; i++) 
274        tmp += m_CostMatrix.getElement(i, classIndex); 
275   
276      c.setElement(0, 1, tmp); 
277   
278      return c; 
279    }

281    /** 
282     * Parses a given list of options. Valid options are:<PRE> 
283     * 
284     * -C cost file 
285     * File name of a cost matrix to use. If this is not supplied, a cost 
286     * matrix will be loaded on demand. The name of the on-demand file 
287     * is the relation name of the training data plus ".cost", and the 
288     * path to the on-demand file is specified with the -D option. 
289     * 
290     * -D directory 
291     * Name of a directory to search for cost files when loading costs on demand 
292     * (default current directory). 
293     * 
294     * -F 
295     *  Use the default cost matrix. 
296     * 
297     * -X costFactor 
298     * The cost of misclassifying a minority class when using the default cost matrix. 
299     * 
300     * -B classname 
301     * Specify a class (must be an AbstractCSB subclass) for 
302     * use as the base Booster. 
303     * 
304     * </PRE> 
305     * 
306     * @param options the list of options as an array of strings 
307     * @exception Exception if an option is not supported 
308     */ 
309    public void setOptions(String[] options) throws Exception{ 
310   
311      String boosterName = Utils.getOption('B', options); 
312      if (boosterName.length() == 0) 
313        setBooster((Booster) Utils.forName(AbstractCSB.class, 
314                                           "oaidtb.boosters.costSensitive.CSB2", options)); 
315      else 
316        setBooster((Booster) Utils.forName(AbstractCSB.class, boosterName, options)); 
317   
318      if(null != m_Booster) 
319        ((MulticlassExtensibleBooster)m_Booster).setUseOwnTrainData(false); 
320   
321      if (Utils.getFlag('X', options)) 
322        setCostMatrixSource(new SelectedTag(DEFAULT_MATRIX, TAGS_MATRIX_SOURCE)); 
323   
324      String defaultCostMatrixFactor = Utils.getOption('F', options); 
325      if (defaultCostMatrixFactor.length() != 0) 
326        setDefaultMatrixCostFactor(Double.parseDouble(defaultCostMatrixFactor)); 
327   
328      String costFile = Utils.getOption('C', options); 
329      if (costFile.length() != 0){ 
330        try{ 
331          setCostMatrix(new CostMatrix(new BufferedReader( 
332            new FileReader(costFile)))); 
333        } 
334        catch (Exception ex){ 
335          // now flag as possible old format cost matrix. Delay cost matrix 
336          // loading until buildClassifer is called 
337          setCostMatrix(null); 
338        } 
339        setCostMatrixSource(new SelectedTag(MATRIX_SUPPLIED, 
340                                            TAGS_MATRIX_SOURCE)); 
341        m_CostFile = costFile; 
342      } 
343      else{ 
344        if (m_MatrixSource != DEFAULT_MATRIX) 
345          setCostMatrixSource(new SelectedTag(MATRIX_ON_DEMAND, 
346                                              TAGS_MATRIX_SOURCE)); 
347      } 
348   
349      String demandDir = Utils.getOption('D', options); 
350      if (demandDir.length() != 0){ 
351        setOnDemandDirectory(new File(demandDir)); 
352      } 
353    }

355    /** 
356     * Gets the current settings of the Classifier. 
357     * 
358     * @return an array of strings suitable for passing to setOptions 
359     */ 
360    public String[] getOptions(){ 
361   
362      String[] otherOptions = super.getOptions(); 
363   
364      String[] options = new String[otherOptions.length + 3]; 
365   
366      int current = 0; 
367   
368      if (m_MatrixSource == MATRIX_SUPPLIED){ 
369        if (m_CostFile != null){ 
370          options[current++] = "-C"; 
371          options[current++] = "" + m_CostFile; 
372        } 
373      } 
374      else if (m_MatrixSource == MATRIX_ON_DEMAND){ 
375        options[current++] = "-O"; 
376        options[current++] = "" + getOnDemandDirectory(); 
377      } 
378      else{ 
379        options[current++] = "-X"; 
380        options[current++] = "-F"; 
381        options[current++] = "" + getDefaultMatrixCostFactor(); 
382      } 
383   
384      System.arraycopy(otherOptions, 0, 
385                       options, current, 
386                       otherOptions.length); 
387   
388      current += otherOptions.length; 
389      while (current < options.length) 
390        options[current++] = ""; 
391   
392      return options; 
393    }
