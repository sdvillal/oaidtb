23   package oaidtb.boosters.costSensitive; 
24    
25   import weka.classifiers.DistributionClassifier; 
26   import weka.classifiers.Evaluation; 
27   import weka.core.Instance; 
28    
29   /** 
30    * One of the small changes proposed for AdaCost in:<p> 
31    * 
32    * <a href="http://www.gscit.monash.edu.au/~kmting/Papers/icml00.ps"> 
33    * Ting, K.M., <i>A Comparative Study of Cost-Sensitive Boosting Algorithms.</i>. 
34    * Proceedings of The Seventeenth International Conference on Machine Learning. pp. 983-990. 
35    * San Francisco: Morgan Kaufmann. Stanford University, June 29 - July 2, 2000. 
36    * </a><p> 
37    * 
38    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
39    * @version $Revision: 1.0 $ 
40    */ 
41   public class AdaCostB2 extends AdaCostB1{ 
42    
43     /** 
44      *  We return to the origin using the same function in AbstractCSB; in other words, not using 
45      *  the cost adjustment function when calculating alfa. 
46      */ 
47     protected double calculateAlfa(DistributionClassifier classifier) throws Exception{ 
48    
49       double tmp = 0; 
50       double sumTmp = 0; 
51    
52       for (int i = 0; i < m_NumInstances; i++){ 
53    
54         Instance instance = m_TrainData.instance(i); 
55         double[] distributionForInstance = classifier.distributionForInstance(instance); 
56         int classifiedAs = oaidtb.misc.Utils.maxIndex(distributionForInstance); 
57    
58         if (classifiedAs == instance.classValue()) 
59           tmp += instance.weight() * distributionForInstance[classifiedAs]; 
60         else 
61           tmp -= instance.weight() * distributionForInstance[classifiedAs]; 
62    
63         sumTmp += instance.weight(); 
64       } 
65    
66       tmp = (tmp + NO_DIVISION_BY_ZERO) / (sumTmp + NO_DIVISION_BY_ZERO); 
67    
68       return Math.log((1 + tmp) / (1 - tmp)) / 2; 
69     } 
70    
71     /** 
72      * Main method for testing this class. 
73      * 
74      * @param argv the options 
75      */ 
76     public static void main(String[] argv){ 
77    
78       try{ 
79         System.out.println(Evaluation.evaluateModel(new AdaCostB2(), argv)); 
80       } 
81       catch (Exception e){ 
82         System.err.println(e.getMessage()); 
83       } 
84     }
