149      if (m_HowClassify instanceof MaximumVoteCriterion) 
150        return MVC; 
151      if (m_HowClassify instanceof MaximumVoteCriterionUsingConfidenceLevels) 
152        return MVC_UCL; 
153      if (m_HowClassify instanceof MinimumExpectedCostCriterion)

275    /** Reweight the train data according with the cost matrix. */ 
276    private void initializeWeightsBasedOnCosts(){ 
277   
278      double[] costs = new double[m_NumClasses]; 
279   
280      //Calculate the cost of misclassifying a class i. 
281      for (int i = 0; i < m_NumClasses; i++) 
282        costs[i] = weka.core.Utils.sum(m_CostMatrix.getRow(i)) - m_CostMatrix.getElement(i, i); 
283   
284      double newWeightsSum = 0; 
285      for (int i = 0; i < m_TrainData.numInstances(); i++){ 
286        Instance instance = m_TrainData.instance(i); 
287        instance.setWeight(instance.weight() * costs[(int) instance.classValue()]); 
288        newWeightsSum += instance.weight(); 
289      } 
290   
291      normalizeWeights(newWeightsSum); 
292    }

294    /** 
295     * Calculate the classifier's vote weight according to its training error. 
296     * 
297     * @param classifier The base classifier 
298     * @return The classifier's vote weight 
299     * @throws Exception If an error occurs 
300     */ 
301    protected double calculateAlfa(DistributionClassifier classifier) throws Exception{ 
302   
303      double tmp = 0; 
304      double sumTmp = 0; 
305   
306      for (int i = 0; i < m_NumInstances; i++){ 
307   
308        Instance instance = m_TrainData.instance(i); 
309        double[] distributionForInstance = classifier.distributionForInstance(instance); 
310        int classifiedAs = oaidtb.misc.Utils.maxIndex(distributionForInstance); 
311   
312        if (classifiedAs == instance.classValue()) 
313          tmp += instance.weight() * distributionForInstance[classifiedAs]; 
314        else 
315          tmp -= instance.weight() * distributionForInstance[classifiedAs]; 
316   
317        sumTmp += instance.weight(); 
318      } 
319   
320      tmp = (tmp + NO_DIVISION_BY_ZERO) / (sumTmp + NO_DIVISION_BY_ZERO); 
321   
322      return Math.log((1 + tmp) / (1 - tmp)) / 2; 
323    }

325    /** 
326     * Boosting method. 
327     * Reset the model and initialize data. 
328     * 
329     * @param data the training data to be used for generating the 
330     * boosted classifier. 
331     * @exception java.lang.Exception if the classifier could not be built successfully 
332     */ 
333    public void buildClassifier(Instances data) throws Exception{ 
334   
335      //Check glogal settings. 
336      if (m_Classifier == null || !(m_Classifier instanceof DistributionClassifier)) 
337        throw new Exception("A base classifier has not been specified or it isn't a DistributionClassifier."); 
338   
339      if (m_UseOwnTrainData){ 
340        //Check the correct format of training instances. 
341        if (data.checkForStringAttributes()) 
342          throw new Exception("Can't handle string attributes!"); 
343   
344        if (data.classAttribute().isNumeric()) 
345          throw new Exception("CSB can't handle a numeric class!"); 
346   
347        //We copy them thus ensuring CSB nor other class mess it up. 
348        m_TrainData = new Instances(data); 
349        m_BoosterReady = false;  //No "return" can be done from here. 
350        m_TrainData.deleteWithMissingClass(); 
351        if (m_TrainData.numInstances() == 0) 
352          throw new Exception("No train instances without class missing!"); 
353      } 
354      else 
355      //We entrust to the "caller class" the correctness of the train data instances. 
356        m_TrainData = data; 
357   
358      //Set up the number of instances. 
359      m_NumInstances = m_TrainData.numInstances(); 
360   
361      //Set up the number of classes. 
362      m_NumClasses = m_TrainData.numClasses(); 
363   
364      //Set up the number of performed iterations. 
365      m_NumIterations = 0; 
366   
367      //Initialize the classifiers array. 
368      m_Classifiers = new WeightedClassifierVector(); 
369   
370      //Set up the cost matrix 
371      if (m_MatrixSource == MATRIX_ON_DEMAND){ 
372        String costName = data.relationName() + CostMatrix.FILE_EXTENSION; 
373        File costFile = new File(getOnDemandDirectory(), costName); 
374        if (!costFile.exists()){ 
375          throw new Exception("On-demand cost file doesn't exist: " + costFile); 
376        } 
377        setCostMatrix(new CostMatrix(new BufferedReader( 
378          new FileReader(costFile)))); 
379      } 
380      else if (m_MatrixSource == MATRIX_SUPPLIED){ 
381        if (m_CostMatrix == null){ 
382          // try loading an old format cost file 
383          m_CostMatrix = new CostMatrix(data.numClasses()); 
384          m_CostMatrix.readOldFormat(new BufferedReader( 
385            new FileReader(m_CostFile))); 
386        } 
387      } 
388      else{ 
389        m_CostMatrix = generateMinorityClassSensitiveCostMatrix(m_TrainData, m_DefaultMatrixCostFactor); 
390        m_MatrixSource = DEFAULT_MATRIX; 
391      } 
392   
393  //    m_CostMatrix = normalizeCostMatrix(m_CostMatrix); 
394   
395      if (m_Debug) 
396        System.err.println("Cost matrix: \n" + m_CostMatrix.toString()); 
397   
398      //Initialize instance's weight & m_NormFactor. 
399      initializeNormFactor(); 
400   
401      if (m_InitializeWeightsUsingCosts) 
402        initializeWeightsBasedOnCosts(); 
403      else 
404        initializeWeights(); 
405   
406      //We can now perform new iterations. 
407      m_BoosterReady = true; 
408   
409      //Perform first "InitialIterations" iterations. 
410      nextIterations(m_InitialIterations); 
411    }

413    /** 
414     * Boosting method. Boosts any classifier that can handle weighted 
415     * instances. 
416     * 
417     * @param numIterations The number of iterations to perform. 
418     * 
419     * @exception java.lang.Exception if the classifier could not be built successfully 
420     */ 
421    protected void buildClassifierWithWeights(int numIterations) throws Exception{ 
422   
423      Classifier baseClassifier; 
424      SerializedObject serializedClassifier = new SerializedObject(m_Classifier); 
425   
426      if (m_Debug) 
427        System.err.println("Boosting without resampling."); 
428   
429      // Do iterations. 
430      for (numIterations--; numIterations >= 0; numIterations--){ 
431   
432        if (m_Debug) 
433          System.err.println("Training classifier " + (m_NumIterations + 1)); 
434   
435        //Copy the base classifier. 
436        baseClassifier = (Classifier) serializedClassifier.getObject(); 
437   
438        // Build the classifier. 
439        baseClassifier.buildClassifier(m_TrainData); 
440   
441        // Calculate the classifier's vote weight 
442        double alfa = calculateAlfa((DistributionClassifier) baseClassifier); 
443   
444        //Weights redistribution. 
445        reweight((DistributionClassifier) baseClassifier, alfa); 
446   
447        //"Commit". 
448        m_Classifiers.add(baseClassifier, alfa); 
449        m_NumIterations++; 
450      } 
451    }

453    /** 
454     * Boosting method. Boosts any classifier using training instances 
455     * weight for resampling. 
456     * 
457     * @param numIterations The number of iterations to perform. 
458     * 
459     * @exception java.lang.Exception if the classifier could not be built successfully 
460     */ 
461    protected void buildClassifierUsingResampling(int numIterations) throws Exception{ 
462   
463      Classifier baseClassifier; 
464      SerializedObject serializedClassifier = new SerializedObject(m_Classifier); 
465      Random randomInstance = new Random(m_ResampleSeed); //Random number generator for resample. 
466      Instances sample;                                   //The resampled training dataset. 
467   
468      if (m_Debug) 
469        System.err.println("Boosting with resampling."); 
470   
471      // Do iterations 
472      for (numIterations--; numIterations >= 0; numIterations--){ 
473   
474        if (m_Debug){ 
475          System.err.println("Training classifier " + (m_NumIterations + 1)); 
476        } 
477   
478        // Resample. 
479        sample = m_TrainData.resampleWithWeights(randomInstance); 
480   
481        //Copy the base classifier. 
482        baseClassifier = (Classifier) serializedClassifier.getObject(); 
483   
484        // Build the classifier. 
485        baseClassifier.buildClassifier(sample); 
486   
487        // Calculate the classifier's vote weight 
488        double alfa = calculateAlfa((DistributionClassifier) baseClassifier); 
489   
490        //Weights redistribution. 
491        reweight((DistributionClassifier) baseClassifier, alfa); 
492   
493        //"Commit" 
494        m_Classifiers.add(baseClassifier, alfa); 
495        m_NumIterations++; 
496      } 
497    }

542    /** 
543     * Constructs a CostMatrix for the data assigning 0 cost to all correct classifications, 
544     * 1 to misclassifying a not minority class and the supplied costFactor to misclassifying 
545     * the minority class. 
546     * 
547     * @param data The instances 
548     * @param costFactor The cost of misclassifying a minority class 
549     * @return The cost matrix constructed 
550     */ 
551    public final static CostMatrix generateMinorityClassSensitiveCostMatrix(Instances data, double costFactor){ 
552   
553      int numClasses = data.numClasses(); 
554      CostMatrix tmp = new CostMatrix(numClasses); 
555      double classCount[] = new double[numClasses]; 
556   
557      for (int i = 0; i < data.numInstances(); i++) 
558        classCount[(int) data.instance(i).classValue()]++; 
559   
560      int minorityClass = oaidtb.misc.Utils.minIndex(classCount); 
561   
562      //Optimize these loops wouldn't be legible 
563      for (int i = 0; i < numClasses; i++) 
564        for (int j = 0; j < numClasses; j++) 
565          if (i != j) 
566            tmp.setElement(j, i, 1); 
567   
568      for (int i = 0; i < numClasses; i++) 
569        if (i != minorityClass) 
570          tmp.setElement(minorityClass, i, costFactor); 
571   
572      return tmp; 
573    }

575    /** 
576     * Normalize a cost matrix so that the sum of all its elements will be 1. 
577     * 
578     * @param costMatrix The cost matriz 
579     * @return The cost matrix "normalized" 
580     */ 
581    public final static CostMatrix normalizeCostMatrix(CostMatrix costMatrix){ 
582   
583      double sumTmp = 0; 
584   
585      //A cost matrix is always squared. 
586      for (int i = 0; i < costMatrix.numRows(); i++) 
587        for (int j = 0; j < costMatrix.numColumns(); j++) 
588          sumTmp += costMatrix.getElement(i, j); 
589   
590      if (sumTmp == 0) 
591        return costMatrix; 
592   
593      CostMatrix normalizedCostMatrix = new CostMatrix(costMatrix.numRows()); 
594   
595      for (int i = 0; i < costMatrix.numRows(); i++) 
596        for (int j = 0; j < costMatrix.numColumns(); j++) 
597          normalizedCostMatrix.setElement(i, j, costMatrix.getElement(i, j) / sumTmp); 
598   
599      return normalizedCostMatrix; 
600    }

639    /** 
640     * Returns description of the boosted classifier. 
641     * 
642     * @return description of the boosted classifier as a string 
643     */ 
644    public String toString(){ 
645   
646      StringBuffer text = new StringBuffer(); 
647   
648      if (m_NumIterations == 0){ 
649        text.append("Booster: No model built yet.\n"); 
650      } 
651      else if (m_NumIterations == 1){ 
652        text.append("Booster: No boosting possible, one classifier used!\n"); 
653        text.append("Cost Matrix: " + m_CostMatrix); 
654        text.append(m_Classifiers.get(0).toString() + "\n"); 
655      } 
656      else{ 
657        text.append("Cost Matrix: " + m_CostMatrix); 
658        text.append("CSB: Base classifiers: \n\n");

660          text.append(m_Classifiers.get(i) + "\n Vote: " + m_Classifiers.getWeight(i) + "\n\n"); 
661        } 
662        text.append("Number of performed Iterations: " + m_NumIterations + "\n"); 
663      } 
664   
665      return text.toString(); 
666    }

678     * Interface for the internal classes that will serve to configure how the 
679     * base classifiers will be combined to obtain the final hypothesis 
680     * for a given instance. 
681     */ 
682    private interface CombinedPredictionModel extends Serializable{ 
683   
684      /**

707    /** Classify using the minimum expected cost criterion. */ 
708    private class MinimumExpectedCostCriterion implements CombinedPredictionModel{ 
709   
710      public double[] distributionForInstance(Instance instance) throws Exception{ 
711   
712        double[] distributionForInstance; 
713        double[] tmp = new double[m_NumClasses]; 
714   
715        for (int i = 0; i < m_NumIterations; i++){ 
716          double weight = m_Classifiers.getWeight(i); 
717          distributionForInstance = ((DistributionClassifier) m_Classifiers.get(i)).distributionForInstance(instance); 
718          int classifiedAs = oaidtb.misc.Utils.maxIndex(distributionForInstance); 
719          double confidence = distributionForInstance[classifiedAs]; 
720          for (int j = 0; j < classifiedAs; j++) 
721            tmp[j] += weight * confidence * m_CostMatrix.getElement(classifiedAs, j); 
722          for (int j = classifiedAs + 1; j < m_NumClasses; j++) 
723            tmp[j] += weight * confidence * m_CostMatrix.getElement(classifiedAs, j); 
724        } 
725   
726        int[] sortedArray = weka.core.Utils.sort(tmp); 
727   
728        double[] swapTmp = new double[m_NumClasses]; 
729        int maxIndex = m_NumClasses - 1; 
730        double sumTmp = 0; 
731        for (int i = 0; i < m_NumClasses; i++){ 
732          sumTmp += swapTmp[sortedArray[maxIndex]] = tmp[sortedArray[i]]; 
733          maxIndex--; 
734        } 
735   
736        for (int i = 0; i < m_NumClasses; i++) 
737          swapTmp[i] /= sumTmp; 
738   
739        return swapTmp; 
740      }

909    /** 
910     * Parses a given list of options. Valid options are:<PRE> 
911     * 
912     * -M <combined prediction selection model> 
913     * Specify how to combine the base classifiers in the hypothesis construction; valid options are 
914     * (case insensitive): 
915     *   - MVC = Maximum vote criterion 
916     *   - MVC_UCL = Maximum vote criterion using confidence levels 
917     *   - MECC = Minimum expected cost criterion 
918     *   - MECC_UCL = Minimum expected cost criterion using confidence levels 
919     * 
920     * -C cost file 
921     * File name of a cost matrix to use. If this is not supplied, a cost 
922     * matrix will be loaded on demand. The name of the on-demand file 
923     * is the relation name of the training data plus ".cost", and the 
924     * path to the on-demand file is specified with the -D option. 
925     * 
926     * -O directory 
927     * Name of a directory to search for cost files when loading costs on demand 
928     * (default current directory). 
929     * 
930     * -U 
931     * Initialize the weights using the costs. 
932     * 
933     * -F 
934     *  Use the default cost matrix. 
935     * 
936     * -X costFactor 
937     * The cost of misclassifying a minority class when using the default cost matrix. 
938     * 
939     * Plus the rest of the superclass options. 
940     * 
941     * </PRE> 
942     * 
943     * @param options the list of options as an array of strings 
944     * @exception Exception if an option is not supported 
945     */ 
946    public void setOptions(String[] options) throws Exception{ 
947   
948      super.setOptions(options); 
949   
950      String combinedPredictionModel = Utils.getOption('M', options); 
951      if (combinedPredictionModel.length() != 0){ 
952        if (combinedPredictionModel.compareToIgnoreCase("MVC") == 0) 
953          setCombinedPredictionModel(new SelectedTag(MVC, CPM_TAGS)); 
954        else if (combinedPredictionModel.compareToIgnoreCase("MVC_UCL") == 0) 
955          setCombinedPredictionModel(new SelectedTag(MVC_UCL, CPM_TAGS)); 
956        else if (combinedPredictionModel.compareToIgnoreCase("MECC") == 0) 
957          setCombinedPredictionModel(new SelectedTag(MECC, CPM_TAGS)); 
958        else if (combinedPredictionModel.compareToIgnoreCase("MECC_UCL") == 0) 
959          setCombinedPredictionModel(new SelectedTag(MECC_UCL, CPM_TAGS)); 
960      } 
961   
962      setInitializeWeightsUsingCosts(Utils.getFlag('U', options)); 
963   
964      if (Utils.getFlag('X', options)) 
965        setCostMatrixSource(new SelectedTag(DEFAULT_MATRIX, TAGS_MATRIX_SOURCE)); 
966   
967      String defaultCostMatrixFactor = Utils.getOption('F', options); 
968      if (defaultCostMatrixFactor.length() != 0) 
969        setDefaultMatrixCostFactor(Double.parseDouble(defaultCostMatrixFactor)); 
970   
971      String costFile = Utils.getOption('C', options); 
972      if (costFile.length() != 0){ 
973        try{ 
974          setCostMatrix(new CostMatrix(new BufferedReader( 
975            new FileReader(costFile)))); 
976        } 
977        catch (Exception ex){ 
978          // now flag as possible old format cost matrix. Delay cost matrix 
979          // loading until buildClassifer is called 
980          setCostMatrix(null); 
981        } 
982        setCostMatrixSource(new SelectedTag(MATRIX_SUPPLIED, 
983                                            TAGS_MATRIX_SOURCE)); 
984        m_CostFile = costFile; 
985      } 
986      else{ 
987        if (m_MatrixSource != DEFAULT_MATRIX) 
988          setCostMatrixSource(new SelectedTag(MATRIX_ON_DEMAND, 
989                                              TAGS_MATRIX_SOURCE)); 
990      } 
991   
992      String demandDir = Utils.getOption('O', options); 
993      if (demandDir.length() != 0){ 
994        setOnDemandDirectory(new File(demandDir)); 
995      } 
996    }

998    /** 
999     * Gets the current settings of the Classifier. 
1000    * 
1001    * @return an array of strings suitable for passing to setOptions 
1002    */ 
1003   public String[] getOptions(){ 
1004  
1005     String[] otherOptions = super.getOptions(); 
1006  
1007     String[] options = new String[otherOptions.length + 6]; 
1008  
1009     int current = 0; 
1010  
1011     String combinedPredictionModelSelected; 
1012     switch (combinedPredictionModelSelected()){ 
1013       case MVC: 
1014         combinedPredictionModelSelected = "MVC"; 
1015         break; 
1016       case MVC_UCL: 
1017         combinedPredictionModelSelected = "MVC_UCL"; 
1018         break; 
1019       case MECC: 
1020         combinedPredictionModelSelected = "MECC"; 
1021         break; 
1022       case MECC_UCL: 
1023         combinedPredictionModelSelected = "MECC_UCL"; 
1024         break; 
1025       default: 
1026         combinedPredictionModelSelected = "Undefined"; 
1027     } 
1028     options[current++] = "-M"; 
1029     options[current++] = "" + combinedPredictionModelSelected; 
1030  
1031     if (isInitializeWeightsUsingCosts()) 
1032       options[current++] = "-U"; 
1033  
1034     if (m_UseOwnTrainData){ 
1035       if (m_MatrixSource == MATRIX_SUPPLIED){ 
1036         if (m_CostFile != null){ 
1037           options[current++] = "-C"; 
1038           options[current++] = "" + m_CostFile; 
1039         } 
1040       } 
1041       else if (m_MatrixSource == MATRIX_ON_DEMAND){ 
1042         options[current++] = "-O"; 
1043         options[current++] = "" + getOnDemandDirectory(); 
1044       } 
1045       else{ 
1046         options[current++] = "-X"; 
1047         options[current++] = "-F"; 
1048         options[current++] = "" + getDefaultMatrixCostFactor(); 
1049       } 
1050     } 
1051  
1052     System.arraycopy(otherOptions, 0, 
1053                      options, current, 
1054                      otherOptions.length); 
1055  
1056     current += otherOptions.length; 
1057     while (current < options.length) 
1058       options[current++] = ""; 
1059  
1060     return options; 
1061   }
