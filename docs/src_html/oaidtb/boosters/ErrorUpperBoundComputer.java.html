23   package oaidtb.boosters; 
24    
25   /** 
26    * Interface for boosters that can calculate any kind of theoretical guaranteed error 
27    * measure upper bound (academical & validation stuff); obviously, it isn' t neccessary 
28    * for the booster to work and it can be harmful for the algorithm performance, 
29    * so the implementations of this interface  must be carefully revised to not become a bottleneck. 
30    * 
31    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
32    * @version $Revision: 1.0 $ 
33    */ 
34   public interface ErrorUpperBoundComputer{ 
35    
36     /** @return The error upper bound string (by example, "Training error upper bound.") */ 
37     String getErrorUpperBoundName(); 
38    
39     /** 
40      * Get if the error bound is being calculated. 
41      * 
42      * @return true if it's being computed. 
43      */ 
44     boolean getCalculateErrorUpperBound(); 
45    
46     /** 
47      * Set (only if no iterations have be done) if the error bound must be 
48      * wether or not calculated. 
49      * 
50      * @param b If the training error upper bound must or not be calculated. 
51      */ 
52     void setCalculateErrorUpperBound(boolean b); 
53    
54     /** 
55      * Get the error bound. 
56      * 
57      * @return The training error upper bound or -1 if no iterations have been done. 
58      */ 
59     double getErrorUpperBound(); 
60   }
