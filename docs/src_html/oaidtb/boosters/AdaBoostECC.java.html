82     /** 
83      * Boosting method. 
84      * Reset the model and initialize data. 
85      * 
86      * @param data the training data to be used for generating the 
87      * boosted classifier. 
88      * @exception java.lang.Exception if the classifier could not be built successfully 
89      */ 
90     public void buildClassifier(Instances data) throws Exception{ 
91    
92       //Check glogal settings. 
93       if (m_Classifier == null) 
94         throw new Exception("A base classifier has not been specified!"); 
95    
96       if (m_Coloring == null) 
97         throw new Exception("The \"coloring function\" hasn't been specified."); 
98    
99       //Check the correct format of training instances. 
100      if (data.checkForStringAttributes()) 
101        throw new Exception("Can't handle string attributes!"); 
102   
103      if (data.classAttribute().isNumeric()) 
104        throw new Exception("AdaBoostECC can't handle a numeric class!"); 
105   
106      //We copy them thus ensuring AdaBoostECC nor other class mess it up. 
107      m_TrainData = new Instances(data); 
108      m_BoosterReady = false;  //No "return" can be done from here. 
109      m_TrainData.deleteWithMissingClass(); 
110   
111      if (m_TrainData.numInstances() == 0) 
112        throw new Exception("No train instances without class missing!"); 
113   
114      //Set up the number of classes. Be careful with binary class problems (not use AdaBoostECC). 
115      if (3 > (m_NumClasses = m_TrainData.numClasses())) 
116        System.err.println("AdaBoostECC could be not useful with binary class problems."); 
117      //throw new Exception("AdaBoostECC is not useful with binary class problems."); 
118   
119      //Set up the number of instances. 
120      m_NumInstances = m_TrainData.numInstances(); 
121   
122      //Initialize the normalization factor. 
123      initializeNormFactor(); 
124   
125      //We store which class belongs to each training instance. 
126      m_OriginalDataClasses = new int[m_NumInstances]; 
127      for (int i = 0; i < m_NumInstances; i++) 
128        m_OriginalDataClasses[i] = (int) m_TrainData.instance(i).classValue(); 
129   
130      //Set up the number of performed iterations. 
131      m_NumIterations = 0; 
132   
133      //Initialize the coloring. 
134      m_Coloring.setProcessedAttribute(-1); //So it will take the class attribute. 
135      m_Coloring.setInputFormat(m_TrainData); 
136   
137      //It's neccesary to set m_TrainData class attribute as a binary one 
138      //to avoid problems with base classifiers (ej. VotedPerceptron). 
139      //This could be done by this line of code: 
140      //m_TrainData=Filter.useFilter(m_TrainData,m_Coloring); 
141      //We do it in a faster way. 
142      int classIndex = m_TrainData.classIndex(); 
143   
144      //Create the binary attribute wich will be the neww class 
145      FastVector my_nominal_values = new FastVector(2); 
146   
147      my_nominal_values.addElement("0"); 
148      my_nominal_values.addElement("1"); 
149   
150      Attribute newBinaryClass = new Attribute(m_TrainData.attribute(m_TrainData.classIndex()).name(), my_nominal_values); 
151   
152      //Remove old class attribute. 
153      m_TrainData.setClassIndex(-1); 
154      m_TrainData.deleteAttributeAt(classIndex); 
155   
156      //Insert and set the new class. 
157      m_TrainData.insertAttributeAt(newBinaryClass, classIndex); 
158      m_TrainData.setClassIndex(classIndex); 
159   
160      //Initialize the classifiers and their weights array. 
161      m_Classifiers = new ArrayList(); 
162   
163      //Initialize the "mislabel distribution". 
164      m_MislabelDistribution = new double[m_NumInstances][m_NumClasses]; 
165      double incorrectLabelWeight = 1.0 / (m_NumInstances * (m_NumClasses - 1)); 
166      for (int i = 0; i < m_NumInstances; i++) 
167        for (int j = 0; j < m_NumClasses; j++) 
168          m_MislabelDistribution[i][j] = (m_OriginalDataClasses[i] != j ? incorrectLabelWeight : 0.0); 
169   
170      //Initialize tha alfas and betas array. 
171      m_AlfasAndBetas = new ArrayList(); 
172   
173      //Now we can perform new iterations. 
174      m_BoosterReady = true; 
175   
176      //Initialize the training error upper bound. 
177      m_TrainingErrorUpperBound = 1; 
178   
179      if (m_Debug && m_CalculateTrainingErrorUpperBound){ 
180        System.err.println("Calculating the training error upper bound."); 
181        if (!m_UseSymmetricVersion && m_NormFactorUsed != Booster.NOT_NORMALIZE) 
182          System.err.println("Warning: because normalization will be done, this upper bound could be inaccurate."); 
183      } 
184   
185      //Perform first "InitialIterations" iterations. 
186      nextIterations(m_InitialIterations); 
187    }

189    /** 
190     * Boosting method. Boosts any classifier that can handle weighted 
191     * instances. 
192     * 
193     * @param numIterations The number of iterations to perform. 
194     * 
195     * @exception java.lang.Exception if the classifier could not be built successfully 
196     */ 
197    protected void buildClassifierWithWeights(int numIterations) throws Exception{ 
198   
199      AlfaAndBetaContainer alfaAndBeta; 
200      Classifier baseClassifier; 
201      SerializedObject serializedClassifier = new SerializedObject(m_Classifier); 
202   
203      if (m_Debug){ 
204        System.err.println("Boosting without resampling."); 
205        if (m_UseSymmetricVersion) 
206          System.err.println("Using the symmetric version."); 
207        else 
208          System.err.println("Using the asymmetric version."); 
209      } 
210   
211      // Do iterations. 
212      for (numIterations--; numIterations >= 0; numIterations--){ 
213   
214        if (m_Debug) 
215          System.err.println("Training classifier " + (m_NumIterations + 1)); 
216   
217        //Colororing genetation & U calculation 
218        double u = calculateU(); 
219   
220        // Weights redistribution. 
221        reweight(u); 
222   
223        //Coloring application. 
224        relabel(); 
225   
226        //Copy the base classifier. 
227        baseClassifier = (Classifier) serializedClassifier.getObject(); 
228   
229        // Build the classifier. 
230        baseClassifier.buildClassifier(m_TrainData); 
231   
232        // Alfa & Beta calculation. 
233        // Mislabel distribution computation. 
234        // Optional training error upper bound calculation. 
235        if (m_UseSymmetricVersion){ 
236          alfaAndBeta = calculateSymmetricVotes(baseClassifier); 
237          m_AlfasAndBetas.add(m_NumIterations, alfaAndBeta); 
238          calculateMislabelDistribution(baseClassifier, alfaAndBeta.alfa, alfaAndBeta.beta); 
239          if (m_CalculateTrainingErrorUpperBound) 
240            updateSymmetricTrainingErrorUpperBound(alfaAndBeta.alfa, u); 
241        } 
242        else{ 
243          alfaAndBeta = calculateAsymmetricVotes(baseClassifier); 
244          m_AlfasAndBetas.add(m_NumIterations, alfaAndBeta); 
245          if (m_CalculateTrainingErrorUpperBound) 
246            calculateMislabelDistribution(baseClassifier, alfaAndBeta.alfa, alfaAndBeta.beta, u); 
247          else 
248            calculateMislabelDistribution(baseClassifier, alfaAndBeta.alfa, alfaAndBeta.beta); 
249        } 
250   
251        if (m_Debug) 
252          System.err.println("\talfa = " + alfaAndBeta.alfa + "  beta = " + alfaAndBeta.beta); 
253   
254        //"Commit" 
255        m_Classifiers.add(baseClassifier); 
256        m_NumIterations++; 
257      } 
258    }

268    protected void buildClassifierUsingResampling(int numIterations) throws Exception{ 
269   
270      Random randomInstance = new Random(m_ResampleSeed); //Random number generator for resample. 
271      Instances sample;                                   //The resampled training dataset. 
272      AlfaAndBetaContainer alfaAndBeta; 
273      Classifier baseClassifier; 
274      SerializedObject serializedClassifier = new SerializedObject(m_Classifier); 
275   
276      if (m_Debug){ 
277        System.err.println("Boosting with resampling."); 
278        if (m_UseSymmetricVersion) 
279          System.err.println("Using the symmetric version."); 
280        else 
281          System.err.println("Using the asymmetric version."); 
282      } 
283   
284      // Do iterations 
285      for (numIterations--; numIterations >= 0; numIterations--){ 
286   
287        if (m_Debug) 
288          System.err.println("Training classifier " + (m_NumIterations + 1)); 
289   
290        //Colororing genetation & U calculation 
291        double u = calculateU(); 
292   
293        //Weights redistribution. 
294        reweight(u); 
295   
296        //Coloring application. 
297        relabel(); 
298   
299        // Resample. 
300        sample = m_TrainData.resampleWithWeights(randomInstance); 
301   
302        //Copy the base classifier. 
303        baseClassifier = (Classifier) serializedClassifier.getObject(); 
304   
305        // Build the classifier. 
306        baseClassifier.buildClassifier(sample); 
307   
308        // Alfa & Beta calculation. 
309        // Mislabel distribution computation. 
310        // Optional training error upper bound calculation. 
311        if (m_UseSymmetricVersion){ 
312          alfaAndBeta = calculateSymmetricVotes(baseClassifier); 
313          m_AlfasAndBetas.add(m_NumIterations, alfaAndBeta); 
314          calculateMislabelDistribution(baseClassifier, alfaAndBeta.alfa, alfaAndBeta.beta); 
315          if (m_CalculateTrainingErrorUpperBound) 
316            updateSymmetricTrainingErrorUpperBound(alfaAndBeta.alfa, u); 
317        } 
318        else{ 
319          alfaAndBeta = calculateAsymmetricVotes(baseClassifier); 
320          m_AlfasAndBetas.add(m_NumIterations, alfaAndBeta); 
321          if (m_CalculateTrainingErrorUpperBound) 
322            calculateMislabelDistribution(baseClassifier, alfaAndBeta.alfa, alfaAndBeta.beta, u); 
323          else 
324            calculateMislabelDistribution(baseClassifier, alfaAndBeta.alfa, alfaAndBeta.beta); 
325        } 
326   
327        if (m_Debug) 
328          System.err.println("\talfa = " + alfaAndBeta.alfa + "  beta = " + alfaAndBeta.beta); 
329   
330        //"Commit" 
331        m_Classifiers.add(baseClassifier); 
332        m_NumIterations++; 
333      }

336    /** 
337     * Alfa and beta claculation in the case of asymmetric version. 
338     * 
339     * @param baseClassifier The classifier used to calculate the votes. 
340     * 
341     * @return Alfa and beta. 
342     * 
343     * @throws java.lang.Exception If a classification error occurs. 
344     */ 
345    private AlfaAndBetaContainer calculateAsymmetricVotes(Classifier baseClassifier) 
346      throws Exception{ 
347   
348      double sumTmpA1 = 0; 
349      double sumTmpA2 = 0; 
350      double sumTmpB1 = 0; 
351      double sumTmpB2 = 0; 
352   
353      for (int i = 0; i < m_NumInstances; i++){ 
354        if (baseClassifier.classifyInstance(m_TrainData.instance(i)) == 1) 
355          if (m_Coloring.getCode(m_NumIterations, m_OriginalDataClasses[i]) == 1) 
356            sumTmpA1 += m_TrainData.instance(i).weight(); 
357          else 
358            sumTmpA2 += m_TrainData.instance(i).weight(); 
359        else if (m_Coloring.getCode(m_NumIterations, m_OriginalDataClasses[i]) == 0) 
360          sumTmpB1 += m_TrainData.instance(i).weight(); 
361        else 
362          sumTmpB2 += m_TrainData.instance(i).weight(); 
363      } 
364   
365      sumTmpA1 += Booster.NO_DIVISION_BY_ZERO; 
366      sumTmpA2 += Booster.NO_DIVISION_BY_ZERO; 
367      sumTmpB1 += Booster.NO_DIVISION_BY_ZERO; 
368      sumTmpB2 += Booster.NO_DIVISION_BY_ZERO; 
369   
370      return new AlfaAndBetaContainer(Math.log(sumTmpA1 / sumTmpA2) / 2, 
371                                      -Math.log(sumTmpB1 / sumTmpB2) / 2); 
372    }

374    /** 
375     * Alfa and beta calculation in the case of symmetric version. 
376     * 
377     * @param baseClassifier The classifier used to calculate the votes. 
378     * 
379     * @return Alfa and beta (= - alfa). 
380     * 
381     * @throws java.lang.Exception If a classification error occurs. 
382     */ 
383    private AlfaAndBetaContainer calculateSymmetricVotes(Classifier baseClassifier) 
384      throws Exception{ 
385   
386      double sumTmpA1 = 0; 
387      double sumTmpA2 = 0; 
388   
389      for (int i = 0; i < m_NumInstances; i++){ 
390        if (baseClassifier.classifyInstance(m_TrainData.instance(i)) == 
391          m_Coloring.getCode(m_NumIterations, m_OriginalDataClasses[i])) 
392          sumTmpA1 += m_TrainData.instance(i).weight(); 
393        else 
394          sumTmpA2 += m_TrainData.instance(i).weight(); 
395      } 
396   
397      sumTmpA1 += Booster.NO_DIVISION_BY_ZERO; 
398      sumTmpA2 += Booster.NO_DIVISION_BY_ZERO; 
399   
400      return new AlfaAndBetaContainer(sumTmpA1 = Math.log(sumTmpA1 / sumTmpA2) / 2, -sumTmpA1); 
401    }

403    /** 
404     * Update the "mislabel distribution, with respect to the last base cassifier generated 
405     * and its weights (alfa and beta). 
406     * 
407     * @param baseClassifier The classifier from we obtain the fresh new information. 
408     * @param alfa Classifier's alfa 
409     * @param beta Classifier's beta 
410     * 
411     * @throws java.lang.Exception If it fails. 
412     */ 
413    protected void calculateMislabelDistribution(Classifier baseClassifier, double alfa, double beta) 
414      throws Exception{ 
415   
416      double sumTmp = 0; // Faster than use Utils.normalize 
417   
418      //Faster than continously call alfaOrBeta(). 
419      alfa /= 2; 
420      beta /= 2; 
421   
422      for (int i = 0; i < m_NumInstances; i++){ 
423        if (baseClassifier.classifyInstance(m_TrainData.instance(i)) == 1) 
424          for (int j = 0; j < m_NumClasses; j++) 
425            sumTmp += 
426              (m_MislabelDistribution[i][j] = m_MislabelDistribution[i][j] * 
427              Math.exp(alfa * m_Coloring.getCode(m_NumIterations, j) - 
428                       alfa * m_Coloring.getCode(m_NumIterations, m_OriginalDataClasses[i]))); 
429        else 
430          for (int j = 0; j < m_NumClasses; j++) 
431            sumTmp += 
432              (m_MislabelDistribution[i][j] = m_MislabelDistribution[i][j] * 
433              Math.exp(beta * m_Coloring.getCode(m_NumIterations, j) - 
434                       beta * m_Coloring.getCode(m_NumIterations, m_OriginalDataClasses[i]))); 
435      } 
436   
437      //Normalize to 1 (not z). 
438      for (int i = 0; i < m_NumInstances; i++) 
439        for (int j = 0; j < m_NumClasses; j++) 
440          m_MislabelDistribution[i][j] = m_MislabelDistribution[i][j] / sumTmp; 
441    }

443    /** 
444     * Update the "mislabel distribution, with respect to the last base cassifier generated 
445     * and its weights (alfa and beta). It also updates the theoretically proved training error 
446     * upper bound in the asymmetric version of the algorithm. 
447     * 
448     * @param baseClassifier The classifier from we obtain the fresh new information. 
449     * @param alfa Classifier's alfa 
450     * @param beta Classifier's beta 
451     * @param u Current iteration u 
452     * 
453     * @throws java.lang.Exception If it fails. 
454     */ 
455    private void calculateMislabelDistribution(Classifier baseClassifier, final double alfa, final double beta, double u) 
456      throws Exception{ 
457   
458      double sumTmp = 0; // Faster than use Utils.normalize 
459   
460      //Faster than continously call alfaOrBeta(). 
461      final double alfaHalf = alfa / 2; 
462      final double betaHalf = beta / 2; 
463   
464      double z = 0; 
465      double sumOfWeights = 0; 
466   
467      for (int i = 0; i < m_NumInstances; i++){ 
468   
469        Instance instance = m_TrainData.instance(i); 
470        sumOfWeights += instance.weight(); 
471   
472        if (baseClassifier.classifyInstance(m_TrainData.instance(i)) == 1){ 
473          z += instance.weight() * 
474            Math.exp(m_Coloring.getCode(m_NumIterations, m_OriginalDataClasses[i]) == 0 ? alfa : -alfa); 
475          for (int j = 0; j < m_NumClasses; j++) 
476            sumTmp += 
477              (m_MislabelDistribution[i][j] = m_MislabelDistribution[i][j] * 
478              Math.exp(alfaHalf * m_Coloring.getCode(m_NumIterations, j) - 
479                       alfaHalf * m_Coloring.getCode(m_NumIterations, m_OriginalDataClasses[i]))); 
480        } 
481        else{ 
482          z += instance.weight() * 
483            Math.exp(m_Coloring.getCode(m_NumIterations, m_OriginalDataClasses[i]) == 0 ? beta : -beta); 
484          for (int j = 0; j < m_NumClasses; j++) 
485            sumTmp += 
486              (m_MislabelDistribution[i][j] = m_MislabelDistribution[i][j] * 
487              Math.exp(betaHalf * m_Coloring.getCode(m_NumIterations, j) - 
488                       betaHalf * m_Coloring.getCode(m_NumIterations, m_OriginalDataClasses[i]))); 
489        } 
490      } 
491   
492      //Deal with the various normalization options. 
493      //WARNING: is this correct? 
494      //if(m_NormFactorUsed != NOT_NORMALIZE){ 
495      //} 
496      //Mediante estas operaciones conseguimos que z sea la que sera si no se hubieran normalizado 
497      //los pesos: 
498      // Suma antes de normalizar = m_LastDistributionWeightSumBeforeNormalize; 
499      // Suma tras normalizar = sumOfWeigths 
500      // m_LastDistributionWeightSumBeforeNormalize / sumOfWeigths --> deshace la normalizacin 
501      z *= (m_LastDistributionWeightSumBeforeNormalize / sumOfWeights); 
502   
503      //Normalize to 1 (we don't use z). 
504      for (int i = 0; i < m_NumInstances; i++) 
505        for (int j = 0; j < m_NumClasses; j++) 
506          m_MislabelDistribution[i][j] = m_MislabelDistribution[i][j] / sumTmp; 
507   
508      //Update the training error upper bound. 
509      m_TrainingErrorUpperBound *= (z * u + 1 - u); 
510   
511      if (m_Debug) 
512        System.err.println("\tTEUB--> " + m_TrainingErrorUpperBound); 
513    }

515    /** 
516     * Updates de training error upper bound in the case of the symmetric version. 
517     * 
518     * @param alfa Current base classifier's alfa. 
519     * @param u Current iteration u. 
520     */ 
521    private void updateSymmetricTrainingErrorUpperBound(double alfa, double u){ 
522   
523      double error = 1 / (Math.exp(2 * alfa) + 1); 
524      double gamma = 0.5 - error; 
525      double update = u * Math.sqrt(1 - 4 * gamma * gamma) + 1 - u; 
526      m_TrainingErrorUpperBound *= update; 
527      if (m_Debug) 
528        System.err.println("\tTEUB--> " + m_TrainingErrorUpperBound); 
529    }

531    /** 
532     * Calculates the class membership probabilities for the given test instance. 
533     * 
534     * @param instance the instance to be classified 
535     * 
536     * @return predicted class probability distribution 
537     * 
538     * @exception java.lang.Exception if instance could not be classified successfully 
539     */ 
540    public double[] distributionForInstance(Instance instance) throws Exception{ 
541   
542      if (m_NumIterations == 0) 
543        throw new Exception("No model built"); 
544   
545      double[] sums = new double[m_NumClasses]; 
546      double alfaOrBeta; 
547   
548      //Necessary to avoid problems (by example, J48.classifyInstance uses numClasses() instance 
549      //method, which must return 2 and not the number of data classes in the original dataset). 
550      instance = new Instance(instance); 
551      instance.setDataset(m_TrainData); 
552   
553      for (int i = 0; i < m_NumIterations; i++){ 
554        alfaOrBeta = (((Classifier) m_Classifiers.get(i)).classifyInstance(instance) == 1 ? 
555          ((AlfaAndBetaContainer) m_AlfasAndBetas.get(i)).alfa : 
556          ((AlfaAndBetaContainer) m_AlfasAndBetas.get(i)).beta); 
557        for (int j = 0; j < m_NumClasses; j++) 
558          sums[j] += (alfaOrBeta * m_Coloring.getCode(i, j)); 
559      } 
560   
561      //Set all classes probabilities >= 1 
562      double minValue = 0; 
563   
564      for (int i = 0; i < m_NumClasses; i++) 
565        if (sums[i] < minValue) 
566          minValue = sums[i]; 
567   
568      minValue = Booster.NO_DIVISION_BY_ZERO - minValue; 
569   
570      for (int i = 0; i < m_NumClasses; i++) 
571        sums[i] += minValue; 
572   
573      //Do it directly, without using Utils.normalize(). 
574      try{ 
575        Utils.normalize(sums); 
576      } 
577      catch (IllegalArgumentException e){ 
578      } // sums[i]==0 para todo i 
579   
580      return sums; 
581    }

583    /** 
584     * Get the classifier's vote for the instance; the (vectorial) sum of all base classifiers vote vectors must be 
585     * the final combined hypothesis of the booster. 
586     * 
587     * @param instance The instance to be classified 
588     * @param classifierIndex The base classifier index 
589     * @return The vote of the base classifier to the overall classifier 
590     * @throws Exception if an error occurs 
591     */ 
592    public double[] getClassifierVote(Instance instance, int classifierIndex) throws Exception{ 
593      if (classifierIndex >= m_NumIterations || classifierIndex < 0) 
594        throw new Exception("Classifier index is invalid"); 
595   
596      double[] sums = new double[m_NumClasses]; 
597      double alfaOrBeta; 
598   
599      //Necessary to avoid problems (by example, J48.classifyInstance uses numClasses() instance 
600      //method, which must return 2 and not the number of data classes in the original dataset). 
601      instance = new Instance(instance); 
602      instance.setDataset(m_TrainData); 
603   
604   
605      alfaOrBeta = (((Classifier) m_Classifiers.get(classifierIndex)).classifyInstance(instance) == 1 ? 
606        ((AlfaAndBetaContainer) m_AlfasAndBetas.get(classifierIndex)).alfa : 
607        ((AlfaAndBetaContainer) m_AlfasAndBetas.get(classifierIndex)).beta); 
608      for (int j = 0; j < m_NumClasses; j++) 
609        sums[j] = (alfaOrBeta * m_Coloring.getCode(classifierIndex, j)); 
610   
611   
612      //Set all classes probabilities >= 1 
613      double minValue = 0; 
614   
615      for (int i = 0; i < m_NumClasses; i++) 
616        if (sums[i] < minValue) 
617          minValue = sums[i]; 
618   
619      minValue = Booster.NO_DIVISION_BY_ZERO - minValue; 
620   
621      for (int i = 0; i < m_NumClasses; i++) 
622        sums[i] += minValue; 
623   
624      return sums; 
625    }

628     * Parses a given list of options. Valid options are:</PRE> 
629     * 
630     * -A 
631     * Use symmetric version instead of asymmetric. 
632     * 
633     * Plus the rest of the superclass options. 
634     * </PRE> 
635     * 
636     * @param options the list of options as an array of strings 
637     * 
638     * @exception java.lang.Exception if an option is not supported 
639     */ 
640    public void setOptions(String[] options) throws Exception{ 
641   
642      setUseSymmetricVersion(Utils.getFlag('A', options)); 
643   
644      super.setOptions(options); 
645    }

647    /** 
648     * Gets the current settings of the Classifier. 
649     * 
650     * @return an array of strings suitable for passing to setOptions 
651     */ 
652    public String[] getOptions(){ 
653   
654      String[] otherOptions = super.getOptions(); 
655   
656      String[] options = new String[otherOptions.length + 1]; 
657   
658      int current = 0; 
659   
660      if (getUseSymmetricVersion()) 
661        options[current++] = "-A"; 
662   
663      System.arraycopy(otherOptions, 0, 
664                       options, current, 
665                       otherOptions.length); 
666   
667      current += otherOptions.length; 
668      while (current < options.length){ 
669        options[current++] = ""; 
670      } 
671      return options; 
672    }

714    /** 
715     * Get the training error upper bound. 
716     * 
717     * @return The training error upper bound or -1 if no iterations have been done. 
718     */ 
719    public double getErrorUpperBound(){ 
720      if (m_NumIterations > 0){ 
721        if (m_CalculateTrainingErrorUpperBound) 
722          return (m_NumClasses - 1) * m_TrainingErrorUpperBound; 
723        return 1; 
724      } 
725      return -1; 
726    }

771    /** 
772     * Returns description of the boosted classifier. 
773     * 
774     * @return description of the boosted classifier as a string 
775     */ 
776    public String toString(){ 
777   
778      StringBuffer text = new StringBuffer(); 
779   
780      if (m_NumIterations == 0){ 
781        text.append("AdaBoostECC: No model built yet.\n"); 
782      } 
783      else if (m_NumIterations == 1){ 
784        text.append("AdaBoostECC: No boosting possible, one classifier used!\n"); 
785        text.append(m_Classifiers.get(0).toString() + "\n"); 
786      } 
787      else{ 
788        text.append("AdaBoostECC: Base classifiers and their weights: \n\n"); 
789   
790        for (int i = 0; i < m_NumIterations; i++){ 
791          text.append("Classifier: " + i + ".\n"); 
792          try{ 
793            text.append("Classes in the subset \"1\": " + m_Coloring.getPartition(i).toString() + "\n"); 
794          } 
795          catch (Exception e){ 
796            text.append("Exception: " + e.toString() + "\n"); 
797          } 
798          text.append("\t Alfa: " + ((AlfaAndBetaContainer) m_AlfasAndBetas.get(i)).alfa); 
799          text.append("   Beta: " + ((AlfaAndBetaContainer) m_AlfasAndBetas.get(i)).beta + ".\n"); 
800          text.append(m_Classifiers.get(i).toString() + "\n\n"); 
801        } 
802        text.append("Number of performed Iterations: " 
803                    + m_NumIterations + "\n"); 
804   
805        if (m_CalculateTrainingErrorUpperBound) 
806          text.append("Training error must be less than " 
807                      + getErrorUpperBound() + "\n"); 
808      } 
809      return text.toString(); 
810    }
