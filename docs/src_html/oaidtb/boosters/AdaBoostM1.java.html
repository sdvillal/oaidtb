79     /** 
80      * Boosting method. 
81      * Reset the model and initialize data. 
82      * 
83      * @param data the training data to be used for generating the 
84      * boosted classifier. 
85      * @exception Exception if the classifier could not be built successfully 
86      */ 
87     public void buildClassifier(Instances data) throws Exception{ 
88    
89       //Check glogal settings. 
90       if (m_Classifier == null) 
91         throw new Exception("A base classifier has not been specified."); 
92    
93       if (m_UseOwnTrainData){ 
94    
95         //Check the correct format of training instances. 
96         if (data.checkForStringAttributes()) 
97           throw new Exception("Can't handle string attributes!"); 
98    
99         if (data.classAttribute().isNumeric()) 
100          throw new Exception("AdaBoostM1 can't handle a numeric class!"); 
101   
102        //We copy them thus ensuring AdaBoostM1 nor other class mess it up. 
103        m_TrainData = new Instances(data); 
104        m_BoosterReady = false;  //No "return" can be done from here. 
105        m_TrainData.deleteWithMissingClass(); 
106        if (m_TrainData.numInstances() == 0) 
107          throw new Exception("No train instances without class missing!"); 
108      } 
109      else 
110      //We entrust to the "caller class" the correctness of the train data instances. 
111        m_TrainData = data; 
112   
113      //Set up the number of classes 
114      m_NumClasses = data.numClasses(); 
115   
116      //Set up the number of instances. 
117      m_NumInstances = m_TrainData.numInstances(); 
118   
119      //Initialize instance's weight & m_NormFactor. 
120      initializeNormFactor(); 
121      initializeWeights(); 
122   
123      //Set up the number of performed iterations. 
124      m_NumIterations = 0; 
125   
126      //Initialize the classifiers and their weights array. 
127      m_Classifiers = new WeightedClassifierVector(); 
128   
129      //We can now perform new iterations. 
130      m_BoosterReady = true; 
131   
132      //Initialize the training error upper bound 
133      initializeErrorUpperBound(); 
134   
135      //Initialize the stop criterion 
136      if (!m_CustomizedBigError) 
137        m_TooBigError = defaultTooBigErrorValue(); 
138   
139      m_TooBigErrorCountDown = m_MaxNumOfTooBigErrors; 
140   
141      //Perform first "InitialIterations" iterations. 
142      nextIterations(m_InitialIterations); 
143    }

145    /** 
146     * Boosting method. Boosts any classifier that can handle weighted 
147     * instances. 
148     * 
149     * @param numIterations The number of iterations to perform. 
150     * 
151     * @exception Exception if the classifier could not be built successfully or the stop criterion is applied 
152     */ 
153    protected void buildClassifierWithWeights(int numIterations) throws Exception{ 
154   
155      double beta; 
156      double epsilon; 
157      Classifier baseClassifier; 
158      SerializedObject serializedClassifier = new SerializedObject(m_Classifier); 
159   
160      if (m_Debug) 
161        System.err.println("Boosting without resampling."); 
162   
163      // Do iterations. 
164      for (numIterations--; numIterations >= 0; numIterations--){ 
165   
166        if (m_Debug) 
167          System.err.println("Training classifier " + (m_NumIterations + 1)); 
168   
169        //Copy the base classifier. 
170        baseClassifier = (Classifier) serializedClassifier.getObject(); 
171   
172        // Build the classifier. 
173        baseClassifier.buildClassifier(m_TrainData); 
174   
175        // Calculate the error. 
176        epsilon = calculateError(baseClassifier); 
177   
178        //Apply the stop criterion 
179        if (mustStopCriterion(epsilon)){ 
180          if (m_Debug) 
181            System.err.println("/**** Exit because of stop criterion ****/ ==> error rate = " + epsilon); 
182          if (m_NumIterations == 0){ 
183            //Use the first base classifier built 
184            beta = calculateBeta(epsilon); 
185            m_Classifiers.add(baseClassifier, beta); 
186            m_NumIterations++; 
187          } 
188          //Reset, nextIterations() 
189          m_TooBigErrorCountDown = m_MaxNumOfTooBigErrors; 
190  //        return; 
191          throw new Exception("Stop criterion applied: iterate stopped"); 
192        } 
193   
194        //Determine the weight to assign to this model. 
195        beta = calculateBeta(epsilon); 
196   
197        if (m_Debug) 
198          System.err.println("\terror rate = " + epsilon + "  beta = " + beta); 
199   
200        //Assign new instances weights 
201        reweight(baseClassifier, beta); 
202   
203        //Update the theoretical training error upper bound 
204        updateErrorUpperBound(epsilon); 
205   
206        //"Commit" 
207        m_Classifiers.add(baseClassifier, beta); 
208        m_NumIterations++; 
209      } 
210    }

212    /** 
213     * Boosting method. Boosts any classifier using training instances 
214     * weight for resampling. 
215     * 
216     * @param numIterations The number of iterations to perform. 
217     * 
218     * @exception Exception if the classifier could not be built successfully or the stop criterion is applied 
219     */ 
220    protected void buildClassifierUsingResampling(int numIterations) throws Exception{ 
221   
222      double epsilon, beta;          //Base classifier's error and weight. 
223      Random randomInstance = new Random(m_ResampleSeed); //Random number generator for resample. 
224      Instances sample;              //The resampled training dataset. 
225      Classifier baseClassifier; 
226      SerializedObject serializedClassifier = new SerializedObject(m_Classifier); 
227   
228      if (m_Debug) 
229        System.err.println("Boosting with resampling."); 
230   
231      // Do iterations 
232      for (numIterations--; numIterations >= 0; numIterations--){ 
233   
234        if (m_Debug){ 
235          System.err.println("Training classifier " + (m_NumIterations + 1)); 
236        } 
237   
238        // Resample. 
239        sample = m_TrainData.resampleWithWeights(randomInstance); 
240   
241        //Copy the base classifier. 
242        baseClassifier = (Classifier) serializedClassifier.getObject(); 
243   
244        // Build the classifier. 
245        baseClassifier.buildClassifier(sample); 
246   
247        //Calculate base classifier's error 
248        epsilon = calculateError(baseClassifier); 
249   
250        //Apply the stop criterion 
251        if (mustStopCriterion(epsilon)){ 
252          if (m_Debug) 
253            System.err.println("/**** Exit because of stop criterion ****/ ==> error rate = " + epsilon); 
254          if (m_NumIterations == 0){ 
255            beta = calculateBeta(epsilon); 
256            m_Classifiers.add(baseClassifier, beta); 
257            m_NumIterations++; 
258          } 
259          //Reset, nextIterations() 
260          m_TooBigErrorCountDown = m_MaxNumOfTooBigErrors; 
261  //        return; 
262          throw new Exception("Stop criterion applied: iterate stopped"); 
263        } 
264   
265        //Determine the weight to assign to this model. 
266        beta = calculateBeta(epsilon); 
267   
268        if (m_Debug) 
269          System.err.println("\terror rate = " + epsilon + "  beta = " + beta); 
270   
271        //Assign new instances weights 
272        reweight(baseClassifier, beta); 
273   
274        //Update the theoretical training error upper bound. 
275        updateErrorUpperBound(epsilon); 
276   
277        //"Commit" 
278        m_Classifiers.add(baseClassifier, beta); 
279        m_NumIterations++; 
280      } 
281    }

283    /** 
284     * Calculate a classifiers error as the weight percentage of training instances misclassified. 
285     * 
286     * @param baseClassifier The base classifier 
287     * 
288     * @return The error in [0,1] 
289     * 
290     * @throws Exception If there is a classification error 
291     */ 
292    private double calculateError(Classifier baseClassifier) throws Exception{ 
293   
294      double epsilon = 0; 
295      double totalSumOfWeights = 0; 
296      Instance instance; 
297   
298      for (int i = 0; i < m_NumInstances; i++){ 
299        instance = m_TrainData.instance(i); 
300        double weight = instance.weight(); 
301        totalSumOfWeights += weight; 
302        if (baseClassifier.classifyInstance(instance) != instance.classValue()) 
303          epsilon += weight; 
304      } 
305   
306      return epsilon / totalSumOfWeights; 
307    }

309    /** 
310     * Calculate the classifier's vote weight in the final combined hypothesis based on its error 
311     * 
312     * @param error The training error of the base classifier. 
313     * 
314     * @return The classifier's vote weight in the final combined hypothesis 
315     */ 
316    protected double calculateBeta(double error){ 
317      return Math.log((1 - error + NO_DIVISION_BY_ZERO) / (error + NO_DIVISION_BY_ZERO)); 
318    }

320    /** 
321     * Assign new weight to every instance in the training dataset based 
322     * in the classifier's vote weight and if it classifies correctly the instance. 
323     * 
324     * We use a "conservative boosting approach", only increasing the weights of misclassified 
325     * instances. This update will be exp(beta) if beta is > 0 or exp (-beta) otherwise. 
326     * 
327     * @param baseClassifier The base classifier 
328     * @param beta Factor in term of which misclassified training instances weight will be updated 
329     * 
330     * @throws Exception If there is a classification error 
331     */ 
332    protected void reweight(Classifier baseClassifier, double beta) throws Exception{ 
333   
334      double newWeightsSum = 0; 
335   
336      if (beta > 0)  //error < 0.5 
337        beta = Math.exp(beta); 
338      else  //error > 0.5; we still increase the weights of misclassified instances 
339        beta = Math.exp(-beta); 
340   
341      for (int i = 0; i < m_NumInstances; i++){ 
342   
343        Instance instance = m_TrainData.instance(i); 
344        double weight = instance.weight(); 
345   
346        if (instance.classValue() != baseClassifier.classifyInstance(instance)) 
347          instance.setWeight(weight * beta); 
348   
349        newWeightsSum += weight; 
350      } 
351   
352      normalizeWeights(newWeightsSum); 
353    }

355    /** 
356     * Return the class predicted (<0 == class 0 and >0 == class 1) and the 
357     * confidence of this prediction (its absolute value). 
358     * 
359     * @param instance The instance to be classified 
360     * 
361     * @return The class predicted and its confidence. 
362     * 
363     * @throws Exception if the instance can´t be classified succesfully. 
364     */ 
365    public double confidenceAndSign(Instance instance) throws Exception{ 
366   
367      double confidenceAndSign = 0; 
368   
369      for (int i = 0; i < m_NumIterations; i++) 
370        if(m_Classifiers.get(i).classifyInstance(instance) == 0) 
371          confidenceAndSign -= m_Classifiers.getWeight(i); 
372        else 
373          confidenceAndSign += m_Classifiers.getWeight(i); 
374   
375      return confidenceAndSign; 
376    }

379     * Return the class predicted (<0 == class 0 and >0 == class 1) and the 
380     * confidence of this prediction (its absolute value) of the classifier indicated. 
381     * 
382     * @param instance The instance to be classified 
383     * @param classifierIndex The classifier's index 
384     * 
385     * @return The class predicted and its confidence. 
386     *

388     */ 
389    public double confidenceAndSign(Instance instance, int classifierIndex) throws Exception{ 
390   
391      if (classifierIndex >= m_NumIterations || classifierIndex < 0) 
392        throw new Exception("Classifier index is invalid"); 
393   
394        if(m_Classifiers.get(classifierIndex).classifyInstance(instance) == 0) 
395          return - m_Classifiers.getWeight(classifierIndex); 
396        else 
397          return m_Classifiers.getWeight(classifierIndex);

503     * If this method return true, the boosting method must stop. 
504     * 
505     * @param error The error of the base classifier in the actual boosting iteration 
506     * 
507     * @return True if we have reached the stop condition, false otherwise.

509    protected boolean mustStopCriterion(double error){ 
510      if (error > m_TooBigError){ 
511        m_TooBigErrorCountDown--; 
512        if (m_TooBigErrorCountDown == 0) 
513          return true; 
514        return false; 
515      } 
516      else{ 
517        if (m_TooBigErrorsMustBeConsecutives) 
518          m_TooBigErrorCountDown = m_MaxNumOfTooBigErrors; 
519        return false; 
520      } 
521    }

569    /** 
570     * Update the theoretical training error upper bound.<PRE> 
571     * 
572     *  - Note that if the error of the base classifier is greater than 1/2 the bound is undetermined, 
573     *    and so its calculation will be aborted. 
574     * 
575     *  - Note that this is related to the current choice of the classifier's vote weight, so it will 
576     *    be invalid if this class is extended playing with its formulae (like in AdaBoostM1W); 
577     *    so the methods related to the training error must be overriden (with a silent noop if no bound 
578     *    must be calculated and returning 1 the method getErrorBound()). 
579     * </PRE> 
580     * 
581     * @param error The current base classifier's error 
582     */ 
583    protected void updateErrorUpperBound(double error){ 
584      if (m_CalculateErrorUpperBound && !Double.isNaN(m_ErrorUpperBound)){ 
585        if (error > 0.5){ 
586          m_ErrorUpperBound = Double.NaN; 
587          if (m_Debug) 
588            System.err.println("Can't bound the training error upper boud: base classifier's error is greater than 1/2."); 
589        } 
590        else{ 
591          m_ErrorUpperBound *= 2 * Math.sqrt(error * (1 - error)); 
592          if (m_Debug) 
593            System.err.println("\tTEUB--> " + m_ErrorUpperBound); 
594        } 
595      } 
596    }

640    /** 
641     * Parses a given list of options. Valid options are:<p> <PRE> 
642     * 
643     * -B 
644     * Calculate the training error upper bound (default false). 
645     * 
646     * -E bigErrorValue 
647     * Set the value considered as "too big error" in the (base classifier) training procedure; it will 
648     * be used as an stop criterion. 
649     *  - If less or equal than zero, it will be reset to the default value. 
650     *  - If bigger than 1.0, no stop criterion will be applied. 
651     * 
652     * -C maxNumOfBigErrors 
653     * Set the number of times after that, happened a value considered as "too big error" in the training procedure, 
654     * the classifier will stop to iterate. 
655     *  - If less or equal than zero, no stop criterion will be applied. 
656     *  - Default: 1 
657     * 
658     * -V 
659     * The "big errors occurrences" mustn't need to be consecutives to provoke the stop. 
660     * 
661     * Plus the rest of the superclass options. 
662     * </PRE> 
663     * 
664     * @param options the list of options as an array of strings 
665     * 
666     * @exception Exception if an option is not supported 
667     */ 
668    public void setOptions(String[] options) throws Exception{ 
669   
670      super.setOptions(options); 
671   
672      setCalculateErrorUpperBound(Utils.getFlag('B', options)); 
673   
674      String bigError = Utils.getOption('E', options); 
675      if (bigError.length() != 0) 
676        setTooBigError(Double.parseDouble(bigError)); 
677   
678      String bigErrorCountDown = Utils.getOption('C', options); 
679      if (bigErrorCountDown.length() != 0) 
680        setMaxNumOfTooBigErrors(Integer.parseInt(bigErrorCountDown)); 
681   
682      setTooBigErrorsMustBeConsecutives(!Utils.getFlag('V', options)); 
683    }

685    /** 
686     * Gets the current settings of the Classifier. 
687     * 
688     * @return an array of strings suitable for passing to setOptions 
689     */ 
690    public String[] getOptions(){ 
691   
692      String[] otherOptions = super.getOptions(); 
693   
694      String[] options = new String[otherOptions.length + 6]; 
695   
696      int current = 0; 
697   
698      if (getCalculateErrorUpperBound()) 
699        options[current++] = "-B"; 
700   
701      if (m_CustomizedBigError){ 
702        options[current++] = "-E"; 
703        options[current++] = "" + getTooBigError(); 
704      } 
705   
706      options[current++] = "-C"; 
707      options[current++] = "" + getMaxNumOfTooBigErrors(); 
708   
709      if (!getTooBigErrorsMustBeConsecutives()) 
710        options[current++] = "-V"; 
711   
712      System.arraycopy(otherOptions, 0, 
713                       options, current, 
714                       otherOptions.length); 
715   
716      current += otherOptions.length; 
717      while (current < options.length){ 
718        options[current++] = ""; 
719      } 
720      return options; 
721    }

765    /** 
766     * Returns description of the boosted classifier. 
767     * 
768     * @return description of the boosted classifier as a string 
769     */ 
770    public String toString(){ 
771   
772      StringBuffer text = new StringBuffer(); 
773   
774      if (m_NumIterations == 0){ 
775        text.append("Booster: No model built yet.\n"); 
776      } 
777      else if (m_NumIterations == 1){ 
778        text.append("Booster: No boosting possible, one classifier used!\n"); 
779        text.append(m_Classifiers.get(0).toString() + "\n"); 
780      } 
781      else{ 
782        text.append("AdaBoostM1: Base classifiers and their weights: \n\n"); 
783        for (int i = 0; i < m_NumIterations; i++){ 
784          text.append(m_Classifiers.get(i).toString() + "\n"); 
785          text.append("**Weight: " + m_Classifiers.getWeight(i) + "\n\n"); 
786        } 
787        text.append("Number of performed Iterations: " 
788                    + m_NumIterations + "\n"); 
789   
790        if (m_CalculateErrorUpperBound) 
791          text.append("Training error upper bound: " 
792                      + getErrorUpperBound() + "\n"); 
793      } 
794   
795      return text.toString(); 
796    }
