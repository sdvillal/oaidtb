144    /** 
145     * Map the x coordinate of a point in a region to the x coordinate of the 
146     * same point as if the region is scaled and translated to the current drawing panel. 
147     * 
148     * Por ejemplo, x(región) = 3.5 ==> x(pantalla) = 9 
149     * 
150     * @param x An x coordinate of a point in the region 
151     * @return The corresponding x coordinate in the screen 
152     */ 
153    public double getXInPanel(double x){ 
154      x -= m_RegionX; //Translate 
155      return x * m_xScaleFactor; //Scale 
156    }

158    /** 
159     * Map the y coordinate of a point in a region to the y coordinate of the 
160     * same point as if the region is scaled and translated to the current drawing panel. 
161     * 
162     * Por ejemplo, y(región) = 3.5 ==> y(pantalla) = 9 
163     * 
164     * @param y An y coordinate of a point in the region 
165     * @return The corresponding y coordinate in the screen 
166     */ 
167    public double getYInPanel(double y){ 
168      y -= m_RegionY; //Translate 
169      return y * m_yScaleFactor; //Scale 
170    }

172    /** 
173     * Map the x coordinate of a point in the panel to the x coordinate of the 
174     * same point as if the panel is scaled and translated to the current region selected. 
175     * 
176     * Por ejemplo, x(pantalla) = 9 ==>  x(región) = 3.5 
177     * 
178     * @param x An x coordinate of a point in the screen 
179     * @return The corresponding x coordinate in the region 
180     */ 
181    public double getXFromPanel(double x){ 
182      return (x + m_RegionX * m_xScaleFactor) / m_xScaleFactor; 
183    }

186     * Map the y coordinate of a point in the panel to the y coordinate of the

192     * @return The corresponding y coordinate in the region

199     * Constructor

203    public DrawingPanel(Point2DInstances pointsDataSet){ 
204   
205      m_Points = pointsDataSet; 
206   
207      //Tenemos que controlar los cambios de tamaño del componente 
208      addComponentListener(new ComponentAdapter(){ 
209        public void componentResized(ComponentEvent e){ 
210          //La imagen que representa a las instancias siempre tendrá el máximo tamaño 
211          //que ha llegado a alcanzar el panel en la pantalla 
212          final int imgW = m_InstancesImage.getWidth(); 
213          final int imgH = m_InstancesImage.getHeight(); 
214          //Máxima anchura que se ha alcanzado hasta ahora 
215          final int w = Math.max(imgW, getWidth()); 
216          //Máxima altura que se ha alcanzado hasta ahora 
217          final int h = Math.max(imgH, getHeight()); 
218          //Si el nuevo tamaño del panel en cualquiera de las dos dimensiones 
219          //es mayor que cualquiera alcanzado hasta ahora 
220          if (w != imgW || h != imgH){ 
221            //Aumentamos el tamaño de la imagen 
222            final BufferedImage bi = new BufferedImage(w, h, BufferedImage.TYPE_INT_ARGB); 
223            final Graphics2D g = bi.createGraphics(); 
224            //Rellenamos la imagen con un color transparente 
225            g.setColor(m_TransparentColor); 
226            g.fillRect(0, 0, w, h); 
227            //Pintamos la imagen anterior 
228            g.drawImage(m_InstancesImage, 0, 0, imgW, imgH, null); 
229            m_InstancesImage = bi; 
230            g.dispose(); 
231          } 
232          if(m_IsZoomedIn){ 
233            /* 
234            Teniemdo en mente la siguiente relación, que debe cumplirse siempre: 
235              m_xScaleFactor * m_regionWidth = getWidth() 
236            es necesario cambiar el tamaño de la región sobre la que se ha hecho zoom 
237            para preservar la lógica del programa (por ejemplo, para que la imagen de hipótesis 
238            recreada no salga "movida" o para que realmente aparezcan todas las 
239            instancias en la región) 
240            */ 
241            m_RegionWidth = (double) getWidth() / m_xScaleFactor; 
242            m_RegionHeight = (double) getHeight() / m_yScaleFactor; 
243            m_UpdateInstances = true; 
244            repaint();  //Mete un "chasquido visual" bastante desagradable, pero no hay otra... 
245          } 
246        } 
247      }); 
248    }

445    /** 
446     * Este es el método principal de la clase; en él se encuentra 
447     * toda la lógica de representación 
448     * 
449     * @param g Un objeto graphics en que se pintará el panel 
450     */ 
451    public void paint(Graphics g){ 
452   
453      //Llamamos al método de la superclase, para luego pintar encima lo que sea necesario 
454      super.paint(g); 
455   
456      //Pintamos la imagen de hipótesis (abajo del todo) 
457      if (m_ShowHypothesisImage && m_HypothesisImage != null) 
458        g.drawImage(m_HypothesisImage, 0, 0, m_HypothesisImage.getWidth(this), m_HypothesisImage.getHeight(this), this); 
459   
460      //Pintamos la imajen de la rejilla (encima de la imagen de hipótesis) 
461      if (m_ShowGrid){ 
462        //Si es necesario la recreamos 
463        if (m_UpdateGrid) 
464          drawGrid(); 
465        if (m_GridImage != null) 
466          g.drawImage(m_GridImage, 0, 0, m_GridImage.getWidth(), m_GridImage.getHeight(), this); 
467      } 
468   
469      //Si tenemos que repintar las instancias 
470      if (m_UpdateInstances){ 
471        //Este puede ser un trabajo costoso, avisamos a los mediadores 
472        notifyMediators(INSTANCES_REPAINT_BEGIN, m_Points); 
473        if (m_IsZoomedIn) 
474          //Elegimos el método más adecuado (y rápido) para pintar las instancias 
475          if (!m_Points.isTrainAndTestGeneratedFromregion()) 
476            drawRegionInstances(m_RegionX, m_RegionY, m_RegionWidth, m_RegionHeight); 
477          else 
478            drawRegionTrainAndTestInstances(); 
479        else if (m_Points.existsTestSet()) 
480          drawTrainAndTestInstances(); 
481        else 
482          drawAllInstances(); 
483        notifyMediators(INSTANCES_REPAINT_FINISH, m_Points); 
484        //Ya no tenemos que hacerlo 
485        m_UpdateInstances = false; 
486      } 
487   
488      //Si procede, pintamos en la pantalla la imagen de instancias (encima del resto de imágenes) 
489      if ((m_ShowTrainInstances || m_ShowTestInstances && m_Points.existsTestSet()) && m_InstancesImage != null) 
490        g.drawImage(m_InstancesImage, 0, 0, m_InstancesImage.getWidth(this), m_InstancesImage.getHeight(this), this); 
491   
492      //Pintamos el "rectángulo transparente" que representa la selección actual 
493      if (m_RegionSelector.existsSelection) 
494        m_RegionSelector.drawSelection((Graphics2D) g); 
495    }

497    /** Dibujar la rejilla en la imagen de rejilla */ 
498    private void drawGrid(){ 
499   
500      //No valid grid has been set up 
501      if (m_GridWidth <= 0 || m_GridHeight <= 0 || m_GridRowHeight <= 0 || m_GridColumnWidth <= 0){ 
502        m_GridImage = null; 
503        m_UpdateGrid = false; 
504        return; 
505      } 
506   
507      //Creamos la imagen 
508      m_GridImage = new BufferedImage(m_GridWidth, m_GridHeight, BufferedImage.TYPE_INT_ARGB); 
509   
510      //La rellenamos con un color transparente 
511      Graphics2D g = m_GridImage.createGraphics(); 
512      g.setColor(m_TransparentColor); 
513      g.fillRect(0, 0, m_GridWidth, m_GridHeight); 
514   
515      //Pintamos las líneas de color negro 
516      g.setColor(Color.black); 
517   
518      //Poner un stroke güeno 
519   
520      //Vertical lines 
521      int coord = 0; 
522      while (coord < m_GridWidth){ 
523        g.drawLine(coord, 0, coord, m_GridHeight); 
524        coord += m_GridColumnWidth; 
525      } 
526   
527      //Horizontal lines 
528      coord = 0; 
529      while (coord < m_GridHeight){ 
530        g.drawLine(0, coord, m_GridWidth, coord); 
531        coord += m_GridRowHeight; 
532      } 
533   
534      //Ya la hemos actualizado 
535      m_UpdateGrid = false; 
536    }

557    /** 
558     * Dibujar las últimas instancias añadidas a la base de datos en la imagen de 
559     * instancias; método pensado para dibujar de golpe y rápidamente un conjunto de instancias 
560     * del mismo color añadidas secuencialmente; no se llama a repaint(), así 
561     * que si se quiere actualizar también el contenido de la 
562     * pantalla se debe llamar a ese método externamente. 
563     * 
564     * @param num El número de instancias a pintar 
565     * @param color El color de dichas instancias 
566     */ 
567    public void drawLastInstances(int num, final Color color){ 
568      //Seleccionamos todas las instancias 
569      final FastVector vector = m_Points.getAllInstancesFastVector(); 
570      final int numInstances = vector.size(); 
571      //Las pintamos en la imagen de instancias 
572      Graphics2D g2 = m_InstancesImage.createGraphics(); 
573      g2.setColor(color); 
574      while (num > 0){ 
575        Instance instance = (Instance) vector.elementAt(numInstances - num); 
576        g2.setColor(color); 
577        g2.fillOval((int) instance.value(0), (int) instance.value(1), m_PointsRadius, m_PointsRadius); 
578        num--; 
579      } 
580      g2.dispose(); 
581    }

583    /** 
584     * Dibuja un conjunto de instancias de un determinado color en la imagen de hipótesis; 
585     * no se llama a repaint(), así que si se quiere actualizar también el contenido de la 
586     * pantalla se debe llamar a ese método externamente. 
587     * 
588     * @param instances Las instancias a pintar 
589     * @param color El color de dichas instancias 
590     */ 
591    public void drawInstances(FastVector instances, final Color color){ 
592      Graphics2D g2 = m_InstancesImage.createGraphics(); 
593      g2.setColor(color); 
594      //Dibujamos todas las instancias en la imagen de instancias 
595      for (int i = 0; i < instances.size(); i++){ 
596        Instance instance = (Instance) instances.elementAt(i); 
597        g2.setColor(color); 
598        g2.fillOval((int) instance.value(0), (int) instance.value(1), m_PointsRadius, m_PointsRadius); 
599      } 
600      g2.dispose(); 
601    }

622     * Pinta todas las intancias que existan en la base de datos en la imagen de instancias

624    private void drawAllInstances(){ 
625   
626      //Si no hay que pintarlas, salimos sin hacer nada 
627      if (!m_ShowTrainInstances) 
628        return; 
629   
630      //Recogemos todas las instancias ordenadas por color 
631      Iterator iterator = m_Points.getAllInstancesIterator(); 
632   
633      Graphics2D g = m_InstancesImage.createGraphics(); 
634   
635      //"Limpiamos" la imagen de instancias 
636      g.setColor(m_TransparentColor); 
637      g.setComposite(AlphaComposite.Src); 
638      g.fillRect(0, 0, m_InstancesImage.getWidth(), m_InstancesImage.getHeight()); 
639   
640      //Nos aprovechamos de que están ordenadas por color 
641      int colorIndex = -1; 
642      while (iterator.hasNext()){ 
643        colorIndex++; 
644        //Preguntamos por cuántas instancias hay de ese color 
645        int countDown = m_Points.getNumPointsOfColor(colorIndex); 
646        g.setColor(m_Points.getColor(colorIndex)); 
647        while (countDown-- > 0){ 
648          Point2DInstances.Point2DInstance p = (Point2DInstances.Point2DInstance) iterator.next(); 
649          g.fillOval((int) p.getX(), (int) p.getY(), m_PointsRadius, m_PointsRadius); 
650        } 
651      } 
652   
653      g.dispose(); 
654    }

691                         m_PointsRadius, m_PointsRadius); 
692            } 
693          } 
694        }

705                       m_PointsRadius, m_PointsRadius); 
706          } 
707        } 
708      } 
709      else{ //Si tenemos que pintar todas las instancias en la región 
710   
711        //Recuperamos todas las instancias que están en la región 
712        FastVector regionInstances = m_Points.getRegion(regionX, regionY, regionWidth, regionHeight);

717            int colorIndex = (int) p.getColor();

719            //Hacemos la transformación de coordenadas correspondiente 
720            g.fillOval((int) Math.round(getXInPanel(p.getX())), (int) Math.round(getYInPanel(p.getY())), 
721                       m_PointsRadius, m_PointsRadius); 
722          } 
723      } 
724   
725      g.dispose(); 
726    }

817    /** Permite seleccionar gráficamente una región dentro del panel. */ 
818    private class RegionSelector extends MouseAdapter implements MouseMotionListener{ 
819   
820      int origX,origY; 
821      //Siempre en coordenadas absolutas de la pantalla 
822      int regionX, regionY, regionWidth, regionHeight; 
823      Color selectionColor = Color.blue.darker(); 
824      boolean existsSelection = false; 
825   
826      //Los tamaños mínimos de una selección 
827      //TODO: Permitir que sea configurable 
828      double minWidth = 0.02, minHeight = 0.02; 
829   
830      /** Recogemos las coordenada donde hemos pulsado el ratón */ 
831      public void mousePressed(MouseEvent e){ 
832        origX = e.getX(); 
833        origY = e.getY(); 
834      } 
835   
836      /** Cancelar la selección */ 
837      public void mouseClicked(MouseEvent e){ 
838        if (existsSelection) 
839          resetSelection(); 
840      } 
841   
842      /** 
843       * Controlamos si hemos realizado una selección (arrastre del ratón con el botón izquierdo) 
844       * 
845       * @param e El evento del ratón 
846       */ 
847      public void mouseReleased(MouseEvent e){ 
848        if (SwingUtilities.isRightMouseButton(e)) //Canceled selection process 
849          resetSelection(); 
850        else if (existsSelection){ 
851          Rectangle2D.Double sel = getLastSelection(); 
852          //Evitamos la selección de una región muy pequeña 
853          if (sel.width < minWidth || sel.height < minHeight) 
854            resetSelection(); 
855          else 
856            notifyMediators(REGION_SELECTED, getLastSelection()); 
857        } 
858      } 
859   
860      /** 
861       * Actualizar la selección actual 
862       * 
863       * @param e El evento del ratón 
864       */ 
865      public void mouseDragged(MouseEvent e){ 
866   
867        existsSelection = true; 
868   
869        //No permitimos que la selección se "salga del panel" 
870        final int w = getWidth(); 
871        final int x = e.getX(); 
872   
873        if(x < 0) 
874          regionWidth = 0; 
875        else if(x > w) 
876          regionWidth = w; 
877        else 
878          regionWidth = x; 
879   
880        //No permitimos que la selección se "salga del panel" 
881        final int h = getHeight(); 
882        final int y = e.getY(); 
883   
884        if(y < 0) 
885          regionHeight = 0; 
886        else if(y > h) 
887          regionHeight = h; 
888        else 
889          regionHeight = y; 
890   
891        //Actualizamos los valores de la región seleccionada 
892        regionX = Math.min(origX, regionWidth); 
893        regionY = Math.min(origY, regionHeight); 
894        regionWidth = Math.abs(origX - regionWidth); 
895        regionHeight = Math.abs(origY - regionHeight); 
896   
897        //Repintamos el "rectángulo transparente" que representa la selección 
898        repaint(); 
899      } 
900   
901      /** @return La última región seleccionada, en coordenadas del dominio del problema  */ 
902      public Rectangle2D.Double getLastSelection(){ 
903        if (!m_IsZoomedIn) 
904          return new Rectangle2D.Double(regionX, regionY, regionWidth, regionHeight); 
905        else{ 
906          final double trueX = getXFromPanel(regionX); 
907          final double trueY = getYFromPanel(regionY); 
908          final double trueWidth = regionWidth / m_xScaleFactor; 
909          final double trueHeight = regionHeight / m_yScaleFactor; 
910          return new Rectangle2D.Double(trueX, trueY, trueWidth, trueHeight); 
911        } 
912      } 
913   
914   
915      public void mouseMoved(MouseEvent e){ 
916      } 
917   
918      /** Dibuja el "rectángulo transparente" que representa la región seleccionada */ 
919      public void drawSelection(final Graphics2D g2){ 
920        g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER, 0.2F)); 
921        g2.setColor(selectionColor); 
922        g2.fillRect(regionX, regionY, regionWidth, regionHeight); 
923      } 
924   
925      /** Borra el "rectángulo transparente" que representa la región seleccionada */ 
926      private void resetSelection(){ 
927        repaint(1, regionX, regionY, regionWidth, regionHeight); 
928        existsSelection = false; 
929        regionX = regionY = regionWidth = regionHeight = 0; 
930      } 
931    }

970   
971      //Tenemos que redibujar 
972      m_UpdateInstances = true; 
973   
974      notifyMediators(ZOOM_IN, new Rectangle2D.Double(regionX, regionY, regionWidth, regionHeight)); 
975    }

987      m_RegionY = 0; 
988      m_RegionWidth = -1;

995      m_UpdateInstances = true;
