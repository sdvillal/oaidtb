151      //Configuramos el aspecto de los mensajes redirigidos desde System.err 
152      SimpleAttributeSet sas = new SimpleAttributeSet(); 
153      StyleConstants.setForeground(sas, Color.blue); 
154      StyleConstants.setItalic(sas, true); 
155      StyleConstants.setBold(sas, true); 
156      //Configurar el thread que se encargará de la redirección */ 
157      m_ErrRedirector = new SystemOutputStreamRedirector(m_Log.getLogText(), sas, true, m_SystemErrFork.isSelected()); 
158   
159      // Iniciar/parar la redirección de mensajes 
160      m_LogSystemErr.addItemListener(new ItemListener(){ 
161        private Thread t; 
162   
163        public void itemStateChanged(ItemEvent e){ 
164          if (m_LogSystemErr.isSelected()){ 
165            t = new Thread(m_ErrRedirector); 
166            t.setPriority(m_ErrRedirector.getPriority()); 
167            t.setDaemon(m_ErrRedirector.isDaemon()); 
168            t.start(); 
169            m_SystemErrFork.setEnabled(true); 
170          } 
171          else{ 
172            t.interrupt(); 
173            m_SystemErrFork.setEnabled(false); 
174          } 
175        } 
176      }); 
177      //Configuración inicial de la redirección 
178      if (m_LogSystemErr.isSelected()) 
179        m_LogSystemErr.setSelected(true); 
180   
181      //Configurar el fork entre System.err y el panel de log 
182      m_SystemErrFork.addItemListener(new ItemListener(){ 
183        public void itemStateChanged(ItemEvent e){ 
184          m_ErrRedirector.setFork(m_SystemErrFork.isSelected()); 
185        } 
186      }); 
187      //Configuración inicial de la redirección 
188      m_ErrRedirector.setFork(m_SystemErrFork.isSelected()); 
189   
190      //Configuración de las etiquetas que muestran las coordenadas reales 
191      //que representa el punto del panel de bibujado sobre el que se encuentra 
192      //el puntero del ratón 
193      m_MouseLocationLabelX.setEditable(false); 
194      m_MouseLocationLabelX.setForeground(Color.blue); 
195      m_MouseLocationLabelX.setBackground(Color.orange); 
196      m_MouseLocationLabelY.setEditable(false); 
197      m_MouseLocationLabelY.setBackground(Color.orange); 
198      m_MouseLocationLabelY.setForeground(Color.blue); 
199      m_DrawingPanel.addMouseMotionListener(new MouseMotionAdapter(){ 
200        public void mouseMoved(MouseEvent e){ 
201          if (!m_DrawingPanel.isZoomedIn()){ 
202            m_MouseLocationLabelX.setText("x= " + e.getX()); 
203            m_MouseLocationLabelY.setText("y= " + e.getY()); 
204          } 
205          else{ 
206            //Obtenemos las coordenadas "reales" del pubto en el dominio del problema 
207            m_MouseLocationLabelX.setText("x= " + Utils.doubleToString(m_DrawingPanel.getXFromPanel(e.getX()), 4, 4)); 
208            m_MouseLocationLabelY.setText("y= " + Utils.doubleToString(m_DrawingPanel.getYFromPanel(e.getY()), 4, 4)); 
209          } 
210        } 
211   
212        public void mouseDragged(MouseEvent e){ 
213          mouseMoved(e); 
214        } 
215      }); 
216   
217      //Añadimos las etiquetas de coordenadas a la parte izquerda de la barra de estado en el panel de log 
218      m_MouseLocationLabelX.setText("x="); 
219      m_MouseLocationLabelY.setText("y="); 
220      JPanel locPanel = new JPanel(new GridLayout(2, 1)); 
221      locPanel.add(m_MouseLocationLabelX); 
222      locPanel.add(m_MouseLocationLabelY); 
223      locPanel.validate(); 
224      locPanel.setBorder(BorderFactory.createLineBorder(Color.green)); 
225      m_Log.addCustomPanel(locPanel); 
226    }

401      /** 
402       * Listener que minimiza / restaura alternativamente un {@link JInternalFrame} 
403       */ 
404      class ShowAndHideFrame implements ActionListener{ 
405   
406        /** El frame a minimizar / restaurar cuando la acción es invocada */ 
407        final JInternalFrame frame; 
408   
409        /** 
410         * Constructor 
411         * 
412         * @param frame El frame a minimizar / restaurar cuando la acción es invocada 
413         */ 
414        public ShowAndHideFrame(JInternalFrame frame){ 
415          this.frame = frame; 
416        } 
417   
418        /** 
419         * Minimiza o restaura el frame 
420         * 
421         * @param e El evento que ha provocado esta llamada a la acción 
422         */ 
423        public void actionPerformed(ActionEvent e){ 
424          try{ 
425            frame.setIcon(!frame.isIcon()); 
426          } 
427          catch (PropertyVetoException ex){ 
428            ex.printStackTrace(); 
429          } 
430        } 
431      } 
432

434      JButton showAndHideIEFrame = new JButton("I.E."); 
435      showAndHideIEFrame.addActionListener(new ShowAndHideFrame(m_InstanceEditingFrame));

437      showAndHideIEFrame.setToolTipText("Mostrar / Ocultar la ventana de edición de instancias" + " (Alt-Izquierda)");

855    /** 
856     * Clase que sirve para llevar el historial de zoom; como consume 
857     * mucha memoria debido a que, en el peor de los casos, 
858     * tiene que almacenar cada vez 2 imágenes, he decidido limitar el historial de zoom a la 
859     * posibilidad de volver sólo a volver a ver todo a escala natural. Otra opción 
860     * más rápida y menos costosa en términos de memoria es hacer zoom símplemente 
861     * sobre lo que se ve en ese momento. 
862     */ 
863    private static class ZoomHistory{ 
864   
865      BufferedImage m_HypothesisSumImage; 
866      BufferedImage m_HypothesisImage; 
867      Rectangle m_GridDimensions; 
868      Rectangle2D.Double m_Region; 
869    }

871    /** Para almacenar el historial de zoom (por ahora, sólo la información de la vista a escala natural 
872     *  ZoomHistory 
873     */ 
874    private ArrayList m_ZoomHistory = new ArrayList(); 
875   
876    /** 
877     * Hace zoom sobre la región especificada. Coordina todos los paneles para que la información sea 
878     * coherente y hace una copia de seguridad en caso de que sea el primer zoom que hacemos desde 
879     * la escala natural. 
880     * 
881     * @param r La región sobre la que hacer zoom, expresada en coordenadas reales del dominio del problema 
882     */ 
883    private void doZoomIn(final Rectangle2D.Double r){ 
884   
885      // Para guardar las coordenadas naturales (panatalla)  que corresponden a las coordenadas 
886      // reales en el dominio del problema pasadas por parámetro 
887      final Rectangle2D.Double r2; 
888   
889      if (!m_DrawingPanel.isZoomedIn()){ 
890        //Guardamos la información para poder volver a restaurar 
891        ZoomHistory naturalScale = new ZoomHistory(); 
892        naturalScale.m_HypothesisSumImage = m_VisualOptionsPanel.getHypothesisSumImage(); 
893        naturalScale.m_HypothesisImage = m_VisualOptionsPanel.getHypothesisImage(); 
894        naturalScale.m_GridDimensions = new Rectangle(m_DrawingPanel.getGridColumnWidth(), 
895                                                      m_DrawingPanel.getGridRowHeight(), 
896                                                      m_DrawingPanel.getGridWidth(), 
897                                                      m_DrawingPanel.getGridHeight()); 
898        naturalScale.m_Region = new Rectangle2D.Double(0, 0, m_DrawingPanel.getWidth(), m_DrawingPanel.getHeight()); 
899        m_ZoomHistory.add(naturalScale); 
900        //Las coordenadas reales coinciden con las naturales 
901        r2 = r; 
902      } 
903      else 
904      //Calculamos las coordenadas naturales (de pantalla) que se corresponden con estas coordenadas 
905      //reales en el dominio del problema pasadas por parámetro 
906        r2 = new Rectangle2D.Double(m_DrawingPanel.getXInPanel(r.x), 
907                                    m_DrawingPanel.getYInPanel(r.y), 
908                                    r.width * m_DrawingPanel.getXScaleFactor(), 
909                                    r.height * m_DrawingPanel.getYScaleFactor()); 
910   
911      //Le decimos al panel ahora tiene que representar la nueva región 
912      m_DrawingPanel.zoomIn(r.x, r.y, r.width, r.height); 
913   
914      //Recuperamos la imagen de hipótesis que contiene el panel de dibujado 
915      BufferedImage hi = m_VisualOptionsPanel.getHypothesisImage(); 
916      //Si existe dicha imagen 
917      if (hi != null){ 
918        // Hay que evitar que cuando se maximiza pete (porque el tamaño de la imagen en el 
919        // panel de opciones visuales es menor que el de la imagen del panel de dibujo 
920        int finalImageWidth = Math.max(m_DrawingPanel.getWidth(), 
921                                       hi.getWidth()); 
922        int finalImageHeight = Math.max(m_DrawingPanel.getHeight(), 
923                                        hi.getHeight()); 
924   
925   
926        //Hacemos corresponder la imagen de hipótesis con lo que vemos en la pantalla 
927        //Demasie de cutre, pensar en algo mejor (+ rápido) 
928        if (hi.getWidth() < finalImageWidth || hi.getHeight() < finalImageHeight){ 
929          m_VisualOptionsPanel.setHypothesisImage(resizeImage(m_VisualOptionsPanel.getHypothesisImage(), 
930                                                              finalImageWidth, 
931                                                              finalImageHeight)); 
932   
933          m_VisualOptionsPanel.setHypothesisSumImage(resizeImage(m_VisualOptionsPanel.getHypothesisSumImage(), 
934                                                                 finalImageWidth, 
935                                                                 finalImageHeight)); 
936        } 
937   
938        //Hacemos zoom de la imagen de hipótesis del panel de opciones visuales 
939        m_VisualOptionsPanel.setHypothesisImage(zoomInImage(m_VisualOptionsPanel.getHypothesisImage(), 
940                                                            r2, finalImageWidth, finalImageHeight)); 
941   
942        //Hacemos zoom de la imagen de hipótesis usando grados de confianza del panel de opciones visuales 
943        m_VisualOptionsPanel.setHypothesisSumImage(zoomInImage(m_VisualOptionsPanel.getHypothesisSumImage(), 
944                                                               r2, finalImageWidth, finalImageHeight)); 
945   
946        //También hacemos zoom, si procede, sobre la imagen de la rejilla 
947        BufferedImage gi = m_DrawingPanel.getGridImage(); 
948        if (gi != null){ 
949          //Hacemos corresponder la imagen de hipótesis con lo que vemos en la pantalla 
950          //Demasie de cutre, pensar en algo mejor (+ rápido) 
951          if (gi.getWidth() < finalImageWidth || gi.getHeight() < finalImageHeight) 
952            gi = resizeImage(gi, finalImageWidth, finalImageHeight); 
953   
954          m_DrawingPanel.setGridImage(zoomInImage(gi, r2, finalImageWidth, finalImageHeight)); 
955        } 
956      } 
957   
958      //Repintamos todo 
959      m_DrawingPanel.clearSelection(); 
960      m_VisualOptionsPanel.getRepaintAction().actionPerformed(null); 
961    }

966    private void doZoomOut(){ 
967   
968      ZoomHistory naturalScale = (ZoomHistory) m_ZoomHistory.get(0);

970      //Recuperamos los datos que tengamos de cómo estaban las cosas antes de hacer zoom in 
971      m_VisualOptionsPanel.setHypothesisImage(naturalScale.m_HypothesisImage);

980      m_DrawingPanel.setUpdateGrid(true); 
981      m_VisualOptionsPanel.getRepaintAction().actionPerformed(null); 
982   
983      //Liberamos la memoria reservada para el historial de zoom 
984      m_ZoomHistory = new ArrayList(); 
985    }

1006    * @param image La imagen en que se encuentra la región a escalar

1008    * @param finalImageWidth El ancho de la imagen que se retornará 
1009    * @param finalImageHeight El alto de la imagen que se retornará 
1010    * 
1011    * @return La imagen con la región escalada o null si image es null 
1012    */ 
1013   private BufferedImage zoomInImage(final BufferedImage image, final Rectangle2D.Double region, 
1014                                     final int finalImageWidth, final int finalImageHeight){ 
1015  
1016     if (image != null){ 
1017       //Comprobar que la región no exceda los límites de la imagen 
1018       int maxWidth = image.getWidth() - (int) Math.round(region.x); 
1019       int maxHeight = image.getHeight() - (int) Math.round(region.y); 
1020  
1021       //El mayor tamaño a escojer de la imagen de zoom 
1022       int width = (int) Math.round(region.width) < maxWidth ? (int) Math.round(region.width) : maxWidth; 
1023       int height = (int) Math.round(region.height) < maxHeight ? (int) Math.round(region.height) : maxHeight; 
1024  
1025       //La imagen que retornaremos 
1026       BufferedImage bi = new BufferedImage(finalImageWidth, finalImageHeight, BufferedImage.TYPE_INT_ARGB); 
1027  
1028       Graphics2D g2 = bi.createGraphics(); 
1029       //TODO: Controlada la calidad de dibujado 
1030       g2.setRenderingHint(RenderingHints.KEY_RENDERING, RenderingHints.VALUE_RENDER_QUALITY); 
1031       //Dibujamos la región de la imagen original para que ocupe toda la imagen que retornaremos 
1032       g2.drawImage(image.getSubimage((int) Math.round(region.x), 
1033                                      (int) Math.round(region.y), 
1034                                      width, 
1035                                      height), 
1036                    0, 0, finalImageWidth, finalImageHeight, null); 
1037       return bi; 
1038     } 
1039     return null; 
1040   }

1042   /** 
1043    * Retorna una nueva imagen de tamaño newWidth y newHeight y con la imagen 
1044    * image pasada como parámetro dibujada en su interior; si 
1045    * newWidth > image.getWidth() y/o newHeight > image.getHeight() 
1046    * se rellena el área restante con un color transparente (0,0,0,0) 
1047    * 
1048    * @param image La imagen original 
1049    * @param newWidth La nueva anchura 
1050    * @param newHeight La nueva altura 
1051    * 
1052    * @return La imagen con el nuevo tamaño o null si image es null 
1053    */ 
1054   private BufferedImage resizeImage(BufferedImage image, final int newWidth, final int newHeight){ 
1055     if (image != null){ 
1056       BufferedImage bi = new BufferedImage(newWidth, newHeight, BufferedImage.TYPE_INT_ARGB); 
1057  
1058       //Rellenamos la imagen con un color transparente 
1059       Graphics2D g2 = bi.createGraphics(); 
1060       g2.setColor(new Color(0, 0, 0, 0)); 
1061       g2.fillRect(0, 0, newWidth, newHeight); 
1062  
1063       //Comprobar si estamos aumentando o disminuyendo el tamaño de la imagen 
1064       final int minWidth = Math.min(image.getWidth(), newWidth); 
1065       final int minHeight = Math.min(image.getHeight(), newHeight); 
1066  
1067       //Pintar la imagen original en la nueva imagen 
1068       g2.drawImage(image.getSubimage(0, 0, minWidth, minHeight), 0, 0, minWidth, minHeight, null); 
1069       return bi; 
1070     } 
1071     return null; 
1072   }
