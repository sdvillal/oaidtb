23   package oaidtb.gui; 
24    
25   import oaidtb.misc.guiUtils.LabeledSlider; 
26    
27   import javax.swing.*; 
28   import javax.swing.event.ChangeEvent; 
29   import javax.swing.event.ChangeListener; 
30   import java.awt.*; 
31   import java.awt.event.ItemEvent; 
32   import java.awt.event.ItemListener; 
33    
34   //Precondiciones: originales >0 
35    
36   /** 
37    * Panel de configuración de la rejilla que se utilizará para crear la imagen de hipótesis 
38    * 
39    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
40    * @version $Revision: 1.0 $ 
41    */ 
42   public class GridDimensionOptionsPanel extends JPanel{ 
43    
44     private double m_XYDimensionsOriginalRatio = 1; 
45     /** Para elegir el número de columnas de la rejilla */ 
46     private LabeledSlider m_XSlider = new LabeledSlider("Grid Columns: "); 
47     /** Para elegir el número de filas de la rejilla */ 
48     private LabeledSlider m_YSlider = new LabeledSlider("Grid Rows: "); 
49    
50     /** La relación original entre la altura y la anchura del panel de dibujado */ 
51       /** Se debe mantener la relación de aspecto original? */ 
52     private JCheckBox m_MaintainAspectRatio = new JCheckBox("Maintain aspect ratio", true); 
53    
54     /** @return El número de columnas de la rejilla */ 
55     public int getColumns(){ 
56       return m_XSlider.getSlider().getValue(); 
57     } 
58    
59     /** @return El número de filas de la rejilla  */ 
60     public int getRows(){ 
61       return m_YSlider.getSlider().getValue(); 
62     } 
63    
64     /** 
65      * Actualizar la anchura asignada como la original del componente a ser dividido por la rejilla 
66      * 
67      * @param xSize La anchura del componente 
68      */ 
69     public void setOriginalXDimensionSize(int xSize){ 
70       //throw an Exception 
71       if (xSize > 0){ 
72         m_XSlider.getSlider().setMaximum(xSize); 
73         updateXYOriginalRatio(); 
74       } 
75     } 
76    
77     /** 
78      * Actualizar la altura asignada como la original del componente a ser dividido por la rejilla 
79      * 
80      * @param ySize La anchura del componente 
81      */ 
82     public void setOriginalYDimensionSize(int ySize){ 
83       //throw an Exception 
84       if (ySize > 0){ 
85         m_YSlider.getSlider().setMaximum(ySize); 
86         updateXYOriginalRatio(); 
87       } 
88     } 
89    
90     /** 
91      * Actualizar la relación entre altura y anchura 
92      */ 
93     private void updateXYOriginalRatio(){ 
94       m_XYDimensionsOriginalRatio = (double) m_XSlider.getSlider().getMaximum() / 
95                                     (double) m_YSlider.getSlider().getMaximum(); 
96       if(m_MaintainAspectRatio.isSelected()) 
97         m_YSlider.getSlider().setValue((int)(m_XSlider.getSlider().getValue()/m_XYDimensionsOriginalRatio)); 
98     } 
99    
100    /** 
101     * Constructor 
102     */ 
103    public GridDimensionOptionsPanel(){ 
104   
105      // Listener para mantener la relación de aspecto cuando se cambia el número de columnas 
106      final ChangeListener maintainRatioListener = new ChangeListener(){ 
107        public void stateChanged(ChangeEvent e){ 
108          m_YSlider.getSlider().setValue((int) Math.round(m_XSlider.getSlider().getValue() / m_XYDimensionsOriginalRatio)); 
109        } 
110      }; 
111   
112      //Valores por defecto, para que no pete 
113      m_XSlider.getSlider().setMinimum(1); 
114      m_XSlider.getSlider().addChangeListener(maintainRatioListener); 
115      m_XSlider.getSlider().setValue(30); 
116   
117      m_YSlider.getSlider().setMinimum(1); 
118      m_YSlider.getSlider().setEnabled(false); 
119   
120      //Checkbox para elegir entre mantener o no la relación de aspecto 
121      m_MaintainAspectRatio.addItemListener(new ItemListener(){ 
122        public void itemStateChanged(ItemEvent e){ 
123          if (m_MaintainAspectRatio.isSelected()){ 
124            m_XSlider.getSlider().addChangeListener(maintainRatioListener); 
125            m_YSlider.getSlider().setEnabled(false); 
126          } 
127          else{ 
128            m_XSlider.getSlider().removeChangeListener(maintainRatioListener); 
129            m_YSlider.getSlider().setEnabled(true); 
130          } 
131        } 
132      }); 
133      m_MaintainAspectRatio.setToolTipText("Mantener la relación de aspecto X/Y del panel de dibujo"); 
134   
135      //Será un BorderLayout 
136      setLayout(new BorderLayout()); 
137   
138      //Añadimos los componentes al panel 
139      JPanel sizeEditorsPanel = new JPanel(new GridLayout(0, 1)); 
140      sizeEditorsPanel.add(m_XSlider.getLabel()); 
141      sizeEditorsPanel.add(m_XSlider.getSlider()); 
142      sizeEditorsPanel.add(m_YSlider.getLabel()); 
143      sizeEditorsPanel.add(m_YSlider.getSlider()); 
144   
145      add(m_MaintainAspectRatio, BorderLayout.SOUTH); 
146      add(sizeEditorsPanel, BorderLayout.CENTER); 
147   
148      //Creamos el borde por defecto 
149      setBorder(BorderFactory.createCompoundBorder( 
150  		 BorderFactory.createTitledBorder("Grid setup"), 
151  		 BorderFactory.createEmptyBorder(0, 5, 5, 5) 
152  		 )); 
153    } 
154  }
