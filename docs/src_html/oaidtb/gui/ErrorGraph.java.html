108    /** Constructor. Generamos el ChartPanel por defecto.  */ 
109    public ErrorGraph(){ 
110   
111      // create the chart... 
112      chart = ChartFactory.createXYChart("Error plot", // chart title 
113                                         "Iteración", // domain axis label 
114                                         "Error", // range axis label 
115                                         dataset, // dataset 
116                                         true     // include legend 
117      ); 
118   
119      // get a reference to the plot for further customisation... 
120      XYPlot plot = chart.getXYPlot(); 
121   
122      plot.getHorizontalValueAxis().setAutoRange(true); 
123      ((NumberAxis) plot.getHorizontalAxis()).setStandardTickUnits(TickUnits.createIntegerTickUnits()); 
124  //    ((NumberAxis)plot.getHorizontalAxis()).setMinimumAxisValue(0); 
125   
126      //Generate the tooltip for each point in the plot 
127      plot.getItemRenderer().setToolTipGenerator(new XYToolTipGenerator(){ 
128        public String generateToolTip(XYDataset data, int series, int item){ 
129          return String.valueOf(data.getSeriesName(series) + ": " + data.getYValue(series, item)); 
130        } 
131      }); 
132   
133      chart.setAntiAlias(false); 
134   
135      // add the chart to a panel... 
136      chartPanel = new ChartPanel(chart); 
137    }

155    /** 
156     * Comprobar si un número de serie es válido (está en el rango definido) 
157     * 
158     * @param serieID El número identificativo de la serie de datos 
159     * @return true si el número se corresponde con una serie, falso si no 
160     */ 
161    private boolean validSerieID(int serieID){ 
162      return serieID < seriesArray.length && serieID > -1; 
163    }

165    /** 
166     * Añade un punto a una serie (si el identificador no es válido, el método 
167     * retorna sin avisar) del gráfico 
168     * 
169     * @param serieID El identificador de la serie 
170     * @param x La abcisa del punto 
171     * @param y La ordenada del punto 
172     */ 
173    public void addPoint(int serieID, double x, double y){ 
174      if (!validSerieID(serieID)) 
175        return;  //throw Exception 
176      seriesArray[serieID].add(new XYDataPair(x, y)); 
177    }

179    /** 
180     * Mostrar o no el gráfico correspondiente a una determinada serie de datos 
181     * 
182     * @param serieID El identificador de la serie 
183     * 
184     * @param represent true si la serie se debe mostrar en el gráfico, false si no 
185     */ 
186    public void setRepresentSerie(int serieID, boolean represent){ 
187      if (!validSerieID(serieID)) 
188        return;  //throw Exception 
189      if (representSerie[serieID] != represent){ 
190        representSerie[serieID] = represent; 
191        if (represent){ 
192          dataset.addSeries(seriesArray[serieID]); 
193          updateBoosterErrorgraph(); 
194          seriesArray[serieID].fireSeriesChanged(); 
195        } 
196        else 
197          dataset.removeSeries(seriesArray[serieID]); 
198      } 
199    }

275    /** 
276     * Una clase que extiende XYSeries para poder añadir datos a cada una de las líneas del gráfico 
277     * bastante más rápido; utiliza un ArrayList de SimpleDoubles para almacenar los datos, pudiendo

324      /**

326       * Presuponemos:

341    /** 
342     * Una clase que extiende XYSeries para poder añadir datos a cada una de las líneas del gráfico 
343     * bastante más rápido; utiliza un ArrayList de SimpleDoubles para almacenar los datos, pudiendo 
344     * este ArrayList ser compartido por más objetos;  ver las notas del método add. 
345     * 
346     * Además esta clase permite pasar un faxtor de normalización que se aplicará cada 
347     * vez que se recuperen los datos. 
348     * 
349     * TODO: Utilizar esta clase para crear líneas de costes representando el coste "relativo" 
350     * en vez del total (al estilo de lo que se hacía en la versión 1.0); permitir elegir 
351     * entre ambas representaciones en el panel de selección de series. 
352     */ 
353    public static class FastArrayListXYSeriesWithNormalizationFactor extends FastArrayListXYSeries{ 
354   
355      /** El factor de normalización */ 
356      private double normFactor = 1; 
357   
358      /** Constructor */ 
359      public FastArrayListXYSeriesWithNormalizationFactor(String name){ 
360        super(name); 
361      } 
362   
363      /** @param normFactor El número por el que serán divididos los valores en cada llamada a getXXX    */ 
364      public void setNormFactor(double normFactor){ 
365        if (normFactor != 0) 
366          this.normFactor = normFactor; 
367      } 
368   
369      /** 
370       * Returns the y-value at the specified index. 
371       * 
372       * @param index The index. 
373       * 
374       * @return The y-value. 
375       */ 
376      public Number getYValue(int index){ 
377        return new SimpleDouble(((SimpleDouble) data.get(index)).value / normFactor); 
378      } 
379   
380      /** 
381       * Return the data pair with the specified index. 
382       * 
383       * @param index The index. 
384       */ 
385      public XYDataPair getDataPair(int index){ 
386        return new XYDataPair(new SimpleInteger(index), new SimpleDouble(((SimpleDouble) data.get(index)).value / normFactor)); 
387      } 
388    }

390    /** 
391     * Un lístener que hace aparecer un menú con las posibles líneas a representar 
392     * cuando el usuario hace click con el botón izquierdo 
393     * sobre un panel que contiene la gráfica 
394     */ 
395    public static class SerieSelectorListener implements ChartMouseListener{ 
396   
397      /** Array (debe corresponderse con las constantes anteriones) de las posibles líneas a representar. */ 
398      private final JCheckBox[] representableSeriesArray; 
399   
400      /** El panel en el que irán las checkboxes de selección */ 
401      private final JPanel serieSelectorPanel; 
402   
403      /** El gráfico de error del que vamos a seleccionar las series a representar */ 
404      private final ErrorGraph errorGraph; 
405   
406      /** Listener que actualiza el gráfico cuando un checkbos es (de)seleccionada */ 
407      private final ItemListener representSerieListener = new ItemListener(){ 
408        public void itemStateChanged(ItemEvent e){ 
409          final JCheckBox source = (JCheckBox) e.getSource(); 
410          int serieIndex; 
411          for (serieIndex = 0; serieIndex < representableSeriesArray.length; serieIndex++) 
412            if (source == representableSeriesArray[serieIndex]) 
413              break; 
414          errorGraph.setRepresentSerie(serieIndex, source.isSelected()); 
415        } 
416      }; 
417   
418      /** 
419       * Constructor 
420       * 
421       * @param errorGraph El gráfico de error del que vamos a elegir qué series representar 
422       */ 
423      public SerieSelectorListener(ErrorGraph errorGraph){ 
424   
425        this.errorGraph = errorGraph; 
426   
427        representableSeriesArray = new JCheckBox[errorGraph.seriesArray.length]; 
428        for (int i = 0; i < representableSeriesArray.length; i++){ 
429          representableSeriesArray[i] = new JCheckBox(errorGraph.seriesArray[i].getName(), 
430                                                      errorGraph.representSerie[i]); 
431          representableSeriesArray[i].setEnabled(false); 
432        } 
433   
434        serieSelectorPanel = new JPanel(new GridLayout(5, 2)); 
435        for (int i = 0; i < representableSeriesArray.length; i++) 
436          serieSelectorPanel.add(representableSeriesArray[i]); 
437        serieSelectorPanel.validate(); 
438   
439        for (int i = 0; i < representableSeriesArray.length; i++) 
440          representableSeriesArray[i].addItemListener(representSerieListener); 
441      } 
442   
443      /** 
444       * Sincroniza la apariencia (seleccionada o no) de las checkboxes 
445       * con las líneas que se están representando o no en el gráfico 
446       */ 
447      public void synchronizeCheckBoxes(){ 
448        for (int i = 0; i < representableSeriesArray.length; i++) 
449          representableSeriesArray[i].setSelected(errorGraph.representSerie[i]); 
450      } 
451   
452      /** 
453       * Mostrar o no el gráfico correspondiente a una determinada serie de datos 
454       * 
455       * @param serieID El identificador de la serie 
456       * 
457       * @param representable true si la serie se debe mostrar en el gráfico, false si no 
458       */ 
459      public void setRepresentableSerie(int serieID, boolean representable){ 
460        if (!errorGraph.validSerieID(serieID)) 
461          return;  //throw Exception 
462        representableSeriesArray[serieID].setEnabled(representable); 
463      }

487      }
