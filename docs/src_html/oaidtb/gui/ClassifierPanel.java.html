246    /** 
247     * Configurar las acciones definidas; por defecto, las metemos en el InputMap

253    public void setUpActions(final InputMap im, final ActionMap am){ 
254   
255      //////////////////////////////////////////////////// 
256      // Construir o iterar (Alt-B) 
257      //////////////////////////////////////////////////// 
258      m_BuildOrIterateAction = new MyAbstractAction("Build_Or_Iterate"){ 
259        public void actionPerformed(ActionEvent e){ 
260          if (isEnabled()) 
261            m_BuildButton.doClick(); 
262        } 
263      }; 
264      m_BuildOrIterateAction.setToolTipText("Construir el clasificador / hacer las siguientes iteraciones" + " (Alt-B)"); 
265      im.put(KeyStroke.getKeyStroke(KeyEvent.VK_B, KeyEvent.ALT_MASK), "Build_Or_Iterate"); 
266      am.put("Build_Or_Iterate", m_BuildOrIterateAction); 
267   
268      //////////////////////////////////////////////////// 
269      // Cancelar (Ctrl-B) 
270      //////////////////////////////////////////////////// 
271      m_CancelBuildOrIterateAction = new MyAbstractAction("Cancel_Build_Or_Iterate"){ 
272        public void actionPerformed(ActionEvent e){ 
273          if (isEnabled()) 
274            m_CancelBuildButton.doClick(); 
275        } 
276      }; 
277      m_CancelBuildOrIterateAction.setEnabled(false); 
278      m_CancelBuildOrIterateAction.setToolTipText("Cancelar la construcción/iteración de un clasificador/booster" + " (Ctrl-B"); 
279      im.put(KeyStroke.getKeyStroke(KeyEvent.VK_B, KeyEvent.CTRL_MASK), "Cancel_Build_Or_Iterate"); 
280      am.put("Cancel_Build_Or_Iterate", m_CancelBuildOrIterateAction); 
281   
282      //////////////////////////////////////////////////// 
283      // Construir un clasificador (no booster) (Alt-B) 
284      //////////////////////////////////////////////////// 
285      m_BuildClassifierAction = new MyAbstractAction("Build_Classifier"){ 
286        public void actionPerformed(ActionEvent e){ 
287          if (!isEnabled() || m_CBThread != null) 
288            return; 
289          m_CBThread = new ClassifierBuildThread(); 
290          //Permitir configurar la prioridad al usuario 
291          m_CBThread.setPriority(Thread.MIN_PRIORITY); 
292          m_CBThread.start(); 
293        } 
294      }; 
295      m_BuildClassifierAction.setToolTipText("Construir un nuevo clasificador" + " (Alt-B)"); 
296   
297      //////////////////////////////////////////////////// 
298      // Iterar un booster (Alt-B) 
299      //////////////////////////////////////////////////// 
300      m_BoosterIterateAction = new MyAbstractAction("Booster_Iterate"){ 
301        public void actionPerformed(ActionEvent e){ 
302          if (!isEnabled() || m_BoosterWorker != null) 
303            return; 
304          m_BoosterWorker = m_BoosterPanel.getNewBoosterIterateWorker(); 
305          m_BoosterWorker.start(); 
306        } 
307      }; 
308      m_BoosterIterateAction.setToolTipText("Realizar nuevas iteraciones" + " (Alt-B)"); 
309   
310      //////////////////////////////////////////////////// 
311      // Cancelar la construcción de un clasificador (Ctrl-B) 
312      //////////////////////////////////////////////////// 
313      m_CancelBuildClassifierAction = new MyAbstractAction("Cancel_Build_Classifier"){ 
314        public void actionPerformed(ActionEvent e){ 
315          if (m_CBThread != null){ 
316            //Seguramente t.interrupt() no hará nada, ver el siguiente comentario. 
317            //Lo matamos de manera abrupta, asíncronamente, ya que los métodos buildClassifier de los clasificadores 
318            //de weka no están preparados para manejar el estado interrupt del thread en que están 
319            //siendo ejecutados. 
320            m_CBThread.stop(); 
321            notifyMediators(CLASSIFIER_BUILD_CANCELED, getClassifier()); 
322          } 
323        } 
324      }; 
325      m_CancelBuildClassifierAction.setEnabled(false); 
326      m_CancelBuildClassifierAction.setToolTipText("Cancelar la construcción del clasificador" + " Ctrl-B"); 
327   
328      //////////////////////////////////////////////////// 
329      // Cancelar la construcción de un booster (Ctrl-B) 
330      //////////////////////////////////////////////////// 
331      m_CancelBoosterIterateAction = new MyAbstractAction("Cancel_Booster_Iterate"){ 
332        public void actionPerformed(ActionEvent e){ 
333          if (m_BoosterWorker != null){ 
334            m_BoosterWorker.interrupt(); 
335            notifyMediators(BOOSTER_ITERATE_CANCELED, getClassifier()); 
336          } 
337        } 
338      }; 
339      m_CancelBoosterIterateAction.setEnabled(false); 
340      m_CancelBoosterIterateAction.setToolTipText("Cancelar la iteración del booster" + " (Ctrl-B"); 
341   
342      //////////////////////////////////////////////////// 
343      // Resetear el problema del booster (Ctrl + Alt + B) 
344      //////////////////////////////////////////////////// 
345      m_ResetAction = new MyAbstractAction("Reset_Booster_Problem"){ 
346        public void actionPerformed(ActionEvent e){ 
347          if (!isEnabled()) 
348            return; 
349          chooseInstances(); 
350          m_TextInfoArea.setText(""); 
351          //Liberar la memoria reservada por el booster para almacenar las instancias de entrenamiento 
352          //y para almacenar los clasificadores base, destruir pues el booster 
353          if (m_Classifier instanceof IterativeUpdatableClassifier){ 
354            m_BoosterPanel.resetGraph(); 
355            try{ 
356              if(m_Classifier instanceof AdaBoostMH){ 
357                ((AdaBoostMH)m_Classifier).purgeTrainData(); 
358                ((AdaBoostMH)m_Classifier).purgeIterations(((AdaBoostMH)m_Classifier).getNumIterationsPerformed()); 
359              } 
360              else{ 
361                ((Booster)m_Classifier).purgeTraindata(); 
362                ((Booster)m_Classifier).purgeIterations(((Booster)m_Classifier).getNumIterationsPerformed()); 
363              } 
364            } 
365            catch(Exception ex){ 
366              ex.printStackTrace(); 
367            } 
368          } 
369        } 
370      }; 
371      m_ResetAction.setToolTipText("Resetear el estado del booster y/o elegir las instancias" + " (Ctrl-Alt-B)"); 
372      im.put(KeyStroke.getKeyStroke(KeyEvent.VK_B, KeyEvent.CTRL_MASK & KeyEvent.CTRL_MASK), "Reset_Booster_Problem"); 
373      am.put("Reset_Booster_Problem", m_ResetAction); 
374    }

1194     /** 
1195      * Método que se encarga de realizar las siguientes iteraciones del booster en cuestión, 
1196      * así como de actualizar la línea que representa algún tipo de seguridad 
1197      * teórica acerca del límite máximo de alguna medida del error (que línea más farragosa) 
1198      * 
1199      * @param numIterations El número de iteraciones a realizar 
1200      * 
1201      * @throws InterruptedException Si se cancela el proceso de iteración 
1202      * @throws Exception Si ocurre un error 
1203      */ 
1204     private void iterate(int numIterations) throws Exception{ 
1205  
1206       //Tal y como está programado ahora mismo, este caso no se puede dar 
1207       if (m_Booster == null) 
1208         throw new Exception("BoosterPanel.iterate: El booster es nulo"); 
1209  
1210       //Para llevar la cuenta de las iteraciones que nos quedan 
1211       int itIndex = 0; 
1212  
1213       //Preparamos la barra de progreso 
1214       m_ProgressBar.setMinimum(0); 
1215       m_ProgressBar.setMaximum(numIterations); 
1216       m_ProgressBar.setStringPainted(true); 
1217  
1218       //En el caso de que tengamos que volver a crear el booster 
1219       if (m_MustInitialize || m_Booster.getNumIterationsPerformed() == 0){ 
1220  
1221         //Elegimos las instancias 
1222         chooseInstances(); 
1223         //Inicializamos el gráfico y los analizadores, y bajamos la bandera de inicialización 
1224         initializeGraph(); 
1225  
1226         //Vamos a realizar la primera iteración 
1227         updatePBStatus(1); 
1228  
1229         if (m_Booster instanceof AdaBoostMH){ 
1230           ((AdaBoostMH) m_Booster).getBooster().setInitialIterations(1); 
1231           ((AdaBoostMH) m_Booster).buildClassifier(m_TrainData); 
1232         } 
1233         else{ 
1234           ((Booster) m_Booster).setInitialIterations(1); 
1235           ((Booster) m_Booster).buildClassifier(m_TrainData); 
1236         } 
1237         //Calculamos, si procede, el límite teórico de alguna medida del error 
1238         if (m_ComputeErrorBound){ 
1239           final double errorUpperBound = ((ErrorUpperBoundComputer) m_Booster).getErrorUpperBound(); 
1240           m_ErrorGraph.addPoint(ErrorGraph.BOOSTER_ERROR_BOUND, 
1241                                 0, 
1242                                 errorUpperBound > 1 ? 1 : errorUpperBound); 
1243         } 
1244  
1245         //Ya hemos hecho la primera iteración 
1246         itIndex = 1; 
1247         notifyMediators(BOOSTER_FIRST_ITERATION, m_Booster); 
1248       } 
1249  
1250       if (m_ComputeErrorBound){ 
1251         //Iteramos calculando el límite teórico de alguna medida del error 
1252         int numIterationsperformed = m_Booster.getNumIterationsPerformed(); 
1253  
1254         for (itIndex++; itIndex <= numIterations; itIndex++){ 
1255  
1256           //Comprobamos que el usuario no haya cancelado el proceso 
1257           if (Thread.interrupted()) 
1258             throw new InterruptedException(); 
1259  
1260           //Actualizar la barra de progreso 
1261           updatePBStatus(itIndex); 
1262  
1263           //Hacer una iteración más 
1264           m_Booster.nextIterations(1); 
1265  
1266           //Recuperamos el límite del error para la iteración 
1267           final double errorUpperBound = ((ErrorUpperBoundComputer) m_Booster).getErrorUpperBound(); 
1268  
1269           //Añadimos el punto a la línea del límite del error en el gráfico 
1270           m_ErrorGraph.addPoint(ErrorGraph.BOOSTER_ERROR_BOUND, 
1271                                 numIterationsperformed, 
1272                                 errorUpperBound > 1 ? 1 : errorUpperBound); 
1273  
1274           //Actualizamos la abcisa para el siguiente punto de la línea del límite del error en el gráfico 
1275           numIterationsperformed++; 
1276         } 
1277       } 
1278       else//No vamos a calcular el límite del error 
1279         for (itIndex++; itIndex <= numIterations; itIndex++){ 
1280  
1281           //Comprobamos que el usuario no haya cancelado el proceso 
1282           if (Thread.interrupted()) 
1283             throw new InterruptedException(); 
1284  
1285           //Actualizamos la barra de progreso 
1286           updatePBStatus(itIndex); 
1287  
1288           //Realizamos una iteración más 
1289           m_Booster.nextIterations(1); 
1290         } 
1291     }
