33   /** 
34    * Un panel que se añade al JColorChooser para permitir al usuario 
35    * reelegir entre los colores de los que ya existen instancias en un 
36    * conjunto de datos "Point2dInstances". 
37    * 
38    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
39    * @version $Revision: 1.0 $ 
40    */ 
41   public class ColorHistoryPanel extends AbstractColorChooserPanel{ 
42    
43     private Point2DInstances m_Colors; 
44    
45     /** 
46      * This method must be call every time before the panel is shown to ensure 
47      * that all the information it contains is correct. 
48      */ 
49     public void remake(){ 
50       removeAll(); 
51       JPanel panel = new JPanel(new GridLayout(0, 8)); 
52       FastVector colors = m_Colors.getColors(); 
53       for (int i = 0; i < colors.size(); i++){ 
54         Color c = (Color) colors.elementAt(i); 
55         JButton button = new JButton(String.valueOf(m_Colors.getNumPointsOfColor(c))); 
56         button.setBackground(c); 
57         button.addActionListener(setColorAction); 
58         panel.add(button); 
59       } 
60       setLayout(new BorderLayout()); 
61       JLabel label = new JLabel("There are " + colors.size() + " colors"); 
62       add(label, BorderLayout.NORTH); 
63       JScrollPane jScrollPane = new JScrollPane(panel); 
64   //    jScrollPane.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_AS_NEEDED); 
65   //    jScrollPane.setHorizontalScrollBarPolicy(JScrollPane.HORIZONTAL_SCROLLBAR_AS_NEEDED); 
66       add(jScrollPane, BorderLayout.CENTER); 
67     } 
68    
69     public Point2DInstances getColors(){ 
70       return m_Colors; 
71     } 
72    
73     public void setColors(Point2DInstances colors){ 
74       m_Colors = colors; 
75     } 
76    
77     public ColorHistoryPanel(Point2DInstances colors){ 
78       super(); 
79       m_Colors = colors; 
80     }

121    // This action takes the background color of the button 
122    // and uses it to set the selected color. 
123    Action setColorAction = new AbstractAction(){ 
124      public void actionPerformed(ActionEvent evt){ 
125        JButton button = (JButton) evt.getSource(); 
126        getColorSelectionModel().setSelectedColor(button.getBackground()); 
127      }
