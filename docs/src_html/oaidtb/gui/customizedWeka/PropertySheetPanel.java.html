25    *   ** It pass to every GenericObjectEditor referenced in the sheet the class name of the object being edited, 
26    *      so they can retrieve special names from the properties configuration file (i.e. Booster@MH). 
27    *   ** Possibility of customize the order in which the properties are displayed by the edited object's 
28    *      "sortProperties" method. 
29    *   ** "More information window" converted to a JDialog (see openHelpFrame) 
30    *   ** Verbose mode added, and by default disbled (I don't like so many messages to the 
31    *      System.err stream in an already validated class)

149    /** 
150     * Sets a new target object for customisation. 
151     * 
152     * @param targ a value of type 'Object' 
153     */ 
154    public synchronized void setTarget(Object targ){ 
155   
156      // used to offset the components for the properties of targ 
157      // if there happens to be globalInfo available in targ 
158      int componentOffset = 0; 
159   
160      // Close any child windows at this point 
161      removeAll(); 
162   
163      GridBagLayout gbLayout = new GridBagLayout(); 
164   
165      setLayout(gbLayout); 
166      setVisible(false); 
167      m_NumEditable = 0; 
168      m_Target = targ; 
169      try{ 
170        BeanInfo bi = Introspector.getBeanInfo(m_Target.getClass()); 
171        m_Properties = bi.getPropertyDescriptors(); 
172        m_Methods = bi.getMethodDescriptors(); 
173      } 
174      catch (IntrospectionException ex){ 
175        System.err.println("PropertySheet: Couldn't introspect"); 
176        return; 
177      } 
178   
179      JTextArea jt = new JTextArea(){ 
180        /** 
181         * Return false, because we don't want this text area to receive the focus, and if this happens 
182         * it will "swallow" it (JDK bug?). 
183         * 
184         * The new JDK 1.4.x gives the setFocusable() method to do this; isFocusTraversable is deprecated there. 
185         * 
186         * The same effect (but with undesired visual collaterals) can be obtained by: jt.setEnabled(false); 
187         * 
188         * @return true if the focus is traversable 
189         */ 
190        public boolean isFocusTraversable(){ 
191          return false; 
192        } 
193      }; 
194   
195      m_HelpText = null; 
196   
197  // En el JDK 1.4, el Introspector no encuentra los métodos 
198  // que existen en Booster (por ejemplo, sortProperties) 
199  // desde AdaBoostECC (2 niveles por debajo en la jerarquía de clases), 
200  // pero class.getMethod sí que lo encuentra 
201  //    Class c = m_Target.getClass(); 
202  //    while(Object.class != c){ 
203  //      Method mola = null; 
204  //        try{ 
205  //          mola = c.getMethod("sortProperties", new Class[]{PropertyDescriptor[].class}); 
206  //        } catch(NoSuchMethodException ex){ 
207  //          System.err.println("No encontrado en " +c.getName()); 
208  //        } 
209  //      if(mola != null) 
210  //        System.err.println(mola.getName() +" encontrado en " +c.getName()); 
211  //      c = c.getSuperclass(); 
212  //    } 
213   
214      //Look for globalInfo & sortProperties methods 
215      for (int i = 0; i < m_Methods.length; i++){ 
216   
217        String name = m_Methods[i].getName(); 
218        Method meth = m_Methods[i].getMethod(); 
219   
220        //If this method exists, it will define in which order and/or which 
221        //properties will be shown for editing. The order can't be defined by 
222        //a BeanInfo class, since no special order is defined 
223        //by the Java Beans specs; in fact, overriding the method 
224        //getPropertyDescriptors() in the target BeanInfo class 
225        //will only be used by the JDK classes when calling to the 
226        //Introspector.getBeanInfo() method to construct the index 
227        //(currently a HashMap)of the properties names over the properties descriptors, 
228        //and once this index is built, the array returned every time by 
229        //Introspector.getBeanInfo().getPropertyDescriptors() is constructed 
230        //using an Iterator over the index, whose order is undefined. 
231        if (name.equals("sortProperties") && 
232          PropertyDescriptor[].class.equals(meth.getReturnType()) 
233          && meth.getParameterTypes()[0] == PropertyDescriptor[].class){ 
234          try{ 
235            m_Properties = (PropertyDescriptor[]) meth.invoke(m_Target, new Object[]{m_Properties}); 
236          } 
237          catch (Exception e){ 
238            System.err.println(e.toString()); 
239          } 
240        } 
241   
242        // Look for a globalInfo method that returns a string 
243        // describing the target 
244        if (name.equals("globalInfo")){ 
245          if (meth.getReturnType().equals(String.class)){ 
246            try{ 
247              Object args[] = {}; 
248              String globalInfo = (String) (meth.invoke(m_Target, args)); 
249              String summary = globalInfo; 
250              int ci = globalInfo.indexOf('.'); 
251              if (ci != -1){ 
252                summary = globalInfo.substring(0, ci + 1); 
253              } 
254              final String className = targ.getClass().getName(); 
255              m_HelpText = new StringBuffer("NAME\n"); 
256              m_HelpText.append(className).append("\n\n"); 
257              m_HelpText.append("SYNOPSIS\n").append(globalInfo).append("\n\n"); 
258              m_HelpBut = new JButton("More"); 
259              m_HelpBut.setMnemonic('m'); 
260              m_HelpBut.setToolTipText("More information about " 
261                                       + className); 
262   
263              m_HelpBut.addActionListener(new ActionListener(){ 
264                public void actionPerformed(ActionEvent a){ 
265                  openHelpFrame(); 
266                  m_HelpBut.setEnabled(false); 
267                } 
268              }); 
269   
270              jt.setFont(new Font("SansSerif", Font.PLAIN, 12)); 
271              jt.setEditable(false); 
272              jt.setLineWrap(true); 
273              jt.setWrapStyleWord(true); 
274              jt.setText(summary); 
275              jt.setBackground(getBackground()); 
276              JPanel jp = new JPanel(); 
277              jp.setBorder(BorderFactory.createCompoundBorder( 
278                BorderFactory.createTitledBorder("About"), 
279                BorderFactory.createEmptyBorder(0, 5, 5, 5) 
280              )); 
281              jp.setLayout(new BorderLayout()); 
282              jp.add(jt, BorderLayout.CENTER); 
283              JPanel p2 = new JPanel(); 
284              p2.setLayout(new BorderLayout()); 
285              p2.add(m_HelpBut, BorderLayout.NORTH); 
286              jp.add(p2, BorderLayout.EAST); 
287              GridBagConstraints gbConstraints = new GridBagConstraints(); 
288              //	    gbConstraints.anchor = GridBagConstraints.EAST; 
289              gbConstraints.fill = GridBagConstraints.BOTH; 
290              //	    gbConstraints.gridy = 0;     gbConstraints.gridx = 0; 
291              gbConstraints.gridwidth = 2; 
292              gbConstraints.insets = new Insets(0, 5, 0, 5); 
293              gbLayout.setConstraints(jp, gbConstraints); 
294              add(jp); 
295              componentOffset = 1; 
296              break; 
297            } 
298            catch (Exception ex){ 
299            } 
300          } 
301        } 
302      } 
303   
304      m_Editors = new PropertyEditor[m_Properties.length]; 
305      m_Values = new Object[m_Properties.length]; 
306      m_Views = new JComponent[m_Properties.length]; 
307      m_Labels = new JLabel[m_Properties.length]; 
308      m_TipTexts = new String[m_Properties.length]; 
309      boolean firstTip = true; 
310      for (int i = 0; i < m_Properties.length; i++){ 
311   
312        // Don't display hidden or expert properties. 
313        if (m_Properties[i].isHidden() || m_Properties[i].isExpert()){ 
314          continue; 
315        } 
316   
317        String name = m_Properties[i].getDisplayName(); 
318        Class type = m_Properties[i].getPropertyType(); 
319        Method getter = m_Properties[i].getReadMethod(); 
320        Method setter = m_Properties[i].getWriteMethod(); 
321   
322        // Only display read/write properties. 
323        if (getter == null || setter == null){ 
324          continue; 
325        } 
326   
327        JComponent view; 
328   
329        try{ 
330          Object args[] = {}; 
331          Object value = getter.invoke(m_Target, args); 
332          m_Values[i] = value; 
333   
334          PropertyEditor editor = null; 
335          Class pec = m_Properties[i].getPropertyEditorClass(); 
336          if (pec != null){ 
337            try{ 
338              editor = (PropertyEditor) pec.newInstance(); 
339            } 
340            catch (Exception ex){ 
341              System.err.println(ex.toString()); 
342              // Drop through. 
343            } 
344          } 
345          if (editor == null){ 
346            editor = PropertyEditorManager.findEditor(type); 
347          } 
348          m_Editors[i] = editor; 
349   
350          // If we can't edit this component, skip it. 
351          if (editor == null){ 
352            // If it's a user-defined property we give a warning. 
353            String getterClass = m_Properties[i].getReadMethod() 
354              .getDeclaringClass().getName(); 
355            if (m_VerboseMode && getterClass.indexOf("java.") != 0){ 
356              System.err.println("Warning: Can't find public property editor" 
357                                 + " for property \"" + name + "\" (class \"" 
358                                 + type.getName() + "\").  Skipping."); 
359            } 
360            continue; 
361          } 
362          if (editor instanceof GenericObjectEditor){ 
363            ((GenericObjectEditor) editor).setSpecializeFromClass(m_Target.getClass().getName()); 
364            ((GenericObjectEditor) editor).setClassType(type); 
365          } 
366          else if (editor instanceof weka.gui.GenericObjectEditor){ 
367            ((weka.gui.GenericObjectEditor) editor).setClassType(type); 
368          } 
369   
370          // Don't try to set null values: 
371          if (value == null){ 
372            // If it's a user-defined property we give a warning. 
373            String getterClass = m_Properties[i].getReadMethod() 
374              .getDeclaringClass().getName(); 
375            if (m_VerboseMode && getterClass.indexOf("java.") != 0){ 
376              System.err.println("Warning: Property \"" + name 
377                                 + "\" has null initial value.  Skipping."); 
378            } 
379            continue; 
380          } 
381   
382          editor.setValue(value); 
383   
384          // now look for a TipText method for this property 
385          String tipName = name + "TipText"; 
386          for (int j = 0; j < m_Methods.length; j++){ 
387            String mname = m_Methods[j].getDisplayName(); 
388            Method meth = m_Methods[j].getMethod(); 
389            if (mname.equals(tipName)){ 
390              if (meth.getReturnType().equals(String.class)){ 
391                try{ 
392                  String tempTip = (String) (meth.invoke(m_Target, args)); 
393                  int ci = tempTip.indexOf('.'); 
394                  if (ci < 0){ 
395                    m_TipTexts[i] = tempTip; 
396                  } 
397                  else{ 
398                    m_TipTexts[i] = tempTip.substring(0, ci); 
399                  } 
400                  if (m_HelpText != null){ 
401                    if (firstTip){ 
402                      m_HelpText.append("OPTIONS\n"); 
403                      firstTip = false; 
404                    } 
405                    m_HelpText.append(name).append(" -- "); 
406                    m_HelpText.append(tempTip).append("\n\n"); 
407                    //jt.setText(m_HelpText.toString()); 
408                  } 
409                } 
410                catch (Exception ex){ 
411                } 
412                break; 
413              } 
414            } 
415          } 
416   
417          // Now figure out how to display it... 
418          if (editor.isPaintable() && editor.supportsCustomEditor()){ 
419            view = new PropertyPanel(editor); 
420          } 
421          else if (editor.getTags() != null){ 
422            view = new PropertyValueSelector(editor); 
423          } 
424          else if (editor.getAsText() != null){ 
425            //String init = editor.getAsText(); 
426            view = new PropertyText(editor); 
427          } 
428          else{ 
429            if(m_VerboseMode) 
430              System.err.println("Warning: Property \"" + name 
431                                 + "\" has non-displayabale editor.  Skipping."); 
432            continue; 
433          } 
434   
435          editor.addPropertyChangeListener(this); 
436   
437        } 
438        catch (InvocationTargetException ex){ 
439          System.err.println("Skipping property " + name 
440                             + " ; exception on target: " 
441                             + ex.getTargetException()); 
442          ex.getTargetException().printStackTrace(); 
443          continue; 
444        } 
445        catch (Exception ex){ 
446          System.err.println("Skipping property " + name 
447                             + " ; exception: " + ex); 
448          ex.printStackTrace(); 
449          continue; 
450        } 
451   
452        m_Labels[i] = new JLabel(name, SwingConstants.RIGHT); 
453        m_Labels[i].setBorder(BorderFactory.createEmptyBorder(10, 10, 0, 5)); 
454        m_Views[i] = view; 
455        GridBagConstraints gbConstraints = new GridBagConstraints(); 
456        gbConstraints.anchor = GridBagConstraints.EAST; 
457        gbConstraints.fill = GridBagConstraints.HORIZONTAL; 
458        gbConstraints.gridy = i + componentOffset; 
459        gbConstraints.gridx = 0; 
460        gbLayout.setConstraints(m_Labels[i], gbConstraints); 
461        add(m_Labels[i]); 
462        JPanel newPanel = new JPanel(); 
463        if (m_TipTexts[i] != null){ 
464          m_Views[i].setToolTipText(m_TipTexts[i]); 
465        } 
466        newPanel.setBorder(BorderFactory.createEmptyBorder(10, 5, 0, 10)); 
467        newPanel.setLayout(new BorderLayout()); 
468        newPanel.add(m_Views[i], BorderLayout.CENTER); 
469        gbConstraints = new GridBagConstraints(); 
470        gbConstraints.anchor = GridBagConstraints.WEST; 
471        gbConstraints.fill = GridBagConstraints.BOTH; 
472        gbConstraints.gridy = i + componentOffset; 
473        gbConstraints.gridx = 1; 
474        gbConstraints.weightx = 100; 
475        gbLayout.setConstraints(newPanel, gbConstraints); 
476        add(newPanel); 
477        m_NumEditable++; 
478      } 
479      if (m_NumEditable == 0){ 
480        JLabel empty = new JLabel("No editable properties", 
481                                  SwingConstants.CENTER); 
482        Dimension d = empty.getPreferredSize(); 
483        empty.setPreferredSize(new Dimension(d.width * 2, d.height * 2)); 
484        empty.setBorder(BorderFactory.createEmptyBorder(10, 5, 0, 10)); 
485        GridBagConstraints gbConstraints = new GridBagConstraints(); 
486        gbConstraints.anchor = GridBagConstraints.CENTER; 
487        gbConstraints.fill = GridBagConstraints.HORIZONTAL; 
488        gbConstraints.gridy = componentOffset; 
489        gbConstraints.gridx = 0; 
490        gbLayout.setConstraints(empty, gbConstraints); 
491        add(empty); 
492      } 
493   
494      validate(); 
495      setVisible(true); 
496    }

498    protected void openHelpFrame(){ 
499   
500      JTextArea ta = new JTextArea(); 
501      ta.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5)); 
502      ta.setLineWrap(true); 
503      ta.setWrapStyleWord(true); 
504      //ta.setBackground(getBackground()); 
505      ta.setEditable(false); 
506      ta.setText(m_HelpText.toString()); 
507      ta.setCaretPosition(0); 
508      final JDialog jf; 
509      if(getTopLevelAncestor() instanceof JDialog) 
510        jf = new JDialog((JDialog)getTopLevelAncestor()); 
511      else 
512        jf = new JDialog(); 
513      jf.setTitle("Information"); 
514      jf.addWindowListener(new WindowAdapter(){ 
515        public void windowClosing(WindowEvent e){ 
516          jf.dispose(); 
517          if (m_HelpFrame == jf){ 
518            m_HelpBut.setEnabled(true); 
519          } 
520        } 
521      }); 
522      jf.getContentPane().setLayout(new BorderLayout()); 
523      jf.getContentPane().add(new JScrollPane(ta), BorderLayout.CENTER); 
524      jf.pack(); 
525      jf.setSize(400, 350); 
526      jf.setLocation(getTopLevelAncestor().getLocationOnScreen().x 
527                     + getTopLevelAncestor().getSize().width, 
528                     getTopLevelAncestor().getLocationOnScreen().y); 
529      jf.setVisible(true); 
530      jf.setModal(false); 
531      m_HelpFrame = jf; 
532    }

545    /** 
546     * Updates the propertysheet when a value has been changed (from outside 
547     * the propertysheet?). 
548     * 
549     * @param evt a value of type 'PropertyChangeEvent' 
550     */ 
551    synchronized void wasModified(PropertyChangeEvent evt){ 
552   
553      //    System.err.println("wasModified"); 
554      if (evt.getSource() instanceof PropertyEditor){ 
555        PropertyEditor editor = (PropertyEditor) evt.getSource(); 
556        for (int i = 0; i < m_Editors.length; i++){ 
557          if (m_Editors[i] == editor){ 
558            PropertyDescriptor property = m_Properties[i]; 
559            Object value = editor.getValue(); 
560            m_Values[i] = value; 
561            Method setter = property.getWriteMethod(); 
562            try{ 
563              Object args[] = {value}; 
564              args[0] = value; 
565              setter.invoke(m_Target, args); 
566            } 
567            catch (InvocationTargetException ex){ 
568              if (ex.getTargetException() 
569                instanceof PropertyVetoException){ 
570                if(m_VerboseMode) 
571                  System.err.println("WARNING: Vetoed; reason is: " 
572                                     + ex.getTargetException().getMessage()); 
573              } 
574              else{ 
575                System.err.println("InvocationTargetException while updating " 
576                                   + property.getName()); 
577              } 
578            } 
579            catch (Exception ex){ 
580              System.err.println("Unexpected exception while updating " 
581                                 + property.getName()); 
582            } 
583            if (m_Views[i] != null && m_Views[i] instanceof PropertyPanel){ 
584              //System.err.println("Trying to repaint the property canvas"); 
585              m_Views[i].repaint(); 
586              revalidate(); 
587            } 
588            break; 
589          } 
590        } 
591      } 
592   
593      // Now re-read all the properties and update the editors 
594      // for any other properties that have changed. 
595      for (int i = 0; i < m_Properties.length; i++){ 
596        Object o; 
597        try{ 
598          Method getter = m_Properties[i].getReadMethod(); 
599          Method setter = m_Properties[i].getWriteMethod(); 
600   
601          if (getter == null || setter == null){ 
602            // ignore set/get only properties 
603            continue; 
604          } 
605   
606          Object args[] = {}; 
607          o = getter.invoke(m_Target, args); 
608        } 
609        catch (Exception ex){ 
610          o = null; 
611        } 
612        if (o == m_Values[i] || (o != null && o.equals(m_Values[i]))){ 
613          // The property is equal to its old value. 
614          continue; 
615        } 
616        m_Values[i] = o; 
617        // Make sure we have an editor for this property... 
618        if (m_Editors[i] == null){ 
619          continue; 
620        } 
621        // The property has changed!  Update the editor. 
622        m_Editors[i].removePropertyChangeListener(this); 
623        m_Editors[i].setValue(o); 
624        m_Editors[i].addPropertyChangeListener(this); 
625        if (m_Views[i] != null){ 
626          //System.err.println("Trying to repaint " + (i + 1)); 
627          m_Views[i].repaint(); 
628        } 
629      } 
630   
631      // Make sure the target bean gets repainted. 
632      if (Beans.isInstanceOf(m_Target, Component.class)){ 
633        ((Component) (Beans.getInstanceOf(m_Target, Component.class))).repaint(); 
634      } 
635    }
