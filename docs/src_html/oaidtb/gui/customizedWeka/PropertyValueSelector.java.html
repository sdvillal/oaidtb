24   package oaidtb.gui.customizedWeka; 
25    
26   import javax.swing.JComboBox; 
27   import javax.swing.ComboBoxModel; 
28   import javax.swing.DefaultComboBoxModel; 
29   import java.awt.event.ItemListener; 
30   import java.awt.event.ItemEvent; 
31   import java.beans.PropertyEditor; 
32    
33   /** 
34    * Support for any PropertyEditor that uses tags. 
35    * 
36    * @author Len Trigg (trigg@cs.waikato.ac.nz) 
37    * @version $Revision: 1.5 $ 
38    */ 
39   class PropertyValueSelector extends JComboBox { 
40    
41     /** The property editor */ 
42     PropertyEditor m_Editor; 
43    
44     /** 
45      * Sets up the editing component with the supplied editor. 
46      * 
47      * @param pe the PropertyEditor 
48      */ 
49     PropertyValueSelector(PropertyEditor pe) { 
50    
51       m_Editor = pe; 
52       Object value = m_Editor.getAsText(); 
53       String tags[] = m_Editor.getTags(); 
54       ComboBoxModel model = new DefaultComboBoxModel(tags) { 
55         public Object getSelectedItem() { 
56   	return m_Editor.getAsText(); 
57         } 
58         public void setSelectedItem(Object o) { 
59   	m_Editor.setAsText((String)o); 
60         } 
61       }; 
62       setModel(model); 
63       setSelectedItem(value); 
64     } 
65   }
