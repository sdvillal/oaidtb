33   /** 
34    * Support for a PropertyEditor that uses text. 
35    * Isn't going to work well if the property gets changed 
36    * somewhere other than this field simultaneously 
37    * 
38    * @author Len Trigg (trigg@cs.waikato.ac.nz) 
39    * @version $Revision: 1.5 $ 
40    */ 
41   class PropertyText extends JTextField{ 
42    
43     /** The property editor */ 
44     private PropertyEditor m_Editor; 
45    
46     /** 
47      * Sets up the editing component with the supplied editor. 
48      * 
49      * @param pe the PropertyEditor 
50      */ 
51     PropertyText(PropertyEditor pe){ 
52    
53       super(pe.getAsText()); 
54       m_Editor = pe; 
55    
56       /*    m_Editor.addPropertyChangeListener(new PropertyChangeListener() { 
57         public void propertyChange(PropertyChangeEvent evt) { 
58   	updateUs(); 
59         } 
60         }); */ 
61       addKeyListener(new KeyAdapter(){ 
62         public void keyReleased(KeyEvent e){ 
63           //	if (e.getKeyCode() == KeyEvent.VK_ENTER) { 
64           updateEditor(); 
65           //	} 
66         } 
67       }); 
68       addFocusListener(new FocusAdapter(){ 
69         public void focusLost(FocusEvent e){ 
70           updateEditor(); 
71         } 
72       }); 
73     } 
74    
75     /** 
76      * Attempts to update the textfield value from the editor. 
77      */ 
78     protected void updateUs(){ 
79       try{ 
80         setText(m_Editor.getAsText()); 
81       } 
82       catch (IllegalArgumentException ex){ 
83         // Quietly ignore. 
84       } 
85     } 
86    
87     /** 
88      * Attempts to update the editor value from the textfield. 
89      */ 
90     protected void updateEditor(){ 
91       try{ 
92         m_Editor.setAsText(getText()); 
93       } 
94       catch (IllegalArgumentException ex){ 
95         // Quietly ignore. 
96       } 
97     } 
98   }
