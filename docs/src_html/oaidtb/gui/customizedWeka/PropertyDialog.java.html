29   package oaidtb.gui.customizedWeka; 
30    
31   import javax.swing.*; 
32   import java.awt.*; 
33   import java.awt.dnd.DropTarget; 
34   import java.awt.event.WindowAdapter; 
35   import java.awt.event.WindowEvent; 
36   import java.beans.PropertyEditor; 
37    
38   /** 
39    * Support for PropertyEditors with custom editors: puts the editor into 
40    * a separate frame. 
41    * 
42    * @author Len Trigg (trigg@cs.waikato.ac.nz) 
43    * @author Santiago David Villalna (sdvb@wanadoo.es) 
44    * @version $Revision: 1.5 $ 
45    */ 
46   public class PropertyDialog extends JDialog{ 
47    
48     /** The property editor */ 
49     private PropertyEditor m_Editor; 
50    
51     /** The custom editor component */ 
52     private Component m_EditorComponent; 
53    
54     /** 
55      * Creates the editor frame. 
56      * 
57      * @param owner The owner of the dialog 
58      * @param pe the PropertyEditor 
59      * @param x initial x coord for the frame 
60      * @param y initial y coord for the frame 
61      */ 
62     public PropertyDialog(Frame owner, PropertyEditor pe, int x, int y){ 
63    
64       super(owner); 
65       initialize(pe, x, y); 
66     } 
67    
68     /** 
69      * Creates the editor frame. 
70      * 
71      * @param owner The owner of the dialog 
72      * @param pe the PropertyEditor 
73      * @param x initial x coord for the frame 
74      * @param y initial y coord for the frame 
75      */ 
76     public PropertyDialog(JDialog owner, PropertyEditor pe, int x, int y){ 
77    
78       super(owner); 
79       initialize(pe, x, y); 
80     } 
81    
82     private void initialize(PropertyEditor pe, int x, int y){ 
83       addWindowListener(new WindowAdapter(){ 
84         public void windowClosing(WindowEvent e){ 
85           e.getWindow().dispose(); 
86         } 
87       }); 
88       getContentPane().setLayout(new BorderLayout()); 
89    
90       m_Editor = pe; 
91       m_EditorComponent = pe.getCustomEditor(); 
92       getContentPane().add(m_EditorComponent, BorderLayout.CENTER); 
93    
94       pack(); 
95       setTitle(pe.getClass().getName()); 
96       setModal(true); 
97       setLocation(x, y); 
98       setVisible(true); 
99     } 
100   
101    /** 
102     * Gets the current property editor. 
103     * 
104     * @return a value of type 'PropertyEditor' 
105     */ 
106    public PropertyEditor getEditor(){ 
107   
108      return m_Editor; 
109    } 
110  }
