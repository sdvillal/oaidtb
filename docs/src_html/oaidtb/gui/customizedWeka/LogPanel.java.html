50   /** 
51    * This panel allows log and status messages to be posted. Log messages

66     /** Displays the log messages */ 
67     private JTextPane m_LogText = new JTextPane(){ 
68       /** 
69        * Evitar que se trague el foco; no lo evita si pinchamos sobre ella, mejor 
70        * especificar cuál es el siguiente compoenete que recibirá el foco ( o mandarlo 
71        * a cycleRoot); este problema está solucionado a partir del JDK 1.4.0_01

73        * @return false

83     /** 
84      * Creates the log panel 
85      */ 
86     public LogPanel(){ 
87       m_LogText.setEditable(false); 
88       m_LogText.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5)); 
89       m_StatusLab.setBorder(BorderFactory.createCompoundBorder( 
90         BorderFactory.createTitledBorder("Status"), 
91         BorderFactory.createEmptyBorder(0, 5, 5, 5))); 
92       JPanel p1 = new JPanel(); 
93       p1.setBorder(BorderFactory.createTitledBorder("Log")); 
94       p1.setLayout(new BorderLayout()); 
95       final JScrollPane js = new JScrollPane(m_LogText); 
96       p1.add(js, BorderLayout.CENTER); 
97       js.getViewport().addChangeListener(new ChangeListener(){ 
98         private int lastHeight; 
99    
100        public void stateChanged(ChangeEvent e){ 
101          JViewport vp = (JViewport) e.getSource(); 
102          int h = vp.getViewSize().height; 
103          if (h != lastHeight){ // i.e. an addition not just a user scrolling 
104            lastHeight = h; 
105            int x = h - vp.getExtentSize().height; 
106            vp.setViewPosition(new Point(0, x)); 
107          } 
108        } 
109      }); 
110      setLayout(new BorderLayout()); 
111      add(p1, BorderLayout.CENTER); 
112      JPanel p2 = new JPanel(new BorderLayout()); 
113      p2.add(m_CustomPanel, BorderLayout.WEST); 
114      p2.add(m_StatusLab, BorderLayout.CENTER); 
115      add(p2,BorderLayout.SOUTH); 
116      addPopups(); 
117    }

171    /** 
172     * Add a popup menu for displaying the amount of free memory 
173     * and running the garbage collector, and another 
174     * to copy the log text to the system clipboard. 
175     */ 
176    private void addPopups(){ 
177      m_LogText.addMouseListener(new MouseAdapter(){ 
178        public void mouseClicked(MouseEvent e){ 
179          if (SwingUtilities.isRightMouseButton(e)){ 
180   
181            JPopupMenu popupMenu = new JPopupMenu(); 
182   
183            JMenuItem clear = new JMenuItem("Clear the text"); 
184            clear.addActionListener(new ActionListener(){ 
185              public void actionPerformed(ActionEvent e2){ 
186                m_LogText.setText(""); 
187              } 
188            }); 
189            popupMenu.add(clear); 
190   
191            JMenuItem copyToClipBoard = new JMenuItem("Copy the text to the clipboard"); 
192            copyToClipBoard.addActionListener(new ActionListener(){ 
193              public void actionPerformed(ActionEvent e2){ 
194                String text = m_LogText.getSelectedText(); 
195                if(text == null) 
196                  text = m_LogText.getText(); 
197                StringSelection ss = new StringSelection(text); 
198                Toolkit.getDefaultToolkit().getSystemClipboard().setContents(ss, null); 
199              } 
200            }); 
201            popupMenu.add(copyToClipBoard); 
202   
203            JMenuItem availMem = new JMenuItem("Available memory"); 
204            availMem.addActionListener(new ActionListener(){ 
205              public void actionPerformed(ActionEvent ee){ 
206                System.gc(); 
207                Runtime currR = Runtime.getRuntime(); 
208                long freeM = currR.freeMemory(); 
209                logMessage("Available memory : " + freeM + " bytes"); 
210              } 
211            }); 
212            popupMenu.add(availMem); 
213   
214            JMenuItem runGC = new JMenuItem("Run garbage collector"); 
215            runGC.addActionListener(new ActionListener(){ 
216              public void actionPerformed(ActionEvent ee){ 
217                statusMessage("Running garbage collector"); 
218                System.gc(); 
219                statusMessage("OK"); 
220              } 
221            }); 
222            popupMenu.add(runGC); 
223   
224            popupMenu.show(m_LogText, e.getX(), e.getY()); 
225          } 
226        } 
227      }); 
228    }

248    /** 
249     * Gets a string containing current date and time. 
250     * 
251     * @return a string containing the date and time. 
252     */ 
253    protected static String getTimestamp(){ 
254   
255      return (new SimpleDateFormat("HH:mm:ss:")).format(new Date()); 
256    } 
257   
258    /** 
259     * Sends the supplied message to the log area. The current timestamp will 
260     * be prepended. 
261     * 
262     * @param message a value of type 'String' 
263     */ 
264    public void logMessage(String message){ 
265   
266      StyledDocument doc = (StyledDocument) m_LogText.getDocument(); 
267   
268      try{ 
269        doc.insertString(doc.getLength(), LogPanel.getTimestamp() + ' ' + message +"\n", null); 
270        m_LogText.setCaretPosition(doc.getLength()); 
271      } 
272      catch(BadLocationException e){} 
273    }

277     * 
278     * @param message the status message 
279     */ 
280    public void statusMessage(String message){ 
281   
282      m_StatusLab.setText(message); 
283    }
