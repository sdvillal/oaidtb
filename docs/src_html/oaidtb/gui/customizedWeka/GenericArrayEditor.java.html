83     /** Listens to buttons being pressed and taking the appropriate action */ 
84     private ActionListener m_InnerActionListener = 
85       new ActionListener(){ 
86    
87         public void actionPerformed(ActionEvent e){ 
88    
89           if (e.getSource() == m_DeleteBut){ 
90             int[] selected = m_ElementList.getSelectedIndices(); 
91             if (selected != null){ 
92               for (int i = 0; i < selected.length; i++){ 
93                 int current = selected[i]; 
94                 m_ListModel.removeElementAt(current); 
95                 if (m_ListModel.size() > current){ 
96                   m_ElementList.setSelectedIndex(current); 
97                 } 
98               } 
99               m_Support.firePropertyChange("", null, null); 
100            } 
101            if (m_ElementList.getSelectedIndex() == -1){ 
102              m_DeleteBut.setEnabled(false); 
103            } 
104          } 
105          else if (e.getSource() == m_AddBut){ 
106            int selected = m_ElementList.getSelectedIndex(); 
107            Object addObj = m_ElementEditor.getValue(); 
108   
109            // Make a full copy of the object using serialization 
110            try{ 
111              SerializedObject so = new SerializedObject(addObj); 
112              addObj = so.getObject(); 
113              if (selected != -1){ 
114                m_ListModel.insertElementAt(addObj, selected); 
115              } 
116              else{ 
117                m_ListModel.addElement(addObj); 
118              } 
119              m_Support.firePropertyChange("", null, null); 
120            } 
121            catch (Exception ex){ 
122              JOptionPane.showMessageDialog(GenericArrayEditor.this, 
123                                            "Could not create an object copy", 
124                                            null, 
125                                            JOptionPane.ERROR_MESSAGE); 
126            } 
127          } 
128        } 
129      };

161    /* This class handles the creation of list cell renderers from the 
162     * property editors. 
163     */ 
164    private class EditorListCellRenderer implements ListCellRenderer{ 
165   
166      /** The class of the property editor for array objects */ 
167      private Class m_EditorClass; 
168   
169      /** The class of the array values */ 
170      private Class m_ValueClass; 
171   
172      /** 
173       * Creates the list cell renderer. 
174       * 
175       * @param editorClass The class of the property editor for array objects 
176       * @param valueClass The class of the array values 
177       */ 
178      public EditorListCellRenderer(Class editorClass, Class valueClass){ 
179        m_EditorClass = editorClass; 
180        m_ValueClass = valueClass; 
181      } 
182   
183      /** 
184       * Creates a cell rendering component. 
185       * 
186       * @param list the list that will be rendered in 
187       * @param value the cell value 
188       * @param index which element of the list to render 
189       * @param isSelected true if the cell is selected 
190       * @param cellHasFocus true if the cell has the focus 
191       * @return the rendering component 
192       */ 
193      public Component getListCellRendererComponent(final JList list, 
194                                                    final Object value, 
195                                                    final int index, 
196                                                    final boolean isSelected, 
197                                                    final boolean cellHasFocus){ 
198        try{ 
199          final PropertyEditor e = (PropertyEditor) m_EditorClass.newInstance(); 
200          if (e instanceof GenericObjectEditor){ 
201            ((GenericObjectEditor) e).setClassType(m_ValueClass); 
202          } 
203          else if (e instanceof weka.gui.GenericObjectEditor){ 
204            //	  ((GenericObjectEditor) e).setDisplayOnly(true); 
205            ((weka.gui.GenericObjectEditor) e).setClassType(m_ValueClass); 
206          } 
207          e.setValue(value); 
208          return new JPanel(){ 
209   
210            public void paintComponent(Graphics g){ 
211   
212              Insets i = this.getInsets(); 
213              Rectangle box = new Rectangle(i.left, i.top, 
214                                            this.getWidth() - i.right, 
215                                            this.getHeight() - i.bottom); 
216              g.setColor(isSelected 
217                         ? list.getSelectionBackground() 
218                         : list.getBackground()); 
219              g.fillRect(0, 0, this.getWidth(), this.getHeight()); 
220              g.setColor(isSelected 
221                         ? list.getSelectionForeground() 
222                         : list.getForeground()); 
223              e.paintValue(g, box); 
224            } 
225   
226            public Dimension getPreferredSize(){ 
227   
228              Font f = this.getFont(); 
229              FontMetrics fm = this.getFontMetrics(f); 
230              return new Dimension(0, fm.getHeight()); 
231            } 
232          }; 
233        } 
234        catch (Exception ex){ 
235          return null; 
236        } 
237      } 
238    }

240    /** 
241     * Updates the type of object being edited, so attempts to find an 
242     * appropriate propertyeditor. 
243     * 
244     * @param o a value of type 'Object' 
245     */ 
246    private void updateEditorType(Object o){ 
247   
248      // Determine if the current object is an array 
249      m_ElementEditor = null; 
250      m_ListModel = null; 
251      removeAll(); 
252      if ((o != null) && (o.getClass().isArray())){ 
253        Class elementClass = o.getClass().getComponentType(); 
254        PropertyEditor editor = PropertyEditorManager.findEditor(elementClass); 
255        Component view = null; 
256        ListCellRenderer lcr = new DefaultListCellRenderer(); 
257        if (editor != null){ 
258          if (editor instanceof GenericObjectEditor){ 
259            ((GenericObjectEditor) editor).setClassType(elementClass); 
260          } 
261          else if (editor instanceof weka.gui.GenericObjectEditor){ 
262            //	  ((GenericObjectEditor) e).setDisplayOnly(true); 
263            ((weka.gui.GenericObjectEditor) editor).setClassType(elementClass); 
264          } 
265   
266          if (editor.isPaintable() && editor.supportsCustomEditor()){ 
267            view = new PropertyPanel(editor); 
268            lcr = new EditorListCellRenderer(editor.getClass(), elementClass); 
269          } 
270          else if (editor.getTags() != null){ 
271            view = new PropertyValueSelector(editor); 
272          } 
273          else if (editor.getAsText() != null){ 
274            view = new PropertyText(editor); 
275          } 
276        } 
277        if (view == null){ 
278          System.err.println("No property editor for class: " 
279                             + elementClass.getName()); 
280        } 
281        else{ 
282          m_ElementEditor = editor; 
283   
284          // Create the ListModel and populate it 
285          m_ListModel = new DefaultListModel(); 
286          m_ElementClass = elementClass; 
287          for (int i = 0; i < Array.getLength(o); i++){ 
288            m_ListModel.addElement(Array.get(o, i)); 
289          } 
290          m_ElementList.setCellRenderer(lcr); 
291          m_ElementList.setModel(m_ListModel); 
292          if (m_ListModel.getSize() > 0){ 
293            m_ElementList.setSelectedIndex(0); 
294            m_DeleteBut.setEnabled(true); 
295          } 
296          else{ 
297            m_DeleteBut.setEnabled(false); 
298          } 
299   
300          try{ 
301            if (m_ListModel.getSize() > 0){ 
302              m_ElementEditor.setValue(m_ListModel.getElementAt(0)); 
303            } 
304            else{ 
305              if (m_ElementEditor instanceof GenericObjectEditor){ 
306                ((GenericObjectEditor) m_ElementEditor).setDefaultValue(); 
307              } 
308              else if (m_ElementEditor instanceof weka.gui.GenericObjectEditor){ 
309                //	  ((GenericObjectEditor) e).setDisplayOnly(true); 
310                ((weka.gui.GenericObjectEditor) m_ElementEditor).setDefaultValue(); 
311              } 
312              else{ 
313                m_ElementEditor.setValue(m_ElementClass.newInstance()); 
314              } 
315            } 
316   
317            JPanel panel = new JPanel(); 
318            panel.setLayout(new BorderLayout()); 
319            panel.add(view, BorderLayout.CENTER); 
320            panel.add(m_AddBut, BorderLayout.EAST); 
321            add(panel, BorderLayout.NORTH); 
322            add(new JScrollPane(m_ElementList), BorderLayout.CENTER); 
323            add(m_DeleteBut, BorderLayout.SOUTH); 
324            m_ElementEditor 
325              .addPropertyChangeListener(new PropertyChangeListener(){ 
326                public void propertyChange(PropertyChangeEvent e){ 
327                  repaint(); 
328                } 
329              }); 
330          } 
331          catch (Exception ex){ 
332            System.err.println(ex.getMessage()); 
333            m_ElementEditor = null; 
334          } 
335        } 
336      } 
337      if (m_ElementEditor == null){ 
338        add(m_Label, BorderLayout.CENTER); 
339      } 
340      m_Support.firePropertyChange("", null, null); 
341      validate(); 
342    }

355    /** 
356     * Gets the current object array. 
357     * 
358     * @return the current object array 
359     */ 
360    public Object getValue(){ 
361   
362      if (m_ListModel == null){ 
363        return null; 
364      } 
365      // Convert the listmodel to an array of strings and return it. 
366      int length = m_ListModel.getSize(); 
367      Object result = Array.newInstance(m_ElementClass, length); 
368      for (int i = 0; i < length; i++){ 
369        Array.set(result, i, m_ListModel.elementAt(i)); 
370      } 
371      return result; 
372    }

397    /** 
398     * Paints a representation of the current classifier. 
399     * 
400     * @param gfx the graphics context to use 
401     * @param box the area we are allowed to paint into 
402     */ 
403    public void paintValue(Graphics gfx, Rectangle box){ 
404   
405      FontMetrics fm = gfx.getFontMetrics(); 
406      int vpad = (box.height - fm.getAscent()) / 2; 
407      String rep = m_ListModel.getSize() + " " + m_ElementClass.getName(); 
408      gfx.drawString(rep, 2, fm.getHeight() + vpad); 
409    }
