135        EDITOR_PROPERTIES = Utils.readProperties(PROPERTY_FILE); 
136        java.util.Enumeration keys = 
137          (java.util.Enumeration) EDITOR_PROPERTIES.propertyNames();

145                                      "Could not read a configuration file for the generic object\n" 
146                                      + "editor. An example file is included with the Weka distribution.\n" 
147                                      + "This file should be named \"" + PROPERTY_FILE + "\" and\n" 
148                                      + "should be placed either in your user home (which is set\n" 
149                                      + "to \"" + System.getProperties().getProperty("user.home") + "\")\n"

390      /** 
391       * Opens an object from a file selected by the user. 
392       * 
393       * @return the loaded object, or null if the operation was cancelled 
394       */ 
395      protected Object openObject(){ 
396   
397        if (m_FileChooser == null){ 
398          createFileChooser(); 
399        } 
400        int returnVal = m_FileChooser.showOpenDialog(this); 
401        if (returnVal == JFileChooser.APPROVE_OPTION){ 
402          File selected = m_FileChooser.getSelectedFile(); 
403          try{ 
404            ObjectInputStream oi = new ObjectInputStream(new BufferedInputStream(new FileInputStream(selected))); 
405            Object obj = oi.readObject(); 
406            oi.close(); 
407            if (!m_ClassType.isAssignableFrom(obj.getClass())){ 
408              throw new Exception("Object not of type: " + m_ClassType.getName()); 
409            } 
410            return obj; 
411          } 
412          catch (Exception ex){ 
413            JOptionPane.showMessageDialog(this, 
414                                          "Couldn't read object: " 
415                                          + selected.getName() 
416                                          + "\n" + ex.getMessage(), 
417                                          "Open object file", 
418                                          JOptionPane.ERROR_MESSAGE); 
419          } 
420        } 
421        return null; 
422      }

433        if (returnVal == JFileChooser.APPROVE_OPTION){ 
434          File sFile = m_FileChooser.getSelectedFile(); 
435          try{ 
436            ObjectOutputStream oo = new ObjectOutputStream(new BufferedOutputStream(new FileOutputStream(sFile))); 
437            oo.writeObject(object); 
438            oo.close(); 
439          } 
440          catch (Exception ex){ 
441            JOptionPane.showMessageDialog(this, 
442                                          "Couldn't write to file: " 
443                                          + sFile.getName() 
444                                          + "\n" + ex.getMessage(), 
445                                          "Save object", 
446                                          JOptionPane.ERROR_MESSAGE); 
447          } 
448        } 
449      }

563    /** Called when the class of object being edited changes. */ 
564    protected String getClassesFromProperties(){ 
565   
566      String className = m_ClassType.getName(); 
567      String typeOptions = null; 
568   
569      if (m_SpecializeFromClass != null) 
570        typeOptions = EDITOR_PROPERTIES.getProperty(className + "@" + m_SpecializeFromClass); 
571   
572      if (typeOptions == null) 
573        typeOptions = EDITOR_PROPERTIES.getProperty(className); 
574   
575      if (typeOptions == null){ 
576        System.err.println("Warning: No configuration property found in\n" 
577                           + PROPERTY_FILE + "\n" 
578                           + "for " + className); 
579      } 
580      return typeOptions; 
581    }

644    /** 
645     * Sets the current Object. If the Object is in the 
646     * Object chooser, this becomes the selected item (and added 
647     * to the chooser if necessary). 
648     * 
649     * @param o an object that must be a Object. 
650     */ 
651    public void setValue(Object o){ 
652   
653      if (m_ClassType == null){ 
654        System.err.println("No ClassType set up for GenericObjectEditor!!"); 
655        return; 
656      } 
657   
658      if (!m_ClassType.isAssignableFrom(o.getClass())){ 
659        System.err.println("setValue object not of correct type!"); 
660        return; 
661      } 
662   
663      setObject(o); 
664   
665      if (m_EditorComponent != null) 
666        m_EditorComponent.updateChooser(); 
667   
668      //If Class.forName(selected).newInstance() fails in the next call, m_Backup will 
669      //not be updated to the last succesfully open object, it will turn back two calls 
670      //unless we do this. 
671      m_Backup = m_Object; 
672    }

674    /** 
675     * Sets the current Object, but doesn't worry about updating 
676     * the state of the object chooser. 
677     * 
678     * @param c a value of type 'Object' 
679     */ 
680    private void setObject(Object c){ 
681   
682  // This should really call equals() for comparison. 
683      boolean trueChange; 
684      if (getValue() != null){ 
685        trueChange = (!c.equals(getValue())); 
686        //System.err.println("GEO::setObject(): Changed? " + trueChange+ getValue().getClass().getName()); 
687      } 
688      else 
689        trueChange = true; 
690   
691      m_Object = c; 
692   
693      if (m_EditorComponent != null){ 
694        m_EditorComponent.updateChildPropertySheet(); 
695        if (trueChange){ 
696          m_Support.firePropertyChange("", null, null); 
697        } 
698      } 
699    }

755    public String getObjectConfigurationText(){ 
756      if (!(m_Enabled && m_Object != null)) 
757        return ""; 
758   
759      String rep = m_Object.getClass().getName(); 
760      int dotPos = rep.lastIndexOf('.'); 
761      if (dotPos != -1){ 
762        rep = rep.substring(dotPos + 1); 
763      } 
764      if (m_Object instanceof OptionHandler) 
765        rep += " " + Utils.joinOptions(((OptionHandler) m_Object).getOptions()); 
766      return rep; 
767    }
