45   public class PropertyPanel extends JPanel{ 
46    
47     /** The property editor */ 
48     private PropertyEditor m_Editor; 
49    
50     /** The currently displayed property dialog, if any */ 
51     private oaidtb.gui.customizedWeka.PropertyDialog m_PD; 
52    
53     /** The appearance when this panel has the focus. */ 
54     private Border m_WithFocusBorder; 
55    
56     /** The appearance when this panel doesn't have the focus. */ 
57     private Border m_WithoutFocusBorder; 
58    
59     /** 
60      * Create the panel with the supplied property editor. 
61      * 
62      * @param pe the PropertyEditor 
63      */ 
64     public PropertyPanel(PropertyEditor pe){ 
65    
66       m_Editor = pe; 
67    
68       m_WithoutFocusBorder = BorderFactory.createEtchedBorder(); 
69       m_WithFocusBorder = BorderFactory.createLineBorder(Color.black); 
70    
71       setBorder(m_WithoutFocusBorder); 
72    
73       setToolTipText("Click to edit properties for this object"); 
74    
75       setOpaque(true); 
76    
77       //m_Editor.getAsText() != null 
78       if (m_Editor instanceof oaidtb.gui.customizedWeka.GenericObjectEditor) 
79         addMouseListener(new MouseAdapter(){ 
80           public void mousePressed(MouseEvent evt){ 
81             if (!isEnabled()) 
82               return; 
83             if (SwingUtilities.isRightMouseButton(evt)) 
84               showPopUp(evt); 
85             else 
86               showPropertyDialog(getLocationOnScreen().x, getLocationOnScreen().y); 
87           } 
88         }); 
89       else 
90         addMouseListener(new MouseAdapter(){ 
91           public void mousePressed(MouseEvent evt){ 
92             if (!isEnabled()) 
93               return; 
94             if (SwingUtilities.isLeftMouseButton(evt)) 
95               showPropertyDialog(getLocationOnScreen().x, getLocationOnScreen().y); 
96           } 
97         }); 
98    
99       addKeyListener(new KeyAdapter(){ 
100        public void keyPressed(KeyEvent e){ 
101          if (!isEnabled()) 
102            return; 
103          if (e.getKeyCode() == KeyEvent.VK_ENTER || e.getKeyCode() == KeyEvent.VK_SPACE) 
104            showPropertyDialog(getLocationOnScreen().x, getLocationOnScreen().y); 
105        } 
106      }); 
107   
108      addFocusListener(new FocusListener(){ 
109        public void focusGained(FocusEvent e){ 
110          setBorder(m_WithFocusBorder); 
111        } 
112   
113        public void focusLost(FocusEvent e){ 
114          setBorder(m_WithoutFocusBorder); 
115        } 
116      }); 
117   
118      Dimension newPref = getPreferredSize(); 
119      newPref.height = getFontMetrics(getFont()).getHeight() * 5 / 4; 
120      newPref.width = newPref.height * 5; 
121      setPreferredSize(newPref); 
122    } 
123   
124    public void showPropertyDialog(int x, int y){ 
125      if(!isEnabled()) 
126        return; 
127      if (m_Editor.getValue() != null){ 
128        if (m_PD == null){ 
129          //Tenemos que hacer que el diálogo interno pertenezca al Frame principal de la 
130          //aplicación o a un JDialog subordinado para que al 
131          //cambiar de aplicación y volver no se pierda el mismo, bloqueando la aplicación 
132          //si es modal (se podría recuperar en ese caso mediante el cambio de ventana con el teclado) 
133          Container container = getTopLevelAncestor(); 
134          while(container instanceof JInternalFrame) 
135            container = ((JInternalFrame)container).getTopLevelAncestor(); 
136          if(container instanceof JDialog) 
137            m_PD = new oaidtb.gui.customizedWeka.PropertyDialog((JDialog)container, m_Editor, x, y); 
138          else 
139            m_PD = new oaidtb.gui.customizedWeka.PropertyDialog((JFrame)container, m_Editor, x, y); 
140        } 
141        else{ 
142          m_PD.setLocation(x, y); 
143          m_PD.setVisible(true); 
144        } 
145      } 
146    }

148    /** 
149     * We allow this panel to receive the focus 
150     * 
151     * @return true if this component can receive the focus 
152     */ 
153    public boolean isFocusTraversable(){ 
154      return true; 
155    }

157    /** 
158     * Paints the component, using the property editor's paint method. 
159     * 
160     * @param g the current graphics context 
161     */ 
162    public void paintComponent(Graphics g){ 
163   
164      Insets i = getInsets(); 
165      Rectangle box = new Rectangle(i.left, i.top, 
166                                    getSize().width - i.left - i.right - 1, 
167                                    getSize().height - i.top - i.bottom - 1); 
168   
169      g.clearRect(i.left, i.top, 
170                  getSize().width - i.right - i.left, 
171                  getSize().height - i.bottom - i.top); 
172      m_Editor.paintValue(g, box); 
173    } 
174   
175    /** 
176     * Shows a popup menu that allows the user to put a textual representation of the object 
177     * being represented in the system clipboard. 
178     * 
179     * @param e The mouse event that provocates the action 
180     */ 
181    private void showPopUp(MouseEvent e){ 
182      JPopupMenu popupMenu = new JPopupMenu(); 
183      JMenuItem copyToClipBoard = new JMenuItem("Copy the text to the clipboard"); 
184      copyToClipBoard.addActionListener(new ActionListener(){ 
185        public void actionPerformed(ActionEvent e2){ 
186          String obj = ((GenericObjectEditor) m_Editor).getObjectName(); 
187          String ops = ((GenericObjectEditor) m_Editor).getObjectOptions(); 
188          if (ops != "") 
189            obj += " " + ops; 
190   
191          StringSelection ss = new StringSelection(obj); 
192   
193          Toolkit.getDefaultToolkit().getSystemClipboard().setContents(ss, null); 
194        } 
195      }); 
196      popupMenu.add(copyToClipBoard); 
197      popupMenu.show(this, e.getX(), e.getY()); 
198    }
