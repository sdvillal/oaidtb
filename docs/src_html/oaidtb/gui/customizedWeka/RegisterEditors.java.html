25   /** 
26    * Clase que sirve para registrar los editores de propiedades necesarios para las 
27    * nuevas clases que queramos especificar que las trate el GOE; 
28    * para no tener que estar trasteando con los fuentes del explorador de weka etc. 
29    * 
30    * Se debe poner bien en el directorio desde donde se ejecuta como una clase sin paquete, 
31    * bien en el paquete al que pertenezca la clase "GenericObjectEditor" en cuestión 
32    * 
33    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
34    * @version $Revision: 1.0 $ 
35    */ 
36   public class RegisterEditors{ 
37    
38     /* Register the property editors we need */ 
39     static{ 
40       java.beans.PropertyEditorManager.registerEditor(weka.core.SelectedTag.class, 
41                                                       weka.gui.SelectedTagEditor.class); 
42       java.beans.PropertyEditorManager.registerEditor(weka.filters.Filter.class, 
43                                                       oaidtb.gui.customizedWeka.GenericObjectEditor.class); 
44       java.beans.PropertyEditorManager.registerEditor(weka.classifiers.Classifier[].class, 
45                                                       oaidtb.gui.customizedWeka.GenericArrayEditor.class); 
46       java.beans.PropertyEditorManager.registerEditor(weka.classifiers.Classifier.class, 
47                                                       oaidtb.gui.customizedWeka.GenericObjectEditor.class); 
48       java.beans.PropertyEditorManager.registerEditor(weka.classifiers.DistributionClassifier.class, 
49                                                       oaidtb.gui.customizedWeka.GenericObjectEditor.class); 
50       java.beans.PropertyEditorManager.registerEditor(oaidtb.boosters.Booster.class, 
51                                                       oaidtb.gui.customizedWeka.GenericObjectEditor.class); 
52       java.beans.PropertyEditorManager.registerEditor(oaidtb.boosters.costSensitive.AbstractCSB.class, 
53                                                       oaidtb.gui.customizedWeka.GenericObjectEditor.class); 
54       java.beans.PropertyEditorManager.registerEditor(oaidtb.filters.AbstractNominalToOCFilter.class, 
55                                                       oaidtb.gui.customizedWeka.GenericObjectEditor.class); 
56       java.beans.PropertyEditorManager.registerEditor(weka.classifiers.CostMatrix.class, 
57                                                       weka.gui.CostMatrixEditor.class); 
58     }
