23   package oaidtb.misc; 
24    
25   import java.util.*; 
26   import java.io.Serializable; 
27    
28   /** 
29    * Class which defines a customized order of elements; in other words, an index of 
30    * objects over they absolute positions in the order (from 0 to numObjects-1). 
31    * 
32    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
33    * @version $Revision: 1.0 $ 
34    */ 
35   public class CustomOrderDefiner implements Serializable{ 
36    
37     /** Absolute position --> object  
38      * @associates Object*/ 
39     protected ArrayList retrieveByOrder = new ArrayList(); 
40    
41     /** Object --> Absolute position  
42      * @associates SimpleInteger*/ 
43     protected HashMap retrieveByObject = new HashMap(); 
44    
45     /** 
46      * Adds an object with index the number of already added objects - 1. 
47      * 
48      * @param o The object 
49      * 
50      * @return True if the object isn't yet in the index, false otherwise 
51      */ 
52     public boolean add(Object o){ 
53       if(retrieveByObject.containsKey(o)) 
54         return false; 
55    
56       retrieveByObject.put(o, new SimpleInteger(retrieveByObject.size())); 
57       retrieveByOrder.add(o); 
58       return true; 
59     } 
60    
61     /** 
62      * Get the position of the specified object 
63      * 
64      * @param o The object we are searching for its position in the order 
65      * 
66      * @return The position or -1 if the object isn't in the index 
67      */ 
68     public int getIndex(Object o){ 
69       SimpleInteger index = (SimpleInteger) retrieveByObject.get(o); 
70       return(index == null ? -1 : index.value); 
71     } 
72    
73     /** 
74      * Get the object in the specified position 
75      * 
76      * @param index The position of the object 
77      * 
78      * @return The object 
79      */ 
80     public Object getObject(int index){ 
81       return retrieveByOrder.get(index); 
82     } 
83    
84     /** @return The number of objects in the index. */ 
85     public int size(){ 
86       return retrieveByObject.size(); 
87     } 
88    
89     /** 
90      * Append all the elements, not already in the index, to the end of this order, 
91      * following the order defined by "other". 
92      * 
93      * @param other The other ordered objects to be merged with this one 
94      */ 
95     public void mergeWith(CustomOrderDefiner other){ 
96       for(int i=0; i<other.retrieveByOrder.size();i++) 
97         add(other.retrieveByOrder.get(i)); 
98     } 
99   }
