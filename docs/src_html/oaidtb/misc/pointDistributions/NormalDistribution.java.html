23   package oaidtb.misc.pointDistributions; 
24    
25   import edu.cornell.lassp.houle.RngPack.RandomElement; 
26   import cern.jet.random.AbstractDistribution; 
27    
28   /** 
29    * Clase que implementa el interfaz {@link PointDistribution}, generando números 
30    * mediante la distribución normal, utilizando la librería de funciones matemáticas 
31    * Colt que se puede encontrar en la 
32    * 
33    * <p><a href="http://tilde-hoschek.home.cern.ch/~hoschek/colt/" > 
34    * Colt distribution homepage 
35    * </a></p> 
36    * 
37    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
38    * @version $Revision: 1.0 $ 
39    */ 
40   public class NormalDistribution extends cern.jet.random.Normal implements PointDistribution{ 
41    
42     public NormalDistribution(){ 
43       super(1, 20, AbstractDistribution.makeDefaultGenerator()); 
44     } 
45    
46     /** 
47      * Se tomará el centro como la media de la distribución 
48      * 
49      * @param center La media de la distribución 
50      */ 
51     public void setCenter(double center){ 
52       setState(center,standardDeviation); 
53     } 
54    
55     public void setStandardDeviation(double standardDeviation){ 
56       setState(mean,standardDeviation); 
57     } 
58    
59     public double getStandardDeviation(){ 
60       return standardDeviation; 
61     } 
62    
63     public NormalDistribution(double mean, double standardDeviation, RandomElement randomGenerator){ 
64       super(mean, standardDeviation, randomGenerator); 
65     } 
66   }
