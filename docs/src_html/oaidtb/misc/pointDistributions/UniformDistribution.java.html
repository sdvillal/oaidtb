23   package oaidtb.misc.pointDistributions; 
24    
25   import cern.jet.random.AbstractDistribution; 
26    
27   /** 
28    * Clase que implementa el interfaz {@link PointDistribution}, generando números 
29    * mediante la distribución uniforme, utilizando la librería de funciones matemáticas 
30    * Colt que se puede encontrar en la 
31    * 
32    * <p><a href="http://tilde-hoschek.home.cern.ch/~hoschek/colt/"> 
33    * Colt distribution homepage 
34    * </a></p> 
35    * 
36    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
37    * @version $Revision: 1.0 $ 
38    */ 
39   public class UniformDistribution extends cern.jet.random.Uniform implements PointDistribution{ 
40    
41     private double radius = 100; 
42    
43     /** @return El "radio" de la distribución */ 
44     public double getRadius(){ 
45       return radius; 
46     } 
47    
48     /** @param radius El "radio" de la distribución */ 
49     public void setRadius(double radius){ 
50       this.radius = radius; 
51     } 
52    
53     /** 
54      * Se generarán números con distribución uniforme en el intervalo 
55      * [center - radius, center + radius] 
56      * @param center 
57      */ 
58     public void setCenter(double center){ 
59       setState(center - radius, center + radius); 
60     } 
61    
62     public UniformDistribution(){ 
63       super(AbstractDistribution.makeDefaultGenerator()); 
64     } 
65   }
