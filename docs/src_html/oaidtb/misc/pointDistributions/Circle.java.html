23   package oaidtb.misc.pointDistributions; 
24    
25   import java.util.Random; 
26    
27   /** 
28    * Clase que implementa el interfaz {@link PointDistribution} y que genera números de 
29    * la siguiente forma: <PRE> 
30    * Se genera un ángulo (a) aleatoriamente, y en las dos siguientes llamadas 
31    * a nextDouble se generan los números: 
32    *    - centro + seno(a) * radio 
33    *    - centro + coseno(a) * radio 
34    * y tras esto se vuelve a elegir un ángulo aleatoriamente; así, la siguiente secuencia 
35    * de llamadas genera aleatoriamente un punto (c1, c2) en la circunferencia de centro (x,y) y radio r: 
36    *   setRadius(r) 
37    *   setCenter(x) 
38    *   c1=nextDouble() 
39    *   setCenter(y) 
40    *   c2=nextDouble() 
41    *</PRE> 
42    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
43    * @version $Revision: 1.0 $ 
44    */ 
45   public class Circle implements PointDistribution{ 
46    
47     /** El centro */ 
48     private double center; 
49     /** El radio */ 
50     private double radius = 30; 
51     /** Es la primera llamada de la secuencia? */ 
52     private boolean isX = true; 
53     /** Generador de ángulos aleatorios */ 
54     private Random random = new Random(); 
55     /** Un ángulo generado aleatoriamente */ 
56     private double currentAngle; 
57    
58     /** @return El radio de la circunferencia */ 
59     public double getRadius(){ 
60       return radius; 
61     } 
62    
63     /** @param radius El radio de la circunferencia  */ 
64     public void setRadius(double radius){ 
65       this.radius = radius; 
66     } 
67    
68     /** @param center El centro de la circunferencia */ 
69     public void setCenter(double center){ 
70       this.center = center; 
71     } 
72    
73     /** @return El siguiente número de la secuencia */ 
74     public double nextDouble(){ 
75       if(isX){ 
76         currentAngle = 360 * random.nextDouble(); 
77         isX = false; 
78         return center + radius * Math.sin(currentAngle); 
79       } 
80       isX = true; 
81       return center + radius * Math.cos(currentAngle); 
82     } 
83   }
