23   package oaidtb.misc.mediator; 
24    
25   /** 
26    * Colleague template. To be used (copy & paste) when direct Colleague inheritance isn't available 
27    * (no multiple inheritance is allowed in Java). 
28    * 
29    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
30    * @version $Revision: 1.0 $ 
31    */ 
32   public class SimpleColleagueTemplate implements Arbitrable{ 
33    
34     //-----------------------------/////////////////////////////////// 
35     //------------------ ARBITRABLE IMPLEMENTATION /////////////////// 
36     //-----------------------------/////////////////////////////////// 
37    
38     // Private field which will be used to implement the Arbitable interface. 
39     private Colleague m_Colleague = new Colleague(this); 
40    
41     public Arbitrable getColleague(){ 
42       return m_Colleague; 
43     } 
44    
45     public void addMediator(Mediator mediator){ 
46       m_Colleague.addMediator(mediator); 
47     } 
48    
49     public void removeMediator(Mediator mediator){ 
50       m_Colleague.removeMediator(mediator); 
51     } 
52    
53     public void notifyMediators(int whatChanged, Object object){ 
54       m_Colleague.notifyMediators(whatChanged, object); 
55     } 
56    
57     public int numberOfMediators(){ 
58       return m_Colleague.numberOfMediators(); 
59     } 
60   }
