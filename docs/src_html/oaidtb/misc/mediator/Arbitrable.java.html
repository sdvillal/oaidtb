23   package oaidtb.misc.mediator; 
24    
25   /** 
26    * Interfaz para las clases que hacen el papel de "colegas" usando esta implementación 
27    * del patrón de disño Mediator. 
28    * 
29    * <p><p> 
30    * Para más información sobre el patrón de diseño Mediator, visitar la página: 
31    *  <a href="http://my.execpc.com/~gopalan/design/behavioral/mediator/mediator.html"> 
32    * http://my.execpc.com/~gopalan/design/behavioral/mediator/mediator.html 
33    * </a> 
34    * 
35    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
36    * @version $Revision: 1.0 $ 
37    */ 
38   public interface Arbitrable{ 
39    
40     /** @return La identidad del objeto que será considerado como el emisor de los mensajes por los mediadores */ 
41       Arbitrable getColleague(); 
42    
43     /** @param mediator Un mediador al que se le pasarán los mensajes generados */ 
44     void addMediator(Mediator mediator); 
45    
46     /** @param mediator Un mediador al que ya no queremos seguir mandando mensajes  */ 
47     void removeMediator(Mediator mediator); 
48    
49     /** @return El número de mediadores conocidos por este objeto {@link Arbitrable} */ 
50     int numberOfMediators(); 
51    
52     /** 
53      * Notificar a todos los mediadores una acción realizada 
54      * por este colega 
55      * 
56      * @param whatChanged Identificador del cambio 
57      * @param object Un objeto que identifique la acción o ayude al mediador a cumplir su función 
58      */ 
59     void notifyMediators(final int whatChanged, final Object object); 
60   }
