23   package oaidtb.misc.mediator; 
24    
25   /** 
26    * Una implementación estándar del interfaz {@link Arbitrable} de la que pueden extender aquellas 
27    * clases que se desea implementen esta interfaz y que no necesiten extender ninguna 
28    * otra clase; si la clase necesita extender alguna otra clase y no se 
29    * desea añadir ninguna funcionalidad especial en la comunicación con 
30    * los mediadores, se puede usar (copy & paste) la clase {@link SimpleColleagueTemplate}, 
31    * que utiliza un objeto Colleague para implentar la interfaz {@link Arbitrable}. 
32    * 
33    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
34    * @version $Revision: 1.0 $ 
35    */ 
36   public class Colleague implements Arbitrable{ 
37    
38     /** El array de mediadores */ 
39     private Mediator[] m_Mediators = new Mediator[]{}; 
40     /** La identidad del objeto que se pasará como fuente del mensaje al mediador */ 
41     private Arbitrable m_ColleagueIdentity; 
42    
43     /** 
44      * Constructor 
45      * @param colleagueIdentity la identidad del objeto que se pasará como fuente del mensaje al mediador 
46      */ 
47     public Colleague(Arbitrable colleagueIdentity){ 
48       m_ColleagueIdentity = colleagueIdentity; 
49     } 
50    
51     /** 
52      * Constructor 
53      * Se pasará this como la identidad del objeto que se pasará como fuente del mensaje al mediador 
54      */ 
55     public Colleague(){ 
56       m_ColleagueIdentity = this; 
57     } 
58    
59     /** @param mediator Un mediador al que se le pasarán los mensajes generados */ 
60     public void addMediator(Mediator mediator){ 
61       Mediator[] tmpArray = new Mediator[m_Mediators.length + 1]; 
62       System.arraycopy(m_Mediators, 0, tmpArray, 0, m_Mediators.length); 
63       tmpArray[m_Mediators.length] = mediator; 
64       m_Mediators = tmpArray; 
65     } 
66    
67     /** @param mediator Un mediador al que ya no queremos seguir mandando mensajes  */ 
68     public void removeMediator(Mediator mediator){ 
69    
70       int mediatorIndex = -1; 
71       for (int i = 0; i < m_Mediators.length; i++){ 
72         if (m_Mediators[i] == mediator){ 
73           mediatorIndex = i; 
74           break; 
75         } 
76       } 
77    
78       if(mediatorIndex == -1) 
79         return; 
80    
81       Mediator[] tmpArray = new Mediator[m_Mediators.length - 1]; 
82       for (int i = 0; i < m_Mediators.length; i++){ 
83         if (m_Mediators[i] == mediator) 
84           continue; 
85         tmpArray[i] = m_Mediators[i]; 
86       } 
87       m_Mediators = tmpArray; 
88     } 
89    
90     /** 
91      * Notificar a todos los mediadores una acción realizada 
92      * por este colega. 
93      * 
94      * @param whatChanged Identificador del cambio 
95      * @param object Un objeto que identifique la acción o ayude al mediador a cumplir su función 
96      */ 
97     public void notifyMediators(int whatChanged, Object object){ 
98       for (int i = 0; i < m_Mediators.length; i++) 
99         m_Mediators[i].colleagueChanged(m_ColleagueIdentity, whatChanged, object); 
100    } 
101   
102    /** @return El número de mediadores conocidos por este objeto {@link Arbitrable} */ 
103    public int numberOfMediators(){ 
104      return m_Mediators.length; 
105    } 
106   
107    /** @return La identidad del objeto que será considerado como el emisor de los mensajes por los mediadores */ 
108    public Arbitrable getColleague(){ 
109      return m_ColleagueIdentity; 
110    } 
111  }
