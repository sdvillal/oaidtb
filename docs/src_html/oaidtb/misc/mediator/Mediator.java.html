23   package oaidtb.misc.mediator; 
24    
25   /** 
26    * Interfaz que hace de mediador entre distintos objetos {@link Arbitrable}, definiendo 
27    * la manera (formato de los mensajes) en que se comunica con ellos. 
28    * 
29    * <p><p> 
30    * Para más información sobre el patrón de diseño Mediator, visitar la página: 
31    *  <a href="http://my.execpc.com/~gopalan/design/behavioral/mediator/mediator.html"> 
32    *  http://my.execpc.com/~gopalan/design/behavioral/mediator/mediator.html 
33    * </a> 
34    * 
35    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
36    * @version $Revision: 1.0 $ 
37    */ 
38   public interface Mediator{ 
39    
40     /** 
41      * Indicar a este mediador que un objeto {@link Arbitrable} ha cambiado (ha realizado alguna 
42      * acción) para que haga las gestiones pertinentes. 
43      * 
44      * @param colleague La identidad del objeto que ha producido el mensaje 
45      * @param changeIndicator Un indicador del tiupo de acción/cambio realizado 
46      * @param object Cualquier objeto que el mediador pueda necesitar; 
47      *               también puede hacer la función de changeIndicator 
48      */ 
49     void colleagueChanged(Arbitrable colleague,final int changeIndicator, final Object object); 
50   }
