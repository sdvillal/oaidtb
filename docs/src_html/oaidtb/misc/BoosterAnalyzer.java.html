38   /** 
39    * A simple class which allows to keep tracking of a booster (& base classifiers) evolution.<PRE> 
40    * Example of use: 
41    * <code> 
42    *   ... 
43    *   Instances instances = new Instances(new FileReader("./data/conus-torus/train.arff")); 
44    *   Instances testInstances = new Instances(new FileReader("./data/conus-torus/test.arff")); 
45    *   instances.setClassIndex(instances.numAttributes() - 1); 
46    *   testInstances.setClassIndex(testInstances.numAttributes() - 1); 
47    * 
48    *   Booster booster = new oaidtb.boosters.AdaBoostECC(); 
49    *   booster.buildClassifier(instances); 
50    * 
51    *   //Create the booster analyzer 
52    *   BoosterAnalyzer ba = new BoosterAnalyzer(booster, instances, null); 
53    *   BoosterAnalyzer baTest = new BoosterAnalyzer(booster, testInstances, null); 
54    * 
55    *   //Inform about what we want keep tracking of 
56    *   ba.setSaveBoosterErrors(true); 
57    *   ba.setSaveBoosterCosts(false); 
58    *   baTest.setSaveBaseClassifiersCosts(false); 
59    *   baTest.setSaveBaseClassifiersErrors(false); 
60    * 
61    *   //Initialize the booster analyzer 
62    *   ba.initialize(); 
63    *   baTest.initialize(); 
64    * 
65    *   //Update the statistics 
66    *   ba.updateStatistics(); 
67    *   baTest.updateStatistics(); 
68    * 
69    *   //Perform a little more iterations 
70    *   booster.nextIterations(10); 
71    * 
72    *   //Update the statistics 
73    *   ba.updateStatistics(); 
74    *   baTest.updateStatistics(); 
75    * 
76    *   ... 
77    * </code> 
78    * </PRE> 
79    * 
80    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a>

172    public BoosterAnalyzer(IterativeUpdatableClassifier booster, Instances data, CostMatrix costMatrix) throws Exception{ 
173   
174      if (data == null || data.numInstances() == 0) 
175        throw new Exception("No se ha especificado el conjunto de datos sobre el que trabajar."); 
176   
177      if (booster == null) 
178        throw new Exception("No se ha especificado el booster sobre el que trabajar."); 
179   
180      m_Data = data; 
181      m_Booster = booster; 
182   
183      m_CostMatrix = costMatrix; 
184      if (isCostSensitiveBooster((Classifier) booster)) 
185        m_CostMatrixGetter = oaidtb.misc.Utils.getCostMatrixGetter((Classifier) booster); 
186    }

188    /** 
189     * Este método debe ser llamado cuando el booster ya haya sido debidamente configurado 
190     * (se hace automáticamente la primera vez que se llama a {@link #updateStatistics} si no 
191     * ha sido invocado antes). Sus cometidos son: 
192     * 
193     * <p> Inicializar la matriz de costos; si es null, se pide al booster que provea una y si éste no 
194     * es capaz se usa la matriz por defecto en la que los costes valen siempre 1 
195     * 
196     * <p> Comprobar que el clasificador realmente sea capaz de clasificar las instancias 
197     * 
198     * <p> Hacer otras inicializaciones 
199     * 
200     * @throws Exception Si el clasificador no es capaz de clasificar las instancias 
201     * @throws Exception Si se falla al invocar el método del booster que provee la matriz de costes 
202     */ 
203    public void initialize() throws Exception{ 
204   
205      if (!oaidtb.misc.Utils.classifierCanClassify((Classifier) m_Booster, m_Data.instance(0))) 
206        throw new Exception("Los datos con los que analizar no son compatibles " 
207                            + "con las instancias de entrenamiento del booster."); 
208   
209      if (m_SaveBaseClassifiersCosts || m_SaveBaseClassifiersErrors) 
210        if ((m_CostMatrix == null || m_CostMatrix.size() != m_Data.numClasses()) && m_CostMatrixGetter != null) 
211          m_CostMatrix = (CostMatrix) m_CostMatrixGetter.invoke(m_Booster, new Object[]{}); 
212   
213      if (m_CostMatrix == null) 
214  // No podemos hacer esto porque getElement(int,int) es final 
215  //      m_CostMatrix = new CostMatrix(0){ 
216  //        public double getElement(int row, int column){ 
217  //          return 0.0; 
218  //        } 
219  //      }; 
220        m_CostMatrix = new CostMatrix(m_Data.numClasses()); 
221   
222      m_BC_VotesForInstances = new double[m_Data.numInstances()][m_Data.numClasses()]; 
223   
224      if(m_Booster instanceof AdaBoostMH) 
225        m_TrainDataFormat = new Instances(((AdaBoostMH)m_Booster).getTrainData(), 0); 
226      else 
227        m_TrainDataFormat = new Instances(((Booster)m_Booster).getTrainData(), 0); 
228   
229      m_ProcessStarted = true; 
230    }

232    /** 
233     * Actualiza las estadísticas, si existen nuevas iteraciones que no han sido utilizadas 
234     * para tal fin. 
235     * 
236     * @throws Exception Si se produce algún error 
237     */ 
238    public void updateStatistics() throws Exception{ 
239      if(!m_ProcessStarted) 
240        initialize(); 
241   
242      //Es un booster simple o es un meta-booster? 
243      if (m_Booster instanceof AdaBoostMH) 
244        while (m_IterationsAnalyzedSoFar < m_Booster.getNumIterationsPerformed()){ 
245          retrieveStatisticsFromMH(m_IterationsAnalyzedSoFar); 
246          m_IterationsAnalyzedSoFar++; 
247        } 
248      else 
249        while (m_IterationsAnalyzedSoFar < m_Booster.getNumIterationsPerformed()){ 
250          retrieveStatistics(m_IterationsAnalyzedSoFar); 
251          m_IterationsAnalyzedSoFar++; 
252        } 
253      m_BiggestCost = (m_BiggestBoosterCost > m_BiggestBC_Cost ? m_BiggestBoosterCost : m_BiggestBC_Cost); 
254    }

256    /** 
257     * Recoge las estadísticas de un booster "convencional" 
258     * 
259     * @param itIndex la iteración de la que recoger las estadísticas 
260     * 
261     * @throws Exception si algo falla... 
262     */ 
263    private void retrieveStatistics(final int itIndex) throws Exception{ 
264   
265      int boosterFails = 0; 
266      double boosterCost = 0; 
267      int bcFails = 0; 
268      double bcCost = 0; 
269   
270      //Nos hacemos con el clasificador base correspondiente 
271      Classifier baseClassifier = ((Booster) m_Booster).getClassifier(itIndex); 
272   
273      for (int i = 0; i < m_Data.numInstances(); i++){ 
274   
275        Instance instance = m_Data.instance(i); 
276        int trueClassValue = (int) instance.classValue(); 
277        int classifiedAs; 
278   
279        //Calculamos los errores del booster 
280        if (m_SaveBoosterErrors || m_SaveBoosterCosts){ 
281   
282          Utils.doubleArraysSum(m_BC_VotesForInstances[i], 
283                                m_Booster.getClassifierVote(instance, itIndex)); 
284          classifiedAs = Utils.maxIndex(m_BC_VotesForInstances[i]); 
285   
286          if (classifiedAs != trueClassValue){ 
287            boosterFails++; 
288            boosterCost += m_CostMatrix.getElement(trueClassValue, classifiedAs); 
289          } 
290        } 
291   
292        //Calculamos los errores del clasificador base 
293        //Un poco ineficiente, dividirlo en dos buclse para realizar esta comprobación sólo una vez 
294        if (m_SaveBaseClassifiersErrors || m_SaveBaseClassifiersCosts){ 
295          //Los clasificadores de weka no deberían depender para clasificar de que la instancia 
296          //pertenezca a algún dataset, pero dependen (por ejemplo, J48 pregunta por el número 
297          //de clases de la instancia a clasificar...) 
298          instance = new Instance(instance); 
299          instance.setDataset(m_TrainDataFormat); 
300          classifiedAs = (int) baseClassifier.classifyInstance(instance); 
301          if (classifiedAs != trueClassValue){ 
302            bcFails++; 
303            bcCost += m_CostMatrix.getElement(trueClassValue, classifiedAs); 
304          } 
305        } 
306      } 
307      saveStatistics(itIndex, boosterFails, boosterCost, bcFails, bcCost); 
308    }

310    /** 
311     * Recoge las estadísticas de un meta-booster 
312     * 
313     * @param itIndex la iteración de la que recoger las estadísticas 
314     * 
315     * @throws Exception si algo falla... 
316     */ 
317    private void retrieveStatisticsFromMH(final int itIndex) throws Exception{ 
318   
319      int boosterFails = 0; 
320      double boosterCost = 0; 
321      int bcFails = 0; 
322      double bcCost = 0; 
323   
324      AdaBoostMH mh = (AdaBoostMH) m_Booster; 
325      //Nos hacemos con los clasificadores base 
326      Classifier[] baseClassifiers = new Classifier[mh.getNumClasses()]; 
327      for (int j = 0; j < mh.getNumClasses(); j++) 
328        baseClassifiers[j] = mh.getBooster(j).getClassifier(itIndex); 
329   
330      for (int i = 0; i < m_Data.numInstances(); i++){ 
331   
332        Instance instance = m_Data.instance(i); 
333        int trueClassValue = (int) instance.classValue(); 
334        int classifiedAs; 
335   
336        if (m_SaveBoosterErrors || m_SaveBoosterCosts){ 
337          Utils.doubleArraysSum(m_BC_VotesForInstances[i], 
338                                m_Booster.getClassifierVote(instance, itIndex)); 
339          classifiedAs = Utils.maxIndex(m_BC_VotesForInstances[i]); 
340          if (classifiedAs != trueClassValue){ 
341            boosterFails++; 
342            boosterCost += m_CostMatrix.getElement(trueClassValue, classifiedAs); 
343          } 
344        } 
345   
346        //Calculamos los errores del clasificador base 
347        //Consideramos el número de fallos total del clasificador base 
348        //como la suma de los errores de cada CB en la iteración 
349        //clasificando la clase correspondiente 
350        if (m_SaveBaseClassifiersErrors || m_SaveBaseClassifiersCosts){ 
351   
352          //PEAZO DE FALLO, 2 HORAS DE DEPURADO POR GAÑÁN 
353          //instance.setDataset(trainDataFormat); 
354          //Los clasificadores de weka no deberían depender para clasificar de que la instancia 
355          //pertenezca a algún dataset, pero dependen (por ejemplo, J48 pregunta por el número 
356          //de clases de la instancia a clasificar...) 
357          instance = new Instance(instance); 
358          instance.setDataset(m_TrainDataFormat); 
359          classifiedAs = (int) baseClassifiers[trueClassValue].classifyInstance(instance); 
360          if (classifiedAs != 1){ 
361            bcFails++; 
362            bcCost += m_CostMatrix.getElement(trueClassValue, classifiedAs); 
363          } 
364        } 
365      } 
366      saveStatistics(itIndex, boosterFails, boosterCost, bcFails, bcCost); 
367    }

369    /** 
370     * Salva las estadísticas en los correspondientes arrays 
371     * 
372     * @param itIndex El número de iteración 
373     * @param boosterFails El número de errores del booster 
374     * @param boosterCost El coste de los errores del booster 
375     * @param bcFails El número de fallos del clasifixador base 
376     * @param bcCost El coste de los errores del clasificador base 
377     */ 
378    private void saveStatistics(final int itIndex, 
379                                final int boosterFails, 
380                                final double boosterCost, 
381                                final int bcFails, 
382                                final double bcCost){ 
383      //Booster error 
384      if (m_SaveBoosterErrors){ 
385        double error = (double) boosterFails / (double) m_Data.numInstances(); 
386        m_BoosterErrors.add(itIndex, new SimpleDouble(error)); 
387        if (m_BiggestBoosterError < error){ 
388          m_BiggestBoosterError = error; 
389          m_BiggestBoosterErrorAt = itIndex; 
390        } 
391        if (m_SmallestBoosterError > error){ 
392          m_SmallestBoosterError = error; 
393          m_SmallestBoosterErrorAt = itIndex; 
394        } 
395      } 
396   
397      //Booster cost 
398      if (m_SaveBoosterCosts){ 
399        m_BoosterCosts.add(itIndex, new SimpleDouble(boosterCost)); 
400        if (m_BiggestBoosterCost < boosterCost){ 
401          m_BiggestBoosterCost = boosterCost; 
402          m_BiggestBoosterCostAt = itIndex; 
403        } 
404        if (m_SmallestBoosterCost > boosterCost){ 
405          m_SmallestBoosterCost = boosterCost; 
406          m_SmallestBoosterCostAt = itIndex; 
407        } 
408      } 
409   
410      //Base classifier error 
411      if (m_SaveBaseClassifiersErrors){ 
412        double error = (double) bcFails / (double) m_Data.numInstances(); 
413        m_BC_Errors.add(itIndex, new SimpleDouble(error)); 
414        if (m_BiggestBC_Error < error){ 
415          m_BiggestBC_Error = error; 
416          m_BiggestBC_ErrorAt = itIndex; 
417        } 
418        if (m_SmallestBC_Error > error){ 
419          m_SmallestBC_Error = error; 
420          m_SmallestBC_ErrorAt = itIndex; 
421        } 
422      } 
423   
424      //Base classifier cost 
425      if (m_SaveBaseClassifiersCosts){ 
426        m_BC_Costs.add(itIndex, new SimpleDouble(bcCost)); 
427        if (m_BiggestBC_Cost < bcCost){ 
428          m_BiggestBC_Cost = bcCost; 
429          m_BiggestBC_CostAt = itIndex; 
430        } 
431        if (m_SmallestBC_Cost > bcCost){ 
432          m_SmallestBC_Cost = bcCost; 
433          m_SmallestBC_CostAt = itIndex; 
434        } 
435      } 
436    }

438    /** 
439     * Método ad-hoc para conocer rápidamente (sin tener que hacer introspección) si 
440     * un booster puede proporcionar una matriz de costes; de momento, se emplea 
441     * la norma de que esto es así si y sólo si pertenece al paquete

445     * @return true si el booster pertenece a {@link oaidtb.boosters.costSensitive}

447    private boolean isCostSensitiveBooster(Classifier booster){ 
448      if (booster instanceof Booster){ 
449        String name = booster.getClass().getName(); 
450        if (name.startsWith("oaidtb.boosters.costSensitive.")){ 
451          return true; 
452        } 
453      } 
454      return false; 
455    }

655    /** @return una descripción en modo texto, CSV (Comma Separated Values) de las estadísticas  */ 
656    public String toCSV(){ 
657      StringBuffer sb = new StringBuffer(); 
658      ArrayList[] collectedStats = new ArrayList[4]; 
659      int numCollected = 0; 
660      sb.append(";;;FORMAT: itIndex"); 
661      if(m_SaveBoosterErrors){ 
662        collectedStats[numCollected] = m_BoosterErrors; 
663        numCollected++; 
664        sb.append(",booster error"); 
665      } 
666      if(m_SaveBoosterCosts){ 
667        collectedStats[numCollected] = m_BoosterCosts; 
668        numCollected++; 
669        sb.append(",booster cost"); 
670      } 
671      if(m_SaveBaseClassifiersErrors){ 
672        collectedStats[numCollected] = m_BC_Errors; 
673        numCollected++; 
674        sb.append(",base classifier error"); 
675      } 
676      if(m_SaveBaseClassifiersCosts){ 
677        collectedStats[numCollected] = m_BC_Costs; 
678        numCollected++; 
679        sb.append(",base classifier cost"); 
680      } 
681      sb.append("\n\n"); 
682   
683      for(int i=0; i < m_IterationsAnalyzedSoFar - 1; i++){ 
684        sb.append(i +","); 
685        for(int j=0; j < numCollected - 1; j++) 
686          sb.append(collectedStats[j].get(i) +","); 
687        sb.append(collectedStats[numCollected - 1].get(i) +"\n"); 
688      } 
689   
690      sb.append(m_IterationsAnalyzedSoFar - 1 +","); 
691      for(int j=0; j < numCollected - 1; j++) 
692        sb.append(collectedStats[j].get(m_IterationsAnalyzedSoFar - 1) +","); 
693      sb.append(collectedStats[numCollected - 1].get(m_IterationsAnalyzedSoFar - 1)); 
694   
695      return sb.toString(); 
696    }
