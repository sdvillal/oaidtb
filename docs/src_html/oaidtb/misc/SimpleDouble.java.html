23   package oaidtb.misc; 
24    
25   /** 
26    * Clase que encapsula un tipo primitivo double en un objeto 
27    * con mayor eficiencia en términos de memoria que la clase Integer. 
28    * 
29    * <a href="http://www.javaworld.com/javaworld/javatips/jw-javatip130.html"> 
30    * Vladimir Roubtsov. <i>Do you know your data size?</i>. 
31    * JavaWorld tips and tricks nº 130 
32    * </a><p> 
33    * 
34    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
35    * @version $Revision: 1.0 $ 
36    */ 
37   public class SimpleDouble extends Number{ 
38    
39     //No lo hacemos final para poder hacer cosas como value++ 
40     public double value; 
41    
42     public SimpleDouble(final double value){ 
43       this.value = value; 
44     } 
45    
46     public boolean equals(Object obj){ 
47       if (obj instanceof SimpleDouble) 
48         return value == ((SimpleDouble) obj).value; 
49       return false; 
50     } 
51    
52     public int hashCode() { 
53       return new Double(value).hashCode(); 
54     } 
55    
56     public String toString() { 
57       return String.valueOf(value); 
58     } 
59    
60     /** 
61      * Returns the value of the specified number as a <code>double</code>. 
62      * This may involve rounding. 
63      * 
64      * @return  the numeric value represented by this object after conversion 
65      *          to type <code>double</code>. 
66      */ 
67     public double doubleValue(){ 
68       return value; 
69     } 
70    
71     /** 
72      * Returns the value of the specified number as a <code>float</code>. 
73      * This may involve rounding. 
74      * 
75      * @return  the numeric value represented by this object after conversion 
76      *          to type <code>float</code>. 
77      */ 
78     public float floatValue(){ 
79       return (float) value; 
80     } 
81    
82     /** 
83      * Returns the value of the specified number as an <code>int</code>. 
84      * This may involve rounding. 
85      * 
86      * @return  the numeric value represented by this object after conversion 
87      *          to type <code>int</code>. 
88      */ 
89     public int intValue(){ 
90       return (int) value; 
91     } 
92    
93     /** 
94      * Returns the value of the specified number as a <code>long</code>. 
95      * This may involve rounding. 
96      * 
97      * @return  the numeric value represented by this object after conversion 
98      *          to type <code>long</code>. 
99      */ 
100    public long longValue(){ 
101      return (long) value; 
102    } 
103  }
