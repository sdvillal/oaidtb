23   package oaidtb.misc.guiUtils; 
24    
25   import javax.swing.text.AttributeSet; 
26   import javax.swing.text.BadLocationException; 
27   import javax.swing.text.PlainDocument; 
28   import java.awt.*; 
29    
30   /** 
31    * Clase que, añadida a un componente de texto con el método setDocument() hace que 
32    * no se permita la inserción de texto que no represente a un entero 
33    * comprendido entre los valores máximo y mínimo especificados 
34    * (por defecto, permite la inclusión de todos los enteros); por supuesto, también 
35    * permite la inserción del símbolo menos. 
36    * 
37    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
38    * @version $Revision: 1.0 $ 
39    */ 
40   public class IntegerDocument extends PlainDocument{ 
41    
42     private int maxValue = Integer.MAX_VALUE; 
43     private int minValue = Integer.MIN_VALUE; 
44    
45     public int getMaxValue(){ 
46       return maxValue; 
47     } 
48    
49     public void setMaxValue(int maxValue){ 
50       this.maxValue = maxValue; 
51     } 
52    
53     public int getMinValue(){ 
54       return minValue; 
55     } 
56    
57     public void setMinValue(int minValue){ 
58       this.minValue = minValue; 
59     } 
60    
61     public void remove(int offs, int len) throws BadLocationException{ 
62       String tmp = getText(0, getLength()).substring(offs, offs + len); 
63       super.remove(offs, len); 
64       String text = getText(0, getLength()); 
65       if (text.length() == 0) 
66         insertString(offs, tmp, null); 
67       else{ 
68         int intValue = Integer.parseInt(text); 
69         if (intValue < minValue || intValue > maxValue) 
70           insertString(offs, tmp, null); 
71       } 
72     } 
73    
74     public void insertString(int offset, String string, AttributeSet attributes) throws BadLocationException{ 
75    
76       if (string == null || "-".equals(string) && minValue >= 0){ 
77         Toolkit.getDefaultToolkit().beep(); 
78         return; 
79       } 
80    
81       int length = getLength(); 
82    
83       String newValue; 
84    
85       if (length == 0){ 
86         if ("-".equals(string)){ 
87           super.insertString(offset, string, attributes); 
88           return; 
89         } 
90         newValue = string; 
91       } 
92       else{ 
93         String currentContent = getText(0, length); 
94         StringBuffer currentBuffer = new StringBuffer(currentContent); 
95         currentBuffer.insert(offset, string); 
96         newValue = currentBuffer.toString(); 
97       } 
98    
99       if(newValue.length() == 0) 
100        return; 
101   
102      try{ 
103        int intValue = Integer.parseInt(newValue); 
104        //TODO: Cambiarlo, pues bajo ciertas circunstancias puede producir q el documento no pueda ser editado 
105        if (intValue > maxValue || intValue < minValue) 
106          throw new NumberFormatException("Invalid value: limits violated."); 
107        super.insertString(offset, string, attributes); 
108      } 
109      catch (NumberFormatException exception){ 
110        Toolkit.getDefaultToolkit().beep(); 
111      } 
112    } 
113  }
