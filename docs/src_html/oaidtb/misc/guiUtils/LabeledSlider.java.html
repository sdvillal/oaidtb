23   package oaidtb.misc.guiUtils; 
24    
25   import javax.swing.*; 
26   import javax.swing.event.ChangeListener; 
27   import javax.swing.event.ChangeEvent; 
28   import java.awt.*; 
29    
30   /** 
31    * Clase que soluciona el bug de guiUtils 4220108 (un JSlider no se pinta cuando se inserta 
32    * en un JDesktopPane...) y que permite tener un JLabel que se actualiza según los valores 
33    * de un JSlider. 
34    * 
35    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
36    * @version $Revision: 1.0 $ 
37    */ 
38   public class LabeledSlider{ 
39    
40     private JSlider slider = new JSlider(){ 
41       boolean primeraVezQueSePinta = true; 
42    
43       //Solución cutre para el bug 4220108 
44       //Ver http://developer.java.sun.com/developer/bugParade/bugs/4220108.html 
45       //A ver si reabren el tema... 
46       public void paint(Graphics g){ 
47         super.paint(g); 
48         if(primeraVezQueSePinta){ 
49           updateUI(); 
50           primeraVezQueSePinta = false; 
51         } 
52       } 
53     }; 
54     private JLabel label = new JLabel(""); 
55     private String labelMessage = ""; 
56    
57     public JLabel getLabel(){ 
58       return label; 
59     } 
60    
61     public JSlider getSlider(){ 
62       return slider; 
63     } 
64    
65     public LabeledSlider(String messagePrefix){ 
66       if(messagePrefix != null) 
67         labelMessage = messagePrefix; 
68    
69       slider.addChangeListener(new ChangeListener(){ 
70         public void stateChanged(ChangeEvent e){ 
71           label.setText(labelMessage +String.valueOf(slider.getValue())); 
72         } 
73       }); 
74     } 
75   }
