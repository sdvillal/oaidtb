23   package oaidtb.misc.guiUtils; 
24    
25   import javax.swing.*; 
26   import java.beans.PropertyChangeEvent; 
27   import java.beans.PropertyChangeListener; 
28    
29   /** 
30    * Clase que encapsula una {@link javax.swing.AbstractAction}, añadiéndola funcionalidades tales 
31    * como facilidad de acceso a la tecla de acceso rápido, al tooltip, 
32    * al icono y al nombre, y la posibilidad de especificar componentes 
33    * que serán habilitados o deshabilitados según la acción sea 
34    * habilitada o deshabilitada 
35    * 
36    * @author <a href="mailto:sdvb@wanadoo.es">Santiago David Villalba Bartolomé</a> 
37    * @version $Revision: 1.0 $ 
38    */ 
39   public abstract class MyAbstractAction extends AbstractAction{ 
40    
41     /** El listener que centralizará la habilitación de los componentes asociados */ 
42     private ActionComponentsAdapter m_EnabeDisableList = null; 
43    
44     public void setAcceleratorKey(KeyStroke keyStroke){ 
45       putValue(Action.ACCELERATOR_KEY, keyStroke); 
46     } 
47    
48     public KeyStroke getAcceleratorKey(){ 
49       return (KeyStroke) getValue(Action.ACCELERATOR_KEY); 
50     } 
51    
52     public void setToolTipText(String toolTip){ 
53       KeyStroke keyStroke = getAcceleratorKey(); 
54       putValue(Action.SHORT_DESCRIPTION, new String(toolTip +(keyStroke != null ? keyStroke.toString() : ""))); 
55     } 
56    
57     public String getTip(){ 
58       return (String) getValue(Action.SHORT_DESCRIPTION); 
59     } 
60    
61     public void setIcon(Icon icon){ 
62       putValue(Action.SMALL_ICON, icon); 
63     } 
64    
65     public Icon getIcon(){ 
66       return (Icon) getValue(Action.SMALL_ICON); 
67     } 
68    
69     public void setName(String name){ 
70       putValue(Action.NAME, name); 
71     } 
72    
73     public String getName(){ 
74       return (String) getValue(Action.NAME); 
75     } 
76    
77     public MyAbstractAction(){} 
78    
79     public MyAbstractAction(String name){ 
80       super(name); 
81     } 
82    
83     public MyAbstractAction(String name, Icon icon){ 
84       super(name, icon); 
85     } 
86    
87     /** 
88      * Añadir un nuevo componente a la lista de componentes cuyo estado de habilitación 
89      * depende de que esta acción esté o no habilitada 
90      * 
91      * @param component Un componente cuyo estado de (des)habilitación será el de la acción 
92      */ 
93     public void addStateDependantComponent(JComponent component){ 
94       if(m_EnabeDisableList == null){ 
95         m_EnabeDisableList = new ActionComponentsAdapter(component); 
96         addPropertyChangeListener(m_EnabeDisableList); 
97       } 
98       else 
99         m_EnabeDisableList.addComponent(component); 
100    } 
101   
102    /** 
103     * Clase que sirve para actualizar los componentes ligados a una acción; 
104     * se puede extender para manejar otras propiedades no estándar de 
105     * todos los componentes (por ejemplo, el texto o el icono de un JButton). 
106     */ 
107    public static class ActionComponentsAdapter implements PropertyChangeListener{ 
108   
109      /** El array de componentes */ 
110      private JComponent[] components; 
111   
112      /** 
113       * Constructor 
114       * 
115       * @param firstComponent El primer componente en la lista 
116       */ 
117      public ActionComponentsAdapter(JComponent firstComponent){ 
118        components = new JComponent[1]; 
119        components[0] = firstComponent; 
120      } 
121   
122      /** 
123       * Añade un nuevo componente a la lista 
124       * 
125       * @param component El componente 
126       */ 
127      public void addComponent(JComponent component){ 
128        JComponent[] tmp = new JComponent[components.length + 1]; 
129        System.arraycopy(components, 0, tmp, 0, components.length); 
130        tmp[components.length] = component; 
131        components = tmp; 
132      } 
133   
134      /** 
135       * Cada vez que la {@link javax.swing.Action} cambie su estado, se cambiará 
136       * también el de los componentes asociados 
137       * 
138       * @param e El evento 
139       */ 
140      public void propertyChange(PropertyChangeEvent e){ 
141        String propertyName = e.getPropertyName(); 
142   
143        if (propertyName.equals("enabled")){ 
144          Boolean enabledState = (Boolean) e.getNewValue(); 
145          for(int i=0; i<components.length;i++) 
146            components[i].setEnabled(enabledState.booleanValue()); 
147        }

149  //        if (propertyName.equals(Action.SHORT_DESCRIPTION)){ 
150  //          String tip = (String) e.getNewValue(); 
151  //          for(int i=0; i<components.length;i++) 
152  //            components[i].setToolTipText(tip); 
153  //        } 
154  //      } 
155      } 
156    }
